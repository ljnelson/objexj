029    
030    import java.util.Arrays;
031    import java.util.List;
032    import java.util.Map;
033    
034    import org.junit.Test;
035    
036    import com.edugility.objexj.engine.Thread;
037    
038    import static org.junit.Assert.assertEquals;
039    import static org.junit.Assert.assertNotNull;
040    import static org.junit.Assert.assertTrue;
041    
042    public class TestCaseProgram4 extends AbstractProgramTestCase<Character> {
043    
044      public TestCaseProgram4() {
045        super();
046      }
047    
048      @Test
049      public void testVariableAssignmentInThePresenceOfMultipleThreads() {
050        final Thread<Character> t = this.run(Arrays.asList('a', 'a', 'b'));
051        assertMatch(t);
052        final Map<Object, Object> variables = t.getVariables();
053        assertNotNull(variables);
054        assertEquals(1, variables.size());
055        assertEquals("bar", variables.get("foo"));
056      }
057    
058    }
