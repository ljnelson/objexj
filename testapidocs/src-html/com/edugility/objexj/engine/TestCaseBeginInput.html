001    package com.edugility.objexj.engine;
002    
003    import java.util.Arrays;
004    import java.util.Collections;
005    import java.util.List;
006    
007    import org.junit.Before;
008    import org.junit.Test;
009    
010    import static org.junit.Assert.*;
011    
012    public class TestCaseBeginInput extends BasicThreadScheduler<Character> {
013    
014      public TestCaseBeginInput() {
015        super();
016      }
017    
018      @Test
019      public void testAtStartWithNullInput() {
020    
021        /*
022         * Set up a (bogus, virtually nonsensical) program that runs two
023         * BeginInput instructions, which in normal circumstances would be
024         * stupid.  The second one should fail, obviously, as the program
025         * counter will no longer be at the beginning.
026         *
027         * This program will have no input (which is legal).
028         */
029    
030        final Program<Character> p = new Program<Character>();
031        p.add(new BeginInput<Character>());
032        p.add(new BeginInput<Character>());
033        assertNotNull(p);
034    
035        final Thread<Character> t = this.newThread("T0", new ProgramCounter<Character>(p), null /* no input */, Thread.VALID_NO_INPUT_POINTER, null, null);
036        assertNotNull(t);
037    
038        t.step();
039        assertEquals(Thread.State.VIABLE, t.getState());
040    
041        t.step();
042        assertEquals(Thread.State.DEAD, t.getState());
043      }  
044    
045      @Test
046      public void testAtStartWithEmptyInput() {

048        /*

050         * BeginInput instructions, which in normal circumstances would be
051         * stupid.  The second one should fail, obviously, as the program
052         * counter will no longer be at the beginning.
