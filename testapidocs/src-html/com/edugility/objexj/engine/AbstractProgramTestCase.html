001    package com.edugility.objexj.engine;
002    
003    import java.io.BufferedReader;
004    import java.io.InputStreamReader;
005    import java.io.IOException;
006    
007    import java.util.Map;
008    
009    import java.util.concurrent.atomic.AtomicInteger;
010    
011    import java.lang.reflect.InvocationTargetException;
012    
013    import java.net.URL;
014    
015    import java.util.ArrayList;
016    import java.util.Collection;
017    import java.util.Iterator;
018    import java.util.List;
019    
020    import java.util.logging.Level;
021    import java.util.logging.Logger;
022    
023    import org.junit.Before;
024    import org.junit.Test;
025    
026    import static org.junit.Assert.*;
027    
028    public abstract class AbstractProgramTestCase<T> implements ThreadScheduler<T> {
029    
030      private static final AtomicInteger ids = new AtomicInteger();
031    
032      protected Program<T> program;
033    
034      protected Collection<Thread<T>> threads;
035    
036      protected AbstractProgramTestCase() {
037        super();
038      }
039    
040      @Before
041      public void before() throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
042        this.threads = new ArrayList<Thread<T>>();
043        this.program = this.getProgram();
044      }
045    
046      public Program<T> getProgram() throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
047        return this.getProgram(String.format("%s.rgx", this.getClass().getSimpleName()));
048      }
049    
050      public Program<T> getProgram(final String resourceName) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
051        assertNotNull(resourceName);
052        final URL program = java.lang.Thread.currentThread().getContextClassLoader().getResource(resourceName);
053        assertNotNull(program);
054        final BufferedReader reader = new BufferedReader(new InputStreamReader(program.openStream()));
055        Program<T> p;
056        try {
057          p = Program.valueOf(reader);
058        } finally {
059          try {
060            reader.close();
061          } catch (final IOException nothingToBeDone) {
062            final Logger logger = Logger.getLogger(this.getClass().getName());
063            if (logger != null && logger.isLoggable(Level.SEVERE)) {
064              logger.logp(Level.SEVERE, this.getClass().getName(), "getProgram", "Exception encountered while trying to close reader", nothingToBeDone);
065            }
066          }
067        }
068        assertNotNull(p);
069        p.setName(resourceName);
070        return p;
071      }
072    
073      @Override
074      public Thread<T> newThread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables) {
075        return new Thread<T>(id, programCounter, items, itemPointer, captureGroups, variables, this);
076      }
077    
078      @Override
079      public boolean schedule(final Thread<T> thread) {
080        assertNotNull(thread);
081        assertNotNull(this.threads);
082        return this.threads.add(thread);
083      }
084    
085      public static final void assertDead(final Thread<?> t) {
086        assertNotNull(t);
087        assertSame(Thread.State.DEAD, t.getState());
088      }
089    
090      public static final void assertMatch(final Thread<?> t) {
091        assertNotNull(t);
092        assertSame(Thread.State.MATCH, t.getState());
093      }
094    
095      public Thread<T> getThread(final List<T> input) {
096        assertNotNull(this.program);    
097        final int inputPointer;
098        if (input == null || input.isEmpty()) {
099          inputPointer = Thread.VALID_NO_INPUT_POINTER;
100        } else {
101          inputPointer = 0;
102        }
103        final Thread<T> t = this.newThread(String.format("T%d", this.ids.getAndIncrement()), new ProgramCounter<T>(this.program), input, inputPointer, null, null);
104        assertNotNull(t);
105        return t;
106      }
107    
108      public Thread<T> run(final List<T> input) {
109        assertNotNull(this.threads);
110        assertTrue(this.threads.isEmpty());
111        Thread<T> thread = this.getThread(input);
112        assertNotNull(thread);
113        this.threads.add(thread);
114        while (!this.threads.isEmpty()) {
115          final Iterator<Thread<T>> iterator = this.threads.iterator();
116          assertNotNull(iterator);
117          assertTrue(iterator.hasNext());
118          thread = iterator.next();
119          assertNotNull(thread);
120          iterator.remove();
121          thread.run();
122          if (Thread.State.MATCH == thread.getState()) {
123            // we have a match
124            break;
125          }
126        }
127        return thread;
128      }
129    
130    }
