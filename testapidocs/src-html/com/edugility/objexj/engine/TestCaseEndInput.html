001    package com.edugility.objexj.engine;
002    
003    import java.util.Arrays;
004    import java.util.List;
005    
006    import org.junit.Before;
007    import org.junit.Test;
008    
009    import static org.junit.Assert.*;
010    
011    public class TestCaseEndInput extends BasicThreadScheduler<Character> {
012    
013      public TestCaseEndInput() {
014        super();
015      }
016    
017      @Test
018      public void testAtEnd() {
019    
020        /*
021         * Set up a (bogus, virtually nonsensical) program that runs two
022         * EndInput instructions, which in normal circumstances would be
023         * stupid.  The second one should fail, obviously, as the program
024         * counter will no longer be at the end.
025         *
026         * This program will have no input (which is legal).
027         */
028    
029        final Program<Character> p = new Program<Character>();
030        p.add(new EndInput<Character>());
031        p.add(new EndInput<Character>());
032        assertNotNull(p);
033    
034        final Thread<Character> t = this.newThread(0, new ProgramCounter<Character>(p), null /* no input */, Thread.VALID_NO_INPUT_POINTER, null, null);
035        assertNotNull(t);
036    
037        t.step();
038        assertEquals(Thread.State.VIABLE, t.getState());
039    
040        t.step();
041        assertEquals(Thread.State.DEAD, t.getState());
042      }  
043    
044    }
