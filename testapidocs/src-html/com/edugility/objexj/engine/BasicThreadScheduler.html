028    package com.edugility.objexj.engine;
029    
030    import java.util.Collection;
031    import java.util.List;
032    import java.util.Map;
033    
034    import com.edugility.objexj.engine.Thread;
035    
036    /**
037     * A trivial implementation of the {@link ThreadScheduler} interface.
038     * This implementation is capable of creating new {@link Thread}s, but
039     * does not ever actually schedule them.
040     *
041     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
042     */
043    public class BasicThreadScheduler<T> implements ThreadScheduler<T> {
044    
045      /**
046       * Creates a new {@link BasicThreadScheduler}.
047       */
048      public BasicThreadScheduler() {
049        super();
050      }
051    
052      /**
053       * Creates a new {@link Thread}.
054       *
055       * @param id the identifier for the new {@link Thread}; may be
056       * {@code null} and is used only for its {@link Object#toString()}
057       * value
058       *
059       * @param programCounter the {@link ProgramCounter} to initialize
060       * the new {@link Thread} with; must not be {@code null}
061       *
062       * @param items the {@link List} of items the new {@link Thread}
063       * will {@linkplain Thread#read() read from}; may be {@code null} or
064       * {@linkplain Collection#isEmpty() empty}
065       *
066       * @param itemPointer an {@code int} indicating which item from the
067       * supplied {@code items} {@link List} to start reading from; may be
068       * a positive {@code int} or equal to {@link
069       * Thread#VALID_NO_INPUT_POINTER}
070       *
071       * @exception IllegalArgumentException if a parameter that cannot be
072       * {@code null} is supplied with a {@code null} value
073       */
074      @Override
075      public Thread<T> newThread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables) {
076        return new Thread<T>(id, programCounter, items, itemPointer, captureGroups, variables, this);
077      }
078    
079      /**
080       * Implements the {@link ThreadScheduler#schedule(Thread)} contract
081       * by returning {@code true}, but otherwise taking no action.
082       * Obviously this default implementation is suitable only for
083       * inconsequential things like certain unit tests where scheduling
084       * itself is not a concern.  Otherwise this method should be
085       * overridden.
086       *
087       * @param thread the {@link Thread} to schedule; ignored
088       *
089       * @return {@code true} if invoked
090       */
091      @Override
092      public boolean schedule(final Thread<T> thread) {
093        return true;
094      }
095      
096    }
