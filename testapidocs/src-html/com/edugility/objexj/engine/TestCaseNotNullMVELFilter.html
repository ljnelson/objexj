011    import org.mvel2.CompileException;

013    import static org.junit.Assert.*;
014    
015    public class TestCaseNotNullMVELFilter extends BasicThreadScheduler<Character> {
016    
017      public TestCaseNotNullMVELFilter() {
018        super();
019      }
020    
021      @Test(expected = CompileException.class)
022      public void testBadMVEL() throws Throwable {
023        final Map<Object, Object> variables = new HashMap<Object, Object>(3);
024        new MVELFilter<Character>("arglebargle + 1").accept(Character.valueOf('a'), variables);
025      }
026      
027      @Test
028      public void testMVELFilter() {
029        final MVELFilter<Character> filter = new MVELFilter<Character>("x = 'foo'; this != null && charValue() == 'a'");
030        final Map<Object, Object> variables = new HashMap<Object, Object>(3);
031        assertTrue(filter.accept(Character.valueOf('a'), variables));
032        assertTrue(variables.containsKey("x"));
033        variables.clear();    
034        assertFalse(filter.accept(Character.valueOf('b'), variables));
035        assertTrue(variables.isEmpty());
036        assertFalse(filter.accept((Character)null, variables));
037      }
038    
039    }
