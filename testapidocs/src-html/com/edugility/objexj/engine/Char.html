028    package com.edugility.objexj.engine;
029    
030    public class Char extends Filter<Character> {
031      
032      private static final long serialVersionUID = 1L;
033    
034      private final char c;
035    
036      public Char(final String operands) {
037        super();
038        if (operands == null) {
039          throw new IllegalArgumentException("operands", new NullPointerException("operands"));
040        } else if (operands.isEmpty()) {
041          throw new IllegalArgumentException("operands.isEmpty()");
042        }
043        this.c = operands.charAt(0);
044      }
045    
046      public Char(final char x) {
047        super();
048        this.c = x;
049      }
050    
051      @Override
052      public boolean accept(final InstructionContext<Character> context) {
053        boolean returnValue = false;
054        if (context != null && context.canRead()) {
055          final Character x = context.read();
056          returnValue = x != null && x.charValue() == this.c;
057        }
058        return returnValue;
059      }
060    
061      @Override
062      public String toString() {
063        return new StringBuilder(super.toString()).append(" ").append(this.c).toString();
064      }
065    
066    }
