314      @Test
315      public void testPrime() throws IOException {
316        build("(java.lang.Character(charValue() == 'f'))/java.lang.Integer");
317          
318        assertNextIs(BEGIN_ATOM);
319        assertNextIs(FILTER);
320        assertNextIs(ZERO_OR_MORE);
321        assertNextIs(CATENATION);
322        assertNextIs(START_SAVING);
323        assertNextIs(CATENATION);
324        assertNextIsFilter(Character.class.getName(), "charValue() == 'f'");
325        assertNextIs(CATENATION);
326        assertNextIs(STOP_SAVING);
327        assertNextIs(CATENATION);
328        assertNextIsFilter(Integer.class.getName());
329        assertNextIs(CATENATION);
330        
331        assertNoMoreTokens();
332      }

336       * Build methods.

339    
340      private final void build(final String input) throws IOException {
341        assertNotNull(input);
342        this.build(new PushbackReader(new StringReader(input)));
343      }
344    
345      private final void build(final PushbackReader pbr) throws IOException {
346        assertNotNull(pbr);
347        this.pbr = pbr;
348        this.pft = new PostfixTokenizer(pbr);
349      }
350    
351    
352      /*
353       * Assert methods.
354       */
355    
356    
357      private final void assertNoMoreTokens() {
358        assertFalse(this.pft.hasNext());
359      }
360    
361      private final void assertNextIs(final Token.Type type) {
362        assertNextIs(this.pft, type);
363      }
364    
365      private final void assertNextIsFilter() {
366        assertNextIsFilter(this.pft, null, null);
367      }
368    
369      private final void assertNextIsFilter(final String filterType) {
370        assertNextIsFilter(this.pft, filterType, null);
371      }
372    
373      private static final void assertNextIs(final PostfixTokenizer pft, final Token.Type type) {
374        assertNotNull(type);
375        assertTrue(pft.hasNext());
376        final Token token = pft.next();
377        assertNotNull(token);
378        assertSame(type, token.getType());
379      }
380    
381      private final void assertNextIsFilter(final String filterType, final String mvel) {
382        assertNextIsFilter(this.pft, filterType, mvel);
383      }
384    
385      private static final void assertNextIsFilter(final PostfixTokenizer pft) {
386        assertNextIsFilter(pft, null, null);
387      }
388    
389      private static final void assertNextIsFilter(final PostfixTokenizer pft, final String filterType) {
390        assertNextIsFilter(pft, filterType, null);
391      }
392    
393      private static final void assertNextIsFilter(final PostfixTokenizer pft, final String filterType, final String mvel) {
394        assertTrue(pft.hasNext());
395        final Token token = pft.next();
396        assertNotNull(token);
397        assertSame(FILTER, token.getType());
398        assertEquals(filterType, token.getFilterType());
399        assertEquals(mvel, token.getValue());
400      }
