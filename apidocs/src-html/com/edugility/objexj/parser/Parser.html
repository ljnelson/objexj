055    public class Parser {
056    
057      private static final class State<T> implements Iterator<Token> {
058    
059        private int groupIndex;
060        
061        private Token token;
062    
063        private int position;
064    
065        private final Deque<Program<T>> stack;
066    
067        private final PostfixTokenizer tokenizer;
068    
069        private State(final PostfixTokenizer tokenizer) {
070          super();
071          if (tokenizer == null) {
072            throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
073          }
074          this.position = -1;
075          this.tokenizer = tokenizer;
076          this.stack = new ArrayDeque<Program<T>>();
077          this.groupIndex = 1;
078        }
079    
080        public final Token getToken() {
081          return this.token;
082        }
083    
084        @Override
085        public final boolean hasNext() {
086          return this.tokenizer.hasNext();
087        }
088    
089        @Override
090        public final Token next() {
091          this.token = this.tokenizer.next();
092          this.position++;
093          return this.token;
094        }
095    
096        @Override
097        public final void remove() {
098          this.tokenizer.remove();
099        }
100    
101        public final int stackSize() {
102          return this.stack.size();
103        }
104    
105        public void push(final Program<T> programFragment) {
106          if (programFragment == null) {
107            throw new IllegalArgumentException("programFragment", new NullPointerException("programFragment"));
108          }
109          this.stack.push(programFragment);
110        }
111    
112        public Program<T> peek() {
113          return this.stack.peek();
114        }
115    
116        public Program<T> pop() {
117          return this.stack.pop();
118        }
119    
120        @Override
121        public String toString() {
122          final Formatter f = new Formatter(new StringBuilder());
123          f.format("Parsing state:%n");
124          f.format("Current token: %s%n", this.getToken());
125          f.format("Stack size: %d%n", this.stackSize());
126          f.format("Stack: %s%n", this.stack);
127          return f.toString();
128        }
129        
130      }
131    
132      public Parser() {
133        super();
134      }
135    
136      public <T> Program<T> parse(final String input) throws IOException {
137        if (input == null) {
138          throw new IllegalArgumentException("input", new NullPointerException("input"));
139        }
140        final StringReader sr = new StringReader(input);
141        final PushbackReader reader = new PushbackReader(sr) {
142            @Override
143            public String toString() {
144              return sr.toString();
145            }
146          };
147        try {
148          return this.parse(new State<T>(new PostfixTokenizer(reader)));
149        } finally {
150          try {
151            reader.close();
152          } catch (final IOException logMe) {
153            final Logger logger = Logger.getLogger(this.getClass().getName());
154            assert logger != null;
155            if (logger.isLoggable(Level.SEVERE)) {
156              logger.logp(Level.SEVERE, this.getClass().getName(), "parse", "Error closing reader", logMe);
157            }
158          }
159        }
160      }
161    
162      public <T> Program<T> parse(final PostfixTokenizer tokenizer) {
163        if (tokenizer == null) {
164          throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
165        }
166        return this.parse(new State<T>(tokenizer));
167      }
168    
169      private final <T> Program<T> parse(final Parser.State<T> parsingState) {
170        if (parsingState == null) {
171          throw new IllegalStateException("parsingState", new NullPointerException("parsingState"));
172        }
173    
174        for (int tokenCount = 0; parsingState.hasNext(); tokenCount++) {
175    
176          final Token token = parsingState.next();
177          if (token != null) {
178    
179            final Token.Type tokenType = token.getType();
180            assert tokenType != null;
181    
182            switch (tokenType) {          
183    
184            case ALTERNATION:
185              this.alternate(parsingState);
186              break;
187    
188            case BEGIN_ATOM:
189              this.beginInput(parsingState);
190              break;
191    
192            case CATENATION:
193              this.catenate(parsingState);
194              break;
195    
196            case END_ATOM:
197              this.endInput(parsingState);
198              break;
199    
200            case FILTER:
201              this.filter(parsingState);
202              break;
203    
204            case ONE_OR_MORE:
205              this.oneOrMore(parsingState);
206              break;
207    
208            case START_SAVING:
209              this.startSaving(parsingState);
210              break;
211    
212            case ZERO_OR_MORE:
213              this.zeroOrMore(parsingState);
214              break;
215    
216            case ZERO_OR_ONE:
217              this.zeroOrOne(parsingState);
218              break;
219    
220            case STOP_SAVING:
221              this.stopSaving(parsingState);
222              break;
223    
224            default:
225              throw new IllegalStateException("Unknown token type: " + tokenType);
226            }
227          }
228    
229        }
230        final Program<T> program;
231        final int stackSize = parsingState.stackSize();
232        switch (stackSize) {
233        case 0:
234          throw new IllegalStateException();
235        case 1:
236          program = new Program<T>();
237          program.add(new Save<T>(0));
238          program.addAll(parsingState.pop());
239          program.add(new Stop<T>(0));
240          break;
241        default:
242          throw new IllegalStateException("Unexpected stack size: " + parsingState);
243        }
244        assert program != null;
245        program.setSource(parsingState.tokenizer);
246        program.add(new Match<T>());
247        return program;
248      }
249    
250      private final <T> void beginInput(final State<T> parsingState) {
251        if (parsingState == null) {
252          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
253        }
254        parsingState.push(Program.singleton(new BeginInput<T>()));
255      }
256    
257      private final <T> void endInput(final State<T> parsingState) {
258        if (parsingState == null) {
259          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
260        }
261        parsingState.push(Program.singleton(new EndInput<T>()));
262      }
263    
264      private final <T> void filter(final State<T> parsingState) {
265        if (parsingState == null) {
266          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
267        }
268        final Token token = parsingState.getToken();
269        assert token != null;
270        assert Token.Type.FILTER == token.getType();
271    
272        boolean exact = false;
273        String filterType = token.getFilterType();
274        if (filterType != null) {
275          final char c = filterType.charAt(0);
276          if (c == '=') {
277            exact = true;        
278          }
279        }
280    
281        parsingState.push(Program.singleton(new InstanceOfMVELFilter<T>(token.getFilterType(), token.getValue())));
282      }
283    
284      private final <T> void catenate(final State<T> parsingState) {
285        if (parsingState == null) {
286          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
287        }
288        assert parsingState.stackSize() >= 2;
289    
290        final Program<T> p2 = parsingState.pop();
291        assert p2 != null;
292    
293        final Program<T> p1 = parsingState.peek();
294        assert p1 != null;
295    
296        /*
297         * Catenation program fragment:
298         *
299         * 34: ...
300         * 35: (p1)
301         * 36: (p2)
302         * 37: ...
303         */
304    
305        p1.addAll(p2);
306      }
307    
308      private final <T> void alternate(final State<T> parsingState) {
309        if (parsingState == null) {
310          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
311        }
312        assert parsingState.stackSize() >= 2;
313    
314        final Program<T> p2 = parsingState.pop();
315        assert p2 != null;
316        assert !p2.isEmpty();
317    
318        final Program<T> p1 = parsingState.pop();
319        assert p1 != null;
320        assert !p1.isEmpty();
321    
322        /*
323         * Alternation program fragment:
324         *
325         *  46: ...
326         *  47: split +1, +(p1.size() + 2)
327         *  48: (p1)
328         *  49: jump +(p2.size() + 1)
329         *  50: (p2)
330         *  51: ...
331         */
332    
333        final Program<T> p0 = new Program<T>();
334        p0.add(new Split<T>(1, p1.size() + 2, true));
335        p0.addAll(p1);
336        final int target = p2.size() + 1;
337        p0.add(new Jump<T>(target, true)); // dangling jump target
338        p0.addAll(p2);
339    
340        parsingState.push(p0);
341      }
342    
343      private final <T> void zeroOrMore(final State<T> parsingState) {
344        if (parsingState == null) {
345          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
346        }
347        assert parsingState.stackSize() >= 1;
348    
349        final Program<T> p1 = parsingState.pop();
350        assert p1 != null;
351    
352        /*
353         * Zero or more program fragment
354         *
355         * 24 ...
356         * 25 split +1, +(p1.size() + 2)
357         * 26 (p1)
358         * 27 jump -(p1.size() + 1)
359         * 28 ...
360         */
361    
362        final Program<T> p0 = new Program<T>();
363        final int target = p1.size() + 2;
364        p0.add(0, new Split<T>(1, target, true)); // dangling split target
365        p0.addAll(p1);
366        p0.add(new Jump<T>(-(p1.size() + 1), true));
367        
368        parsingState.push(p0);
369      }
370    
371      private final <T> void zeroOrOne(final State<T> parsingState) {
372        if (parsingState == null) {
373          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
374        }
375        assert parsingState.stackSize() >= 1;
376    
377        final Program<T> p1 = parsingState.peek();
378        assert p1 != null;
379        assert !p1.isEmpty();
380    
381        /*
382         * Zero or one program fragment:
383         *
384         * 17 ...
385         * 18 split +1, +(p1.size() + 1)
386         * 19 (p1)
387         * 20 ...
388         */
389    
390        final int target = p1.size() + 1;
391        p1.add(0, new Split<T>(1, target, true)); // dangling split target
392        
393      }
394    
395      private final <T> void oneOrMore(final State<T> parsingState) {
396        if (parsingState == null) {
397          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
398        }
399        assert parsingState.stackSize() >= 1;
400        
401        final Program<T> p1 = parsingState.peek();
402        assert p1 != null;
403        assert !p1.isEmpty();
404    
405        /*
406         * One or more program fragment:
407         *
408         * 66: ...
409         * 67: (p1)
410         * 68: split -(p1.size()), +1
411         * 69: ...
412         */
413    
414        p1.add(new Split<T>(-p1.size(), 1, true));
415      }
416    
417      private final <T> void startSaving(final State<T> parsingState) {
418        if (parsingState == null) {
419          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
420        }
421        parsingState.push(Program.singleton(new Save<T>(Integer.valueOf(parsingState.groupIndex++))));
422      }
423    
424      private final <T> void stopSaving(final State<T> parsingState) {
425        if (parsingState == null) {
426          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
427        }
428        parsingState.push(Program.singleton(new Stop<T>(Integer.valueOf(--parsingState.groupIndex))));
429      }
