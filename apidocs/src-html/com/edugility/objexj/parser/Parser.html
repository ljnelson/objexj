058    /**
059     * A parser that consumes {@link Token}s from a {@link PostfixTokenizer}.
060     *
061     * @author <a href="http://about.me/lairdnelson"
062     * target="_parent">Laird Nelson</a>
063     */
064    public class Parser {
065    
066      /**
067       * Creates a new {@link Parser}.
068       */
069      public Parser() {
070        super();
071      }
072    
073      /**
074       * Constructs a new {@link PostfixTokenizer} and associated {@link
075       * Reader} machinery and then calls the {@link
076       * #parse(PostfixTokenizer)} method and returns its result.
077       *
078       * @param input the {@link String} representation of a {@link
079       * Program}; must not be {@code null}
080       *
081       * @exception IllegalArgumentException if {@code input} is {@code
082       * null}
083       *
084       * @exception IOException if the {@link String} could not be read
085       *
086       * @see #parse(PostfixTokenizer)
087       *
088       * @see PostfixTokenizer
089       */
090      public <T> Program<T> parse(final String input) throws IOException, ParseException {
091        if (input == null) {
092          throw new IllegalArgumentException("input", new NullPointerException("input"));
093        }
094        final StringReader sr = new StringReader(input);
095        final PushbackReader reader = new PushbackReader(sr) {
096            @Override
097            public String toString() {
098              return sr.toString();
099            }
100          };
101        try {
102          return this.parse(new State<T>(new PostfixTokenizer(reader)));
103        } finally {
104          try {
105            reader.close();
106          } catch (final IOException logMe) {
107            final Logger logger = Logger.getLogger(this.getClass().getName());
108            assert logger != null;
109            if (logger.isLoggable(Level.SEVERE)) {
110              logger.logp(Level.SEVERE, this.getClass().getName(), "parse", "Error closing reader", logMe);
111            }
112          }
113        }
114      }
115    
116      /**
117       * Receives a feed of {@link Token}s from the supplied {@link
118       * PostfixTokenizer} and parses them into a {@link Program}.
119       *
120       * <p>This method never returns {@code null}.</p>
121       *
122       * @param <T> the type of {@link Object} the returned {@link
123       * Program} will be capable of matching
124       *
125       * @param tokenizer the {@link PostfixTokenizer} that provides a
126       * token stream; must not be {@code null}
127       *
128       * @return a non-{@code null} {@link Program}
129       *
130       * @exception IllegalArgumentException if {@code tokenizer} is
131       * {@code null}
132       */
133      public <T> Program<T> parse(final PostfixTokenizer tokenizer) throws ParseException {
134        if (tokenizer == null) {
135          throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
136        }
137        return this.parse(new State<T>(tokenizer));
138      }
139    
140      /**
141       * Parses the supplied {@link State} into a {@link Program}.
142       *
143       * @param <T> the type of {@link Object} the returned {@link
144       * Program} will be capable of matching
145       *
146       * @param parsingState the {@link State} to parse; really just a
147       * state-holding wrapper around a {@link PostfixTokenizer}; must not
148       * be {@code null}
149       *
150       * @return a non-{@code null} {@link Program}
151       *
152       * @exception IllegalArgumentException if {@code parsingState} is
153       * {@code null}
154       */
155      private final <T> Program<T> parse(final Parser.State<T> parsingState) throws ParseException {
156        if (parsingState == null) {
157          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
158        }
159    
160        while (parsingState.hasNext()) {
161          final Token token = parsingState.next();
162          if (token != null) {
163    
164            final Token.Type tokenType = token.getType();
165            assert tokenType != null;
166    
167            switch (tokenType) {          
168    
169            case ALTERNATION:
170              this.alternate(parsingState);
171              break;
172    
173            case BEGIN_ATOM:
174              this.beginInput(parsingState);
175              break;
176    
177            case CATENATION:
178              this.catenate(parsingState);
179              break;
180    
181            case END_ATOM:
182              this.endInput(parsingState);
183              break;
184    
185            case FILTER:
186              this.filter(parsingState);
187              break;
188    
189            case ONE_OR_MORE:
190              this.oneOrMore(parsingState);
191              break;
192    
193            case START_SAVING:
194              this.startSaving(parsingState);
195              break;
196    
197            case ZERO_OR_MORE:
198              this.zeroOrMore(parsingState);
199              break;
200    
201            case ZERO_OR_ONE:
202              this.zeroOrOne(parsingState);
203              break;
204    
205            case STOP_SAVING:
206              this.stopSaving(parsingState);
207              break;
208    
209            default:
210              parsingState.setErrorIndex(parsingState.getIndex());
211              throw new ParseException(String.format("Unknown token type: %s", tokenType), parsingState.getErrorIndex());
212            }
213          }
214    
215        }
216        final Program<T> program;
217        final int stackSize = parsingState.stackSize();
218        switch (stackSize) {
219        case 0:
220          throw new IllegalStateException();
221        case 1:
222          program = new Program<T>();
223          program.add(new Save<T>(0));
224          program.addAll(parsingState.pop());
225          program.add(new Stop<T>(0));
226          break;
227        default:
228          throw new IllegalStateException("Unexpected stack size: " + parsingState);
229        }
230        assert program != null;
231        program.setSource(parsingState.tokenizer);
232        program.add(new Match<T>());
233        return program;
234      }
235    
236      private final <T> void beginInput(final State<T> parsingState) {
237        if (parsingState == null) {
238          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
239        }
240        parsingState.push(Program.singleton(new BeginInput<T>()));
241      }
242    
243      private final <T> void endInput(final State<T> parsingState) {
244        if (parsingState == null) {
245          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
246        }
247        parsingState.push(Program.singleton(new EndInput<T>()));
248      }
249    
250      private final <T> void filter(final State<T> parsingState) {
251        if (parsingState == null) {
252          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
253        }
254        final Token token = parsingState.getToken();
255        assert token != null;
256        assert Token.Type.FILTER == token.getType();
257    
258        boolean exact = false;
259        String filterType = token.getFilterType();
260        if (filterType != null) {
261          final char c = filterType.charAt(0);
262          if (c == '=') {
263            exact = true;        
264          }
265        }
266    
267        parsingState.push(Program.singleton(new InstanceOfMVELFilter<T>(token.getFilterType(), token.getValue())));
268      }
269    
270      private final <T> void catenate(final State<T> parsingState) {
271        if (parsingState == null) {
272          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
273        }
274        assert parsingState.stackSize() >= 2;
275    
276        final Program<T> p2 = parsingState.pop();
277        assert p2 != null;
278    
279        final Program<T> p1 = parsingState.peek();
280        assert p1 != null;
281    
282        /*
283         * Catenation program fragment:
284         *
285         * 34: ...
286         * 35: (p1)
287         * 36: (p2)
288         * 37: ...
289         */
290    
291        p1.addAll(p2);
292      }
293    
294      private final <T> void alternate(final State<T> parsingState) {
295        if (parsingState == null) {
296          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
297        }
298        assert parsingState.stackSize() >= 2;
299    
300        final Program<T> p2 = parsingState.pop();
301        assert p2 != null;
302        assert !p2.isEmpty();
303    
304        final Program<T> p1 = parsingState.pop();
305        assert p1 != null;
306        assert !p1.isEmpty();
307    
308        /*
309         * Alternation program fragment:
310         *
311         *  46: ...
312         *  47: split +1, +(p1.size() + 2)
313         *  48: (p1)
314         *  49: jump +(p2.size() + 1)
315         *  50: (p2)
316         *  51: ...
317         */
318    
319        final Program<T> p0 = new Program<T>();
320        p0.add(new Split<T>(1, p1.size() + 2, true));
321        p0.addAll(p1);
322        final int target = p2.size() + 1;
323        p0.add(new Jump<T>(target, true)); // dangling jump target
324        p0.addAll(p2);
325    
326        parsingState.push(p0);
327      }
328    
329      private final <T> void zeroOrMore(final State<T> parsingState) {
330        if (parsingState == null) {
331          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
332        }
333        assert parsingState.stackSize() >= 1;
334    
335        final Program<T> p1 = parsingState.pop();
336        assert p1 != null;
337    
338        /*
339         * Zero or more program fragment
340         *
341         * 24 ...
342         * 25 split +1, +(p1.size() + 2)
343         * 26 (p1)
344         * 27 jump -(p1.size() + 1)
345         * 28 ...
346         */
347    
348        final Program<T> p0 = new Program<T>();
349        final int target = p1.size() + 2;
350        p0.add(0, new Split<T>(1, target, true)); // dangling split target
351        p0.addAll(p1);
352        p0.add(new Jump<T>(-(p1.size() + 1), true));
353        
354        parsingState.push(p0);
355      }
356    
357      private final <T> void zeroOrOne(final State<T> parsingState) {
358        if (parsingState == null) {
359          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
360        }
361        assert parsingState.stackSize() >= 1;
362    
363        final Program<T> p1 = parsingState.peek();
364        assert p1 != null;
365        assert !p1.isEmpty();
366    
367        /*
368         * Zero or one program fragment:
369         *
370         * 17 ...
371         * 18 split +1, +(p1.size() + 1)
372         * 19 (p1)
373         * 20 ...
374         */
375    
376        final int target = p1.size() + 1;
377        p1.add(0, new Split<T>(1, target, true)); // dangling split target
378        
379      }
380    
381      private final <T> void oneOrMore(final State<T> parsingState) {
382        if (parsingState == null) {
383          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
384        }
385        assert parsingState.stackSize() >= 1;
386        
387        final Program<T> p1 = parsingState.peek();
388        assert p1 != null;
389        assert !p1.isEmpty();
390    
391        /*
392         * One or more program fragment:
393         *
394         * 66: ...
395         * 67: (p1)
396         * 68: split -(p1.size()), +1
397         * 69: ...
398         */
399    
400        p1.add(new Split<T>(-p1.size(), 1, true));
401      }
402    
403      private final <T> void startSaving(final State<T> parsingState) {
404        if (parsingState == null) {
405          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
406        }
407        parsingState.push(Program.singleton(new Save<T>(Integer.valueOf(parsingState.groupIndex++))));
408      }
409    
410      private final <T> void stopSaving(final State<T> parsingState) {
411        if (parsingState == null) {
412          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
413        }
414        parsingState.push(Program.singleton(new Stop<T>(Integer.valueOf(--parsingState.groupIndex))));
415      }

418      /*
419       * Nested and inner classes.
420       */
421    
422    
423      private static final class State<T> extends ParsePosition implements Iterator<Token> {
424    
425        private int groupIndex;
426        
427        private Token token;
428    
429        private final Deque<Program<T>> stack;
430    
431        private final PostfixTokenizer tokenizer;
432    
433        private State(final PostfixTokenizer tokenizer) {
434          super(-1);
435          if (tokenizer == null) {
436            throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
437          }
438          this.tokenizer = tokenizer;
439          this.stack = new ArrayDeque<Program<T>>();
440          this.groupIndex = 1;
441        }
442    
443        public final Token getToken() {
444          return this.token;
445        }
446    
447        @Override
448        public final boolean hasNext() {
449          return this.tokenizer.hasNext();
450        }
451    
452        @Override
453        public final Token next() {
454          this.token = this.tokenizer.next();
455          this.setIndex(this.getIndex() + 1);
456          return this.token;
457        }
458    
459        @Override
460        public final void remove() {
461          this.tokenizer.remove();
462        }
463    
464        public final int stackSize() {
465          return this.stack.size();
466        }
467    
468        public void push(final Program<T> programFragment) {
469          if (programFragment == null) {
470            throw new IllegalArgumentException("programFragment", new NullPointerException("programFragment"));
471          }
472          this.stack.push(programFragment);
473        }
474    
475        public Program<T> peek() {
476          return this.stack.peek();
477        }
478    
479        public Program<T> pop() {
480          return this.stack.pop();
481        }
482    
483        @Override
484        public String toString() {
485          final Formatter f = new Formatter(new StringBuilder());
486          f.format("Parsing state:%n");
487          f.format("Current token: %s%n", this.getToken());
488          f.format("Stack size: %d%n", this.stackSize());
489          f.format("Stack: %s%n", this.stack);
490          return f.toString();
491        }
492        
493      }
