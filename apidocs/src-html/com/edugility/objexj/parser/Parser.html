058    /**
059     * A parser that consumes {@link Token}s from a {@link PostfixTokenizer}.
060     *
061     * @author <a href="http://about.me/lairdnelson"
062     * target="_parent">Laird Nelson</a>
063     */
064    public class Parser {
065    
066      /**
067       * Creates a new {@link Parser}.
068       */
069      public Parser() {
070        super();
071      }
072    
073      /**
074       * Constructs a new {@link PostfixTokenizer} and associated {@link
075       * Reader} machinery and then calls the {@link
076       * #parse(PostfixTokenizer)} method and returns its result.
077       *
078       * @param <T> the type of {@link Object} the resulting {@link
079       * Program} will work with
080       *
081       * @param input the {@link String} representation of a {@link
082       * Program}; must not be {@code null}
083       *
084       * @exception IllegalArgumentException if {@code input} is {@code
085       * null}
086       *
087       * @exception IOException if the {@link String} could not be read
088       *
089       * @see #parse(PostfixTokenizer)
090       *
091       * @see PostfixTokenizer
092       */
093      public <T> Program<T> parse(final String input) throws IOException, ParseException {
094        if (input == null) {
095          throw new IllegalArgumentException("input", new NullPointerException("input"));
096        }
097        final StringReader sr = new StringReader(input);
098        final PushbackReader reader = new PushbackReader(sr) {
099            @Override
100            public String toString() {
101              return sr.toString();
102            }
103          };
104        try {
105          return this.parse(new State<T>(new PostfixTokenizer(reader)));
106        } finally {
107          try {
108            reader.close();
109          } catch (final IOException logMe) {
110            final Logger logger = Logger.getLogger(this.getClass().getName());
111            assert logger != null;
112            if (logger.isLoggable(Level.SEVERE)) {
113              logger.logp(Level.SEVERE, this.getClass().getName(), "parse", "Error closing reader", logMe);
114            }
115          }
116        }
117      }
118    
119      /**
120       * Receives a feed of {@link Token}s from the supplied {@link
121       * PostfixTokenizer} and parses them into a {@link Program}.
122       *
123       * <p>This method never returns {@code null}.</p>
124       *
125       * @param <T> the type of {@link Object} the returned {@link
126       * Program} will be capable of matching
127       *
128       * @param tokenizer the {@link PostfixTokenizer} that provides a
129       * token stream; must not be {@code null}
130       *
131       * @return a non-{@code null} {@link Program}
132       *
133       * @exception IllegalArgumentException if {@code tokenizer} is
134       * {@code null}
135       */
136      public <T> Program<T> parse(final PostfixTokenizer tokenizer) throws ParseException {
137        if (tokenizer == null) {
138          throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
139        }
140        return this.parse(new State<T>(tokenizer));
141      }
142    
143      /**
144       * Parses the supplied {@link State} into a {@link Program}.
145       *
146       * @param <T> the type of {@link Object} the returned {@link
147       * Program} will be capable of matching
148       *
149       * @param parsingState the {@link State} to parse; really just a
150       * state-holding wrapper around a {@link PostfixTokenizer}; must not
151       * be {@code null}
152       *
153       * @return a non-{@code null} {@link Program}
154       *
155       * @exception IllegalArgumentException if {@code parsingState} is
156       * {@code null}
157       */
158      private final <T> Program<T> parse(final Parser.State<T> parsingState) throws ParseException {
159        if (parsingState == null) {
160          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
161        }
162    
163        while (parsingState.hasNext()) {
164          final Token token = parsingState.next();
165          if (token != null) {
166    
167            final Token.Type tokenType = token.getType();
168            assert tokenType != null;
169    
170            switch (tokenType) {          
171    
172            case ALTERNATION:
173              this.alternate(parsingState);
174              break;
175    
176            case BEGIN_ATOM:
177              this.beginInput(parsingState);
178              break;
179    
180            case CATENATION:
181              this.catenate(parsingState);
182              break;
183    
184            case END_ATOM:
185              this.endInput(parsingState);
186              break;
187    
188            case FILTER:
189              this.filter(parsingState);
190              break;
191    
192            case ONE_OR_MORE:
193              this.oneOrMore(parsingState);
194              break;
195    
196            case START_SAVING:
197              this.startSaving(parsingState);
198              break;
199    
200            case ZERO_OR_MORE:
201              this.zeroOrMore(parsingState);
202              break;
203    
204            case ZERO_OR_ONE:
205              this.zeroOrOne(parsingState);
206              break;
207    
208            case STOP_SAVING:
209              this.stopSaving(parsingState);
210              break;
211    
212            default:
213              parsingState.setErrorIndex(parsingState.getIndex());
214              throw new ParseException(String.format("Unknown token type: %s", tokenType), parsingState.getErrorIndex());
215            }
216          }
217    
218        }
219        final Program<T> program;
220        final int stackSize = parsingState.stackSize();
221        switch (stackSize) {
222        case 0:
223          throw new IllegalStateException();
224        case 1:
225          program = new Program<T>();
226          program.add(new Save<T>(0));
227          program.addAll(parsingState.pop());
228          program.add(new Stop<T>(0));
229          break;
230        default:
231          throw new IllegalStateException("Unexpected stack size: " + parsingState);
232        }
233        assert program != null;
234        program.setSource(parsingState.tokenizer);
235        program.add(new Match<T>());
236        return program;
237      }
238    
239      private final <T> void beginInput(final State<T> parsingState) {
240        if (parsingState == null) {
241          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
242        }
243        parsingState.push(Program.singleton(new BeginInput<T>()));
244      }
245    
246      private final <T> void endInput(final State<T> parsingState) {
247        if (parsingState == null) {
248          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
249        }
250        parsingState.push(Program.singleton(new EndInput<T>()));
251      }
252    
253      private final <T> void filter(final State<T> parsingState) {
254        if (parsingState == null) {
255          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
256        }
257        final Token token = parsingState.getToken();
258        assert token != null;
259        assert Token.Type.FILTER == token.getType();
260    
261        boolean exact = false;
262        String filterType = token.getFilterType();
263        if (filterType != null) {
264          final char c = filterType.charAt(0);
265          if (c == '=') {
266            exact = true;        
267          }
268        }
269    
270        parsingState.push(Program.singleton(new InstanceOfMVELFilter<T>(token.getFilterType(), token.getValue())));
271      }
272    
273      private final <T> void catenate(final State<T> parsingState) {
274        if (parsingState == null) {
275          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
276        }
277        assert parsingState.stackSize() >= 2;
278    
279        final Program<T> p2 = parsingState.pop();
280        assert p2 != null;
281    
282        final Program<T> p1 = parsingState.peek();
283        assert p1 != null;
284    
285        /*
286         * Catenation program fragment:
287         *
288         * 34: ...
289         * 35: (p1)
290         * 36: (p2)
291         * 37: ...
292         */
293    
294        p1.addAll(p2);
295      }
296    
297      private final <T> void alternate(final State<T> parsingState) {
298        if (parsingState == null) {
299          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
300        }
301        assert parsingState.stackSize() >= 2;
302    
303        final Program<T> p2 = parsingState.pop();
304        assert p2 != null;
305        assert !p2.isEmpty();
306    
307        final Program<T> p1 = parsingState.pop();
308        assert p1 != null;
309        assert !p1.isEmpty();
310    
311        /*
312         * Alternation program fragment:
313         *
314         *  46: ...
315         *  47: split +1, +(p1.size() + 2)
316         *  48: (p1)
317         *  49: jump +(p2.size() + 1)
318         *  50: (p2)
319         *  51: ...
320         */
321    
322        final Program<T> p0 = new Program<T>();
323        p0.add(new Split<T>(1, p1.size() + 2, true));
324        p0.addAll(p1);
325        final int target = p2.size() + 1;
326        p0.add(new Jump<T>(target, true)); // dangling jump target
327        p0.addAll(p2);
328    
329        parsingState.push(p0);
330      }
331    
332      private final <T> void zeroOrMore(final State<T> parsingState) {
333        if (parsingState == null) {
334          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
335        }
336        assert parsingState.stackSize() >= 1;
337    
338        final Program<T> p1 = parsingState.pop();
339        assert p1 != null;
340    
341        /*
342         * Zero or more program fragment
343         *
344         * 24 ...
345         * 25 split +1, +(p1.size() + 2)
346         * 26 (p1)
347         * 27 jump -(p1.size() + 1)
348         * 28 ...
349         */
350    
351        final Program<T> p0 = new Program<T>();
352        final int target = p1.size() + 2;
353        p0.add(0, new Split<T>(1, target, true)); // dangling split target
354        p0.addAll(p1);
355        p0.add(new Jump<T>(-(p1.size() + 1), true));
356        
357        parsingState.push(p0);
358      }
359    
360      private final <T> void zeroOrOne(final State<T> parsingState) {
361        if (parsingState == null) {
362          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
363        }
364        assert parsingState.stackSize() >= 1;
365    
366        final Program<T> p1 = parsingState.peek();
367        assert p1 != null;
368        assert !p1.isEmpty();
369    
370        /*
371         * Zero or one program fragment:
372         *
373         * 17 ...
374         * 18 split +1, +(p1.size() + 1)
375         * 19 (p1)
376         * 20 ...
377         */
378    
379        final int target = p1.size() + 1;
380        p1.add(0, new Split<T>(1, target, true)); // dangling split target
381        
382      }
383    
384      private final <T> void oneOrMore(final State<T> parsingState) {
385        if (parsingState == null) {
386          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
387        }
388        assert parsingState.stackSize() >= 1;
389        
390        final Program<T> p1 = parsingState.peek();
391        assert p1 != null;
392        assert !p1.isEmpty();
393    
394        /*
395         * One or more program fragment:
396         *
397         * 66: ...
398         * 67: (p1)
399         * 68: split -(p1.size()), +1
400         * 69: ...
401         */
402    
403        p1.add(new Split<T>(-p1.size(), 1, true));
404      }
405    
406      private final <T> void startSaving(final State<T> parsingState) {
407        if (parsingState == null) {
408          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
409        }
410        parsingState.push(Program.singleton(new Save<T>(Integer.valueOf(parsingState.groupIndex++))));
411      }
412    
413      private final <T> void stopSaving(final State<T> parsingState) {
414        if (parsingState == null) {
415          throw new IllegalArgumentException("parsingState", new NullPointerException("parsingState"));
416        }
417        parsingState.push(Program.singleton(new Stop<T>(Integer.valueOf(--parsingState.groupIndex))));
418      }

421      /*
422       * Nested and inner classes.
423       */
424    
425    
426      private static final class State<T> extends ParsePosition implements Iterator<Token> {
427    
428        private int groupIndex;
429        
430        private Token token;
431    
432        private final Deque<Program<T>> stack;
433    
434        private final PostfixTokenizer tokenizer;
435    
436        private State(final PostfixTokenizer tokenizer) {
437          super(-1);
438          if (tokenizer == null) {
439            throw new IllegalArgumentException("tokenizer", new NullPointerException("tokenizer"));
440          }
441          this.tokenizer = tokenizer;
442          this.stack = new ArrayDeque<Program<T>>();
443          this.groupIndex = 1;
444        }
445    
446        public final Token getToken() {
447          return this.token;
448        }
449    
450        @Override
451        public final boolean hasNext() {
452          return this.tokenizer.hasNext();
453        }
454    
455        @Override
456        public final Token next() {
457          this.token = this.tokenizer.next();
458          this.setIndex(this.getIndex() + 1);
459          return this.token;
460        }
461    
462        @Override
463        public final void remove() {
464          this.tokenizer.remove();
465        }
466    
467        public final int stackSize() {
468          return this.stack.size();
469        }
470    
471        public void push(final Program<T> programFragment) {
472          if (programFragment == null) {
473            throw new IllegalArgumentException("programFragment", new NullPointerException("programFragment"));
474          }
475          this.stack.push(programFragment);
476        }
477    
478        public Program<T> peek() {
479          return this.stack.peek();
480        }
481    
482        public Program<T> pop() {
483          return this.stack.pop();
484        }
485    
486        @Override
487        public String toString() {
488          final Formatter f = new Formatter(new StringBuilder());
489          f.format("Parsing state:%n");
490          f.format("Current token: %s%n", this.getToken());
491          f.format("Stack size: %d%n", this.stackSize());
492          f.format("Stack: %s%n", this.stack);
493          return f.toString();
494        }
495        
496      }
