046    public class PostfixTokenizer implements Iterator<Token> {
047    
048      private static final long serialVersionUID = 1L;
049    
050      private enum State {
051        BEGIN_ATOM, END_ATOM, START, START_SAVING_OR_FILTER, INVALID, FILTER, MVEL, END_OF_FILTER, OPERATOR, LEFT_PAREN, RIGHT_PAREN, STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE, END
052      }
053    
054      private final Map<Integer, Token> tokens;
055    
056      private transient Logger logger;
057    
058      private transient final PushbackReader reader;
059    
060      final transient LinkedList<Token> output;
061    
062      private transient Throwable error;
063    
064      private transient State state;
065    
066      private transient int position;
067    
068      private transient final Deque<Token> stack;
069    
070      public PostfixTokenizer(final PushbackReader reader) throws IOException {
071        super();
072        this.logger = Logger.getLogger(this.getClass().getName());
073        assert this.logger != null;
074        if (reader == null) {
075          throw new IllegalArgumentException("reader", new NullPointerException("reader"));
076        }
077        this.tokens = new HashMap<Integer, Token>();
078        this.setupTokens();
079        this.output = new LinkedList<Token>();
080        this.stack = new ArrayDeque<Token>();
081        this.reader = reader;
082        this.state = State.START;
083        this.prime();
084      }
085    
086      private static final String buildIllegalStateExceptionMessage(final Reader r, final int c, final int position) {
087        return String.format("Unexpected character (%c) at position %d in Reader %s", c, position, r);
088      }
089    
090      private final void prime() throws IOException {
091        if (this.state == State.INVALID) {
092          throw new IllegalStateException();
093        }
094    
095        final StringBuilder sb = new StringBuilder();
096        int parenCount = 0;
097        int c = -1;
098    
099        READ_LOOP:
100        for (; (c = this.read()) != -1; this.position++) {
101          switch (this.state) {
102    
103            /*
104             * In this state switch block, "forwarding" states (states
105             * that peek ahead to determine what real state to forward to)
106             * are commented // LIKE THIS, and "real" states are commented
107             * //
108             * // LIKE THIS
109             * //
110             */
111    
112    
113            // START
114          case START:
115            switch (c) {
116    
117            case '^':
118              this.unread(c);
119              this.state = State.BEGIN_ATOM;
120              break;
121    
122            case '(':
123              this.insertBeginAtomZeroOrMoreAndCatenation();
124              this.unread(c);
125              this.state = State.LEFT_PAREN;
126              break;
127    
128            default:
129              if (Character.isWhitespace(c)) {
130                continue READ_LOOP;
131              } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
132                this.insertBeginAtomZeroOrMoreAndCatenation();
133                sb.append((char)c);
134                this.state = State.FILTER;
135              } else {
136                throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
137              }
138              break;
139            }
140            break;
141    
142    
143            //
144            // BEGIN_ATOM
145            //
146          case BEGIN_ATOM:
147            assert c == '^';
148            assert sb.length() == 0;
149            this.output.add(this.tokenFor(c));
150            this.insertCatenation();
151            assert this.state == State.START_SAVING_OR_FILTER;
152            break READ_LOOP;
153    
154    
155            //
156            // END_ATOM
157            //
158          case END_ATOM:
159            assert c == '$';
160            this.insertCatenation();
161            this.output.add(this.tokenFor(c));
162            sb.setLength(0);
163            this.state = State.END;
164            break READ_LOOP;
165    
166    
167            //
168            // LEFT_PAREN
169            //
170          case LEFT_PAREN:
171            assert c == '(';
172            /*
173             * Shunting-yard algorithm: "If the token is a left
174             * parenthesis, then push it onto the stack."
175             *
176             * In our case, a parenthesis is both a grouping operator and
177             * an atom-- a signal to the parser that a START_SAVING
178             * instruction should be fired.  So we also have to add it to
179             * the output.
180             *
181             * TENTATIVE: since it's an atom, we also need to pretend we
182             * found a concatenation operator right afterwards.
183             */
184    
185            // First find a START_SAVING (atom) token.
186            final Token token = new Token(Token.Type.START_SAVING);
187            
188            // TODO: support named capture groups
189    
190            // Add it to the output (it's an atom):
191            this.output.add(token);
192    
193            // Then pretend we found a concatenation.
194            this.insertCatenation();
195    
196            // Then pretend we found a START_GROUP.
197            final Token t = new Token(Token.Type.START_GROUP);
198            assert t != null;
199            this.push(t);
200    
201            this.state = State.START_SAVING_OR_FILTER;
202            break READ_LOOP;
203    
204    
205            //
206            // RIGHT_PAREN
207            //
208          case RIGHT_PAREN:
209            assert c == ')'; // OK
210            /*
211             * Shunting-yard algorithm:
212             * "If the token is a right parenthesis:
213             * " * Until the token at the top of the stack is a left parenthesis, pop operators off the stack onto the output queue.
214             * " * Pop the left parenthesis from the stack, but not onto the output queue.
215             * " * If the token at the top of the stack is a function token, pop it onto the output queue.
216             * " * If the stack runs out without finding a left parenthesis, then there are mismatched parentheses."
217             */
218    
219            final int originalOutputSize = this.output.size();
220            while (!this.stack.isEmpty() && this.stack.peek().getType() != Token.Type.START_GROUP) {
221              this.output.add(this.stack.pop());
222            }
223            if (this.stack.isEmpty()) {
224              throw new IllegalStateException("Mismatched parentheses");
225            }
226    
227            assert Token.Type.START_GROUP == this.stack.peek().getType();
228            this.stack.pop();
229    
230            // The original shunting-yard algorithm discards the parens
231            // because in that algorithm they are used only for grouping.
232            // In regular expressions, they are also atoms.  So they
233            // belong in the output queue. We already added the
234            // START_SAVING token, so now we need to add the STOP_SAVING.
235    
236            // Concatenate...
237            this.insertCatenation();
238    
239            // ...with stop saving:
240            this.output.add(new Token(Token.Type.STOP_SAVING));
241    
242            this.state = State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
243            break READ_LOOP; // only because we added something to the output queue
244    
245    
246            // START_SAVING_OR_FILTER
247          case START_SAVING_OR_FILTER:
248            if (Character.isWhitespace(c)) {
249              continue READ_LOOP;
250            } else if (c == '(') {
251              // Forward to LEFT_PAREN state
252              this.unread(c);
253              this.state = State.LEFT_PAREN;
254            } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
255              // Forward to FILTER state
256              sb.append((char)c);
257              this.state = State.FILTER;
258            } else {
259              throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
260            }
261            break;
262    
263    
264            // FILTER (really more like START_OF_FILTER?)
265          case FILTER:
266            switch (c) {
267    
268            case '(':
269              // Bumped into an MVEL guard.  Create the token, but don't
270              // return it yet.  Fill in its filterType.  Transition to
271              // the MVEL state and have it fill in the details.
272              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
273              parenCount = 1;
274              sb.setLength(0);
275              this.state = State.MVEL;
276              break;
277    
278            case ')': // un-MVELed atom; just ran into a RIGHT_PAREN; OK
279              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
280              sb.setLength(0);
281              this.unread(c);
282              this.state = State.RIGHT_PAREN;
283              break READ_LOOP;
284    
285            case '+':
286            case '*':
287            case '?':
288            case '|':
289            case '/':
290            case ',':
291              // Bumped into a genuine operator.  So create a FILTER token
292              // without an accompanying MVEL guard, and forward to the
293              // OPERATOR state.
294              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
295              sb.setLength(0);
296              this.unread(c);
297              this.state = State.OPERATOR;
298              break READ_LOOP;
299    
300            case '$':
301              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
302              sb.setLength(0);
303              this.unread(c);
304              this.state = State.END_ATOM;
305              break READ_LOOP;
306    
307            default:
308              if (Character.isJavaIdentifierPart(c)) {
309                sb.append((char)c);
310              } else if (Character.isWhitespace(c)) {
311                continue READ_LOOP;
312              } else if (c == '.') {
313                sb.append('.');
314                c = this.read();
315                if (c == -1) {
316                  this.unread(c);
317                } else if (Character.isJavaIdentifierStart(c)) {
318                  sb.append((char)c);
319                } else {
320                  throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
321                }
322              } else {
323                throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
324              }
325              break;
326            }
327            break;
328    
329    
330            // MVEL
331          case MVEL:
332            assert parenCount > 0;
333            switch (c) {
334    
335            case '(':
336              parenCount++;
337              sb.append((char)c);
338              break;
339    
340            case ')': // Maybe a paren inside an MVEL block; maybe the paren that closes it; OK
341              parenCount--;
342              if (parenCount == 0) {
343                final Token filter = this.output.getLast();
344                assert filter != null;
345                assert filter.getFilterType() != null;
346                assert filter.getType() == Token.Type.FILTER;
347                filter.setValue(sb.toString());
348                sb.setLength(0);
349                this.state = State.END_OF_FILTER;
350                break READ_LOOP;
351              } else if (parenCount < 0) {
352                throw new IllegalStateException("Mismatched parentheses");
353              } else {
354                sb.append((char)c);
355              }
356              break;
357    
358            default:
359              sb.append((char)c);
360              break;
361            }
362            break;
363    
364    
365            // END_OF_FILTER
366          case END_OF_FILTER:
367            switch (c) {
368    
369            case '$':
370              sb.setLength(0);
371              this.unread(c);
372              this.state = State.END_ATOM;
373              break;
374    
375            case ')': // OK
376              sb.setLength(0);
377              this.unread(c);
378              this.state = State.RIGHT_PAREN;
379              break;
380    
381            case '+':
382            case '*':
383            case '?':
384            case '|':
385            case '/':
386            case ',':
387              // Forward to OPERATOR state
388              sb.setLength(0);
389              this.unread(c);
390              this.state = State.OPERATOR;
391              break;
392    
393            default:
394              if (Character.isWhitespace(c)) {
395                continue READ_LOOP;
396              } else {
397                throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
398              }
399    
400            }
401            break;
402    
403    
404            //
405            // OPERATOR
406            //
407          case OPERATOR:
408            sb.setLength(0);
409            if (Character.isWhitespace(c)) {
410              continue READ_LOOP;
411            } else if (this.handleOperator(this.tokenFor(c))) {
412              break READ_LOOP;
413            }
414            break;
415    
416    
417            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE
418            //
419            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE permits
420            // only those operators that come after a (potentially
421            // augmented) filter token, or a group of such tokens, and
422            // weeds out others before proceeding to the OPERATOR state.
423            //
424            // This handles closing of submatches (")") and end-of-input
425            // ("$") as special (legal) cases.  Otherwise the operator has
426            // to be catenation or alternation.
427            //
428            // In these cases we forward to the OPERATOR state.
429            //
430            // Any other character causes an IllegalStateException.
431          case STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE:
432            switch (c) {
433    
434            case ')': // stop saving
435              this.unread(c);
436              this.state = State.RIGHT_PAREN;
437              break;
438    
439            case '$': // end of input
440              this.unread(c);
441              this.state = State.END_ATOM;
442              break;
443    
444            case '/': // catenation
445            case ',': // catenation
446            case '|': // alternation
447              this.reader.unread(c);
448              this.state = State.OPERATOR;
449              break;
450    
451            default:
452              if (Character.isWhitespace(c)) {
453                continue READ_LOOP;
454              } else {
455                throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
456              }
457            }
458            break;
459    
460    
461            //
462            // END
463            //
464          case END:
465            if (!Character.isWhitespace(c)) {
466              throw new IllegalStateException(buildIllegalStateExceptionMessage(this.reader, c, this.position));
467            }
468            break;
469    
470            //
471            // DEFAULT
472            //
473          default:
474            throw new IllegalStateException("Unexpected state: " + this.state);
475          }
476        } // READ_LOOP
477    
478        if (c == -1) {
479          switch (this.state) {
480          case FILTER:
481            this.output.add(new Token(Token.Type.FILTER, sb.toString()));
482            sb.setLength(0);
483            break;
484          case END:
485            break;
486          default:
487            if (this.logger.isLoggable(Level.WARNING)) {
488              this.logger.logp(Level.WARNING, this.getClass().getName(), "prime", "Unhandled cleanup in state {0} with stringbuffer {1}", new Object[] { state, sb });
489            }
490            break;
491          }
492          this.state = State.END;
493          while (!this.stack.isEmpty()) {
494            final Token top = this.stack.peek();
495            assert top != null;
496            final Token.Type type = top.getType();
497            assert type != null;
498    
499            if (type == Token.Type.START_GROUP || type == Token.Type.STOP_GROUP) {
500              throw new IllegalStateException("Mismatched parentheses");
501            }
502            this.output.add(this.stack.pop());
503          }
504        }
505    
506      }

508      private final Token createFilterToken(final StringBuilder sb) {
509        if (sb == null) {
510          throw new IllegalArgumentException("sb", new NullPointerException("sb"));
511        }
512        return new Token(Token.Type.FILTER, sb.toString());
513      }
514    
515      private final void insertBeginAtomZeroOrMoreAndCatenation() {
516        this.output.add(this.tokenFor('^'));
517        this.insertCatenation();
518        this.insertZeroOrMoreAnything();
519        this.insertCatenation();
520      }
521    
522      private final void insertZeroOrMoreAnything() {
523        this.output.add(new Token(Token.Type.FILTER, "java.lang.Object"));
524        this.handleOperator(this.tokenFor('*'));
525      }
526    
527      private final boolean insertCatenation() {
528        return this.handleOperator(this.tokenFor('/'));
529      }
530    
531      private final boolean handleOperator(final Token o1) {
532        if (o1 == null) {
533          throw new IllegalArgumentException("o1", new NullPointerException("o1"));
534        } else if (!o1.isOperator()) {
535          throw new IllegalArgumentException("!o1.isOperator()");
536        }
537        Token o2;
538        boolean added = false;
539        while (!this.stack.isEmpty() && this.assertIsOperator(o2 = this.stack.peek())) {
540          if (o1.compareTo(o2) <= 0) {
541            added = this.output.add(this.stack.pop()) || added;
542          } else {
543            break;
544          }
545        }
546        this.push(o1);
547        this.state = this.stateAfterOperator(o1);
548        return added;
549      }
550    
551      private final void push(final Token token) {
552        if (this.logger.isLoggable(Level.FINER)) {
553          this.logger.entering(this.getClass().getName(), "push", token);
554        }
555        if (token == null) {
556          throw new IllegalArgumentException("token", new NullPointerException("token"));
557        } else if (Token.Type.STOP_GROUP == token.getType()) {
558          throw new IllegalArgumentException("token.getType() == Token.Type.STOP_GROUP");
559        }
560        this.stack.push(token);
561        if (this.logger.isLoggable(Level.FINER)) {
562          this.logger.exiting(this.getClass().getName(), "push");
563        }
564      }
565    
566      private final State stateAfterOperator(final Token token) {
567        if (token == null) {
568          throw new IllegalArgumentException("token", new NullPointerException("token"));
569        }
570        final Token.Type type = token.getType();
571        assert type != null;
572        switch (type) {
573        case ONE_OR_MORE:
574        case ZERO_OR_MORE:
575        case ZERO_OR_ONE:
576          return State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
577        case ALTERNATION:
578        case CATENATION:
579          return State.START_SAVING_OR_FILTER;
580        default:
581          throw new IllegalArgumentException("token: " + token);
582        }
583      }
584    
585      private final void setupTokens() {
586        this.tokens.clear();
587        this.tokens.put(Integer.valueOf('^'), new Token(Token.Type.BEGIN_ATOM));
588        this.tokens.put(Integer.valueOf('$'), new Token(Token.Type.END_ATOM));
589        this.tokens.put(Integer.valueOf('+'), new Token(Token.Type.ONE_OR_MORE));
590        this.tokens.put(Integer.valueOf('*'), new Token(Token.Type.ZERO_OR_MORE));
591        this.tokens.put(Integer.valueOf('?'), new Token(Token.Type.ZERO_OR_ONE));
592        this.tokens.put(Integer.valueOf('|'), new Token(Token.Type.ALTERNATION));
593        final Token catenation = new Token(Token.Type.CATENATION);
594        this.tokens.put(Integer.valueOf('/'), catenation);
595        this.tokens.put(Integer.valueOf(','), catenation);
596        this.tokens.put(Integer.valueOf('('), new Token(Token.Type.START_SAVING));
597        this.tokens.put(Integer.valueOf(')'), new Token(Token.Type.STOP_SAVING));
598      }
599    
600      private final Token tokenFor(final int c) {
601        final Token token = this.tokens.get(Integer.valueOf(c));
602        if (token == null) {
603          throw new IllegalArgumentException("c: " + (char)c);
604        }
605        return token;
606      }
607    
608      private final boolean assertIsOperator(final Token token) {
609        if (token == null) {
610          throw new IllegalArgumentException("token", new NullPointerException("token"));
611        }
612        if (!token.isOperator()) {
613          throw new IllegalArgumentException("!token.isOperator(): " + token);
614        }
615        return true;
616      }
617    
618      private final int read() throws IOException {
619        final int c = this.reader.read();
620        ++this.position;
621        return c;
622      }
623    
624      private final int unread(final int c) throws IOException {
625        this.reader.unread(c);
626        return --this.position;
627      }
628    
629      @Override
630      public boolean hasNext() {
631        if (this.error != null) {
632          throw new IllegalStateException(this.error);
633        }
634        return !this.output.isEmpty();
635      }
636    
637      @Override
638      public Token next() {
639        if (this.error != null) {
640          throw (NoSuchElementException)new NoSuchElementException().initCause(this.error);
641        } else if (this.output.isEmpty()) {
642          throw new NoSuchElementException();
643        }
644        final Token returnValue = this.output.removeFirst();
645        if (this.output.isEmpty()) {
646          try {
647            this.prime();
648          } catch (final RuntimeException ise) {
649            this.error = ise;
650            this.state = State.INVALID;
651            this.output.clear();
652          } catch (final IOException io) {
653            this.error = io;
654            this.state = State.INVALID;
655            this.output.clear();
656          }
657        }
658        return returnValue;
659      }
660    
661      @Override
662      public void remove() {
663        throw new UnsupportedOperationException();
664      }
665    
666      @Override
667      public String toString() {
668        return this.reader.toString();
669      }
