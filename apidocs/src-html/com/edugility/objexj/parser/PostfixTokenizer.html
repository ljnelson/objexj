050    /**
051     * A {@link ParsePosition} and an {@link Iterator} of {@link Token}s
052     * that converts a {@link PushbackReader} of a textual objexj pattern
053     * into a series of {@link Token}s in postfix order.
054     *
055     * <p>A {@link PostfixTokenizer} is the lexer component of the
056     * lex-then-parse series of events that go into compiling an objexj
057     * pattern.</p>
058     *
059     * @author <a href="http://about.me/lairdnelson"
060     * target="_parent">Laird Nelson</a>
061     *
062     * @see Parser
063     *
064     * @see <a
065     * href="http://en.wikipedia.org/wiki/Shunting-yard_algorithm">Shunting-yard
066     * algorithm. (2013, March 14). In Wikipedia, The Free
067     * Encyclopedia. Retrieved 20:42, March 14, 2013, from
068     * http://en.wikipedia.org/w/index.php?title=Shunting-yard_algorithm&oldid=544128320</a>
069     */
070    public class PostfixTokenizer extends ParsePosition implements Iterator<Token> {
071    
072      /**
073       * The version of this class for {@link Serializable serialization
074       * purposes}.
075       */
076      private static final long serialVersionUID = 1L;
077    
078      /**
079       * A state a {@link PostfixTokenizer} may be in.
080       *
081       * @author <a href="http://about.me/lairdnelson"
082       * target="_parent">Laird Nelson</a>
083       */
084      private enum State {
085    
086        /**
087         * A {@link PostfixTokenizer.State} indicating the {@link
088         * PostfixTokenizer} is expecting a begin-input marker (usually
089         * {@code ^}).
090         */
091        BEGIN_ATOM, 
092    
093        /**
094         * A {@link PostfixTokenizer.State} indicating the {@link
095         * PostfixTokenizer} is expecting an end-of-input marker (usually
096         * {@code $})
097         */
098        END_ATOM, 
099    
100        /**
101         * A {@link PostfixTokenizer.State} indicating the {@link
102         * PostfixTokenizer} is beginning its lexing operation.
103         */
104        START, 
105    
106        /**
107         * A {@link PostfixTokenizer.State} indicating the {@link
108         * PostfixTokenizer} is expecting either a start-saving marker
109         * (usually {@code (}) or a filter token.
110         */
111        START_SAVING_OR_FILTER, 
112    
113        /**
114         * A {@link PostfixTokenizer.State} indicating the {@link
115         * PostfixTokenizer} is in an invalid state and is essentially
116         * useless.
117         */
118        INVALID, 
119    
120        /**
121         * A {@link PostfixTokenizer.State} indicating the {@link
122         * PostfixTokenizer} is expecting a filter token.
123         */
124        FILTER, 
125    
126        /**
127         * A {@link PostfixTokenizer.State} indicating the {@link
128         * PostfixTokenizer} is expecting an <a
129         * href="http://mvel.codehaus.org/">MVEL</a> expression modifier
130         * to a filter token.
131         */
132        MVEL, 
133    
134        /**
135         * A {@link PostfixTokenizer.State} indicating the {@link
136         * PostfixTokenizer} has just completed parsing a filter token.
137         */
138        END_OF_FILTER,
139    
140        /**
141         * A {@link PostfixTokenizer.State} indicating the {@link
142         * PostfixTokenizer} is expecting an operator token.
143         */
144        OPERATOR, 
145    
146        /**
147         * A {@link PostfixTokenizer.State} indicating the {@link
148         * PostfixTokenizer} is expecting a left parenthesis character (a
149         * token that represents both a begin-saving operator and a
150         * precedence adjuster).
151         */
152        LEFT_PAREN, 
153    
154        /**
155         * A {@link PostfixTokenizer.State} indicating the {@link
156         * PostfixTokenizer} is expecting a right parenthesis character (a
157         * token that represents both a stop-saving operator and a
158         * precedence adjuster).
159         */
160        RIGHT_PAREN, 
161    
162        /**
163         * A {@link PostfixTokenizer.State} indicating the {@link
164         * PostfixTokenizer} could forward to several other states.
165         */
166        STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE, 
167    
168        /**
169         * A {@link PostfixTokenizer.State} indicating the {@link
170         * PostfixTokenizer} is at the end of its lexing.
171         */
172        END
173      }
174    
175      /**
176       * A {@link Map} of {@link Token}s indexed by their character
177       * representations.  The representations are stored as {@link
178       * Integer}s rather than {@link Character}s for convenience since
179       * certain {@link Reader} operations return {@code int}s, not {@code
180       * char}s.
181       *
182       * <p>This field is never {@code null}.</p>
183       *
184       * @see #setupTokens(Map)
185       */
186      private final Map<Integer, Token> tokens;
187    
188      /**
189       * The {@link Logger} used by this {@link PostfixTokenizer} to log messages.
190       *
191       * <p>This field is never {@code null} in normal situations; it
192       * might be {@code null} in some rare serialization edge cases.</p>
193       */
194      private transient Logger logger;
195    
196      /**
197       * The {@link PushbackReader} responsible for reading characters
198       * from a textual objexj pattern.
199       *
200       * <p>This field is never {@code null} in normal situations.  It
201       * might be {@code null} in some rare serialization edge cases.</p>
202       */
203      private transient final PushbackReader reader;
204    
205      /**
206       * A {@link LinkedList} of {@link Token}s that is iterated over by
207       * this {@link PostfixTokenizer}.
208       *
209       * <p>This field is never {@code null} in normal situations.  It
210       * might be {@code null} in some rare serialization edge cases.</p>
211       */
212      final transient LinkedList<Token> output;
213    
214      /**
215       * A {@link Throwable} encountered during iteration.
216       *
217       * <p>This field may be {@code null}.</p>
218       */
219      private transient Throwable error;
220    
221      /**
222       * The {@link PostfixTokenizer.State} this {@link PostfixTokenizer}
223       * is currently in.
224       *
225       * <p>This field may be {@code null}.</p>
226       */
227      private transient State state;
228    
229      /**
230       * The current position of this {@link PostfixTokenizer} within its
231       * associated {@link #reader}.
232       */
233      private transient int position;
234    
235      /**
236       * A {@link Deque} of {@link Token}s used during lexing.
237       *
238       * <p>This field is never {@code null} in normal situations.  It
239       * might be {@code null} in some rare serialization edge cases.</p>
240       */
241      private transient final Deque<Token> stack;
242    
243      /**
244       * Creates a new {@link PostfixTokenizer}.
245       *
246       * @param reader the {@link PushbackReader} that produces characters
247       * to read; must not be {@code null}
248       *
249       * @exception IllegalArgumentException if {@code reader} is {@code
250       * null}
251       *
252       * @exception IOException if the supplied {@link PushbackReader} has
253       * trouble reading
254       *
255       * @exception ParseException if the supplied {@link PushbackReader}
256       * yields an unparseable pattern
257       */
258      public PostfixTokenizer(final PushbackReader reader) throws IOException, ParseException {
259        super(0);
260        this.logger = Logger.getLogger(this.getClass().getName());
261        assert this.logger != null;
262        if (reader == null) {
263          throw new IllegalArgumentException("reader", new NullPointerException("reader"));
264        }
265        this.output = new LinkedList<Token>();
266        this.stack = new ArrayDeque<Token>();
267        this.reader = reader;
268        this.state = State.START;
269        this.tokens = new HashMap<Integer, Token>();
270        this.setupTokens(this.tokens);
271        this.prime();
272      }
273    
274      /**
275       * Returns the position of this {@link PostfixTokenizer} within its
276       * associated {@link PushbackReader}.
277       *
278       * @return the position of this {@link PostfixTokenizer} within its
279       * associated {@link PushbackReader}
280       */
281      @Override
282      public int getIndex() {
283        return this.position;
284      }
285    
286      /**
287       * Sets the position of this {@link PostfixTokenizer} within its
288       * associated {@link PushbackReader}.
289       *
290       * <p>This method does not actually advance the {@link
291       * PushbackReader}; it is simply used to update an internal variable
292       * that keeps track of where the {@link PushbackReader} is within
293       * its stream.</p>
294       *
295       * @param index the new position
296       */
297      @Override
298      public void setIndex(final int index) {
299        this.position = index;
300      }
301    
302      /**
303       * Builds an appropriate exception message for an unexpected
304       * character during lexing.
305       *
306       * @param c the unexpected character
307       *
308       * @return a non-{@code null} error message
309       */
310      private final String buildExceptionMessage(final int c) {
311        return String.format("Unexpected character (%c) at position %d in Reader %s", c, this.getErrorIndex(), this.reader);
312      }
313    
314      /**
315       * Lexes as little as possible to instantiate a {@link Token} so
316       * that {@linkplain #next() iteration} can then take place.
317       *
318       * @exception IOException if a reading error occurs
319       *
320       * @exception ParseException if a non-sensical textual pattern was
321       * encountered
322       */
323      private final void prime() throws IOException, ParseException {
324        if (this.state == State.INVALID) {
325          throw new IllegalStateException();
326        }
327    
328        final StringBuilder sb = new StringBuilder();
329        int parenCount = 0;
330        int c = -1;
331    
332        READ_LOOP:
333        for (; (c = this.read()) != -1; this.setIndex(this.getIndex() + 1)) {
334          switch (this.state) {
335    
336    
337            /*
338             * In this state switch block, "forwarding" states (states
339             * that peek ahead to determine what real state to forward to)
340             * are commented // LIKE THIS, and "real" states are commented
341             * //
342             * // LIKE THIS
343             * //
344             */
345    
346    
347            // START
348          case START:
349            switch (c) {
350    
351            case '^':
352              this.unread(c);
353              this.state = State.BEGIN_ATOM;
354              break;
355    
356            case '(':
357              this.insertBeginAtomZeroOrMoreAndCatenation();
358              this.unread(c);
359              this.state = State.LEFT_PAREN;
360              break;
361    
362            default:
363              if (Character.isWhitespace(c)) {
364                continue READ_LOOP;
365              } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
366                this.insertBeginAtomZeroOrMoreAndCatenation();
367                sb.append((char)c);
368                this.state = State.FILTER;
369              } else {
370                this.setErrorIndex(this.getIndex());
371                throw new ParseException(buildExceptionMessage(c), this.getIndex());
372              }
373              break;
374            }
375            break;
376    
377    
378            //
379            // BEGIN_ATOM
380            //
381          case BEGIN_ATOM:
382            assert c == '^';
383            assert sb.length() == 0;
384            this.output.add(this.tokenFor(c));
385            this.insertCatenation();
386            assert this.state == State.START_SAVING_OR_FILTER;
387            break READ_LOOP;
388    
389    
390            //
391            // END_ATOM
392            //
393          case END_ATOM:
394            assert c == '$';
395            this.insertCatenation();
396            this.output.add(this.tokenFor(c));
397            sb.setLength(0);
398            this.state = State.END;
399            break READ_LOOP;
400    
401    
402            //
403            // LEFT_PAREN
404            //
405          case LEFT_PAREN:
406            assert c == '(';
407            /*
408             * Shunting-yard algorithm: "If the token is a left
409             * parenthesis, then push it onto the stack."
410             *
411             * In our case, a parenthesis is both a grouping operator and
412             * an atom-- a signal to the parser that a START_SAVING
413             * instruction should be fired.  So we also have to add it to
414             * the output.
415             *
416             * TENTATIVE: since it's an atom, we also need to pretend we
417             * found a concatenation operator right afterwards.
418             */
419    
420            // First find a START_SAVING (atom) token.
421            final Token token = new Token(Token.Type.START_SAVING);
422            
423            // TODO: support named capture groups
424    
425            // Add it to the output (it's an atom):
426            this.output.add(token);
427    
428            // Then pretend we found a concatenation.
429            this.insertCatenation();
430    
431            // Then pretend we found a START_GROUP.
432            final Token t = new Token(Token.Type.START_GROUP);
433            assert t != null;
434            this.push(t);
435    
436            this.state = State.START_SAVING_OR_FILTER;
437            break READ_LOOP;
438    
439    
440            //
441            // RIGHT_PAREN
442            //
443          case RIGHT_PAREN:
444            assert c == ')'; // OK
445            /*
446             * Shunting-yard algorithm:
447             * "If the token is a right parenthesis:
448             * " * Until the token at the top of the stack is a left parenthesis, pop operators off the stack onto the output queue.
449             * " * Pop the left parenthesis from the stack, but not onto the output queue.
450             * " * If the token at the top of the stack is a function token, pop it onto the output queue.
451             * " * If the stack runs out without finding a left parenthesis, then there are mismatched parentheses."
452             */
453    
454            final int originalOutputSize = this.output.size();
455            while (!this.stack.isEmpty() && this.stack.peek().getType() != Token.Type.START_GROUP) {
456              this.output.add(this.stack.pop());
457            }
458            if (this.stack.isEmpty()) {
459              this.setErrorIndex(this.getIndex());
460              throw new ParseException("Mismatched parentheses", this.getIndex());
461            }
462    
463            assert Token.Type.START_GROUP == this.stack.peek().getType();
464            this.stack.pop();
465    
466            // The original shunting-yard algorithm discards the parens
467            // because in that algorithm they are used only for grouping.
468            // In regular expressions, they are also atoms.  So they
469            // belong in the output queue. We already added the
470            // START_SAVING token, so now we need to add the STOP_SAVING.
471    
472            // Concatenate...
473            this.insertCatenation();
474    
475            // ...with stop saving:
476            this.output.add(new Token(Token.Type.STOP_SAVING));
477    
478            this.state = State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
479            break READ_LOOP; // only because we added something to the output queue
480    
481    
482            // START_SAVING_OR_FILTER
483          case START_SAVING_OR_FILTER:
484            if (Character.isWhitespace(c)) {
485              continue READ_LOOP;
486            } else if (c == '(') {
487              // Forward to LEFT_PAREN state
488              this.unread(c);
489              this.state = State.LEFT_PAREN;
490            } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
491              // Forward to FILTER state
492              sb.append((char)c);
493              this.state = State.FILTER;
494            } else {
495              this.setErrorIndex(this.getIndex());
496              throw new ParseException(buildExceptionMessage(c), this.getIndex());
497            }
498            break;
499    
500    
501            // FILTER (really more like START_OF_FILTER?)
502          case FILTER:
503            switch (c) {
504    
505            case '(':
506              // Bumped into an MVEL guard.  Create the token, but don't
507              // return it yet.  Fill in its filterType.  Transition to
508              // the MVEL state and have it fill in the details.
509              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
510              parenCount = 1;
511              sb.setLength(0);
512              this.state = State.MVEL;
513              break;
514    
515            case ')': // un-MVELed atom; just ran into a RIGHT_PAREN; OK
516              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
517              sb.setLength(0);
518              this.unread(c);
519              this.state = State.RIGHT_PAREN;
520              break READ_LOOP;
521    
522            case '+':
523            case '*':
524            case '?':
525            case '|':
526            case '/':
527            case ',':
528              // Bumped into a genuine operator.  So create a FILTER token
529              // without an accompanying MVEL guard, and forward to the
530              // OPERATOR state.
531              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
532              sb.setLength(0);
533              this.unread(c);
534              this.state = State.OPERATOR;
535              break READ_LOOP;
536    
537            case '$':
538              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
539              sb.setLength(0);
540              this.unread(c);
541              this.state = State.END_ATOM;
542              break READ_LOOP;
543    
544            default:
545              if (Character.isJavaIdentifierPart(c)) {
546                sb.append((char)c);
547              } else if (Character.isWhitespace(c)) {
548                continue READ_LOOP;
549              } else if (c == '.') {
550                sb.append('.');
551                c = this.read();
552                if (c == -1) {
553                  this.unread(c);
554                } else if (Character.isJavaIdentifierStart(c)) {
555                  sb.append((char)c);
556                } else {
557                  this.setErrorIndex(this.getIndex());
558                  throw new ParseException(buildExceptionMessage(c), this.getIndex());
559                }
560              } else {
561                this.setErrorIndex(this.getIndex());
562                throw new ParseException(buildExceptionMessage(c), this.getIndex());
563              }
564              break;
565            }
566            break;
567    
568    
569            // MVEL
570          case MVEL:
571            assert parenCount > 0;
572            switch (c) {
573    
574            case '(':
575              parenCount++;
576              sb.append((char)c);
577              break;
578    
579            case ')': // Maybe a paren inside an MVEL block; maybe the paren that closes it; OK
580              parenCount--;
581              if (parenCount == 0) {
582                final Token filter = this.output.getLast();
583                assert filter != null;
584                assert filter.getFilterType() != null;
585                assert filter.getType() == Token.Type.FILTER;
586                filter.setValue(sb.toString());
587                sb.setLength(0);
588                this.state = State.END_OF_FILTER;
589                break READ_LOOP;
590              } else if (parenCount < 0) {
591                this.setErrorIndex(this.getIndex());
592                throw new ParseException("Mismatched parentheses", this.getIndex());
593              } else {
594                sb.append((char)c);
595              }
596              break;
597    
598            default:
599              sb.append((char)c);
600              break;
601            }
602            break;
603    
604    
605            // END_OF_FILTER
606          case END_OF_FILTER:
607            switch (c) {
608    
609            case '$':
610              sb.setLength(0);
611              this.unread(c);
612              this.state = State.END_ATOM;
613              break;
614    
615            case ')': // OK
616              sb.setLength(0);
617              this.unread(c);
618              this.state = State.RIGHT_PAREN;
619              break;
620    
621            case '+':
622            case '*':
623            case '?':
624            case '|':
625            case '/':
626            case ',':
627              // Forward to OPERATOR state
628              sb.setLength(0);
629              this.unread(c);
630              this.state = State.OPERATOR;
631              break;
632    
633            default:
634              if (Character.isWhitespace(c)) {
635                continue READ_LOOP;
636              } else {
637                this.setErrorIndex(this.getIndex());
638                throw new ParseException(buildExceptionMessage(c), this.getIndex());
639              }
640    
641            }
642            break;
643    
644    
645            //
646            // OPERATOR
647            //
648          case OPERATOR:
649            sb.setLength(0);
650            if (Character.isWhitespace(c)) {
651              continue READ_LOOP;
652            } else if (this.handleOperator(this.tokenFor(c))) {
653              break READ_LOOP;
654            }
655            break;
656    
657    
658            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE
659            //
660            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE permits
661            // only those operators that come after a (potentially
662            // augmented) filter token, or a group of such tokens, and
663            // weeds out others before proceeding to the OPERATOR state.
664            //
665            // This handles closing of submatches (")") and end-of-input
666            // ("$") as special (legal) cases.  Otherwise the operator has
667            // to be catenation or alternation.
668            //
669            // In these cases we forward to the OPERATOR state.
670            //
671            // Any other character causes an ParseException.
672          case STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE:
673            switch (c) {
674    
675            case ')': // stop saving
676              this.unread(c);
677              this.state = State.RIGHT_PAREN;
678              break;
679    
680            case '$': // end of input
681              this.unread(c);
682              this.state = State.END_ATOM;
683              break;
684    
685            case '/': // catenation
686            case ',': // catenation
687            case '|': // alternation
688              this.unread(c);
689              this.state = State.OPERATOR;
690              break;
691    
692            default:
693              if (Character.isWhitespace(c)) {
694                continue READ_LOOP;
695              } else {
696                this.setErrorIndex(this.getIndex());
697                throw new ParseException(buildExceptionMessage(c), this.getIndex());
698              }
699            }
700            break;
701    
702    
703            //
704            // END
705            //
706          case END:
707            if (!Character.isWhitespace(c)) {
708              this.setErrorIndex(this.getIndex());
709              throw new ParseException(buildExceptionMessage(c), this.getIndex());
710            }
711            break;
712    
713            //
714            // DEFAULT
715            //
716          default:
717            this.setErrorIndex(this.getIndex());
718            throw new ParseException("Unexpected state: " + this.state, this.getIndex());
719          }
720        } // READ_LOOP
721    
722        if (c == -1) {
723          switch (this.state) {
724          case FILTER:
725            this.output.add(new Token(Token.Type.FILTER, sb.toString()));
726            sb.setLength(0);
727            break;
728          case END:
729            break;
730          default:
731            if (this.logger.isLoggable(Level.WARNING)) {
732              this.logger.logp(Level.WARNING, this.getClass().getName(), "prime", "Unhandled cleanup in state {0} with stringbuffer {1}", new Object[] { state, sb });
733            }
734            break;
735          }
736          this.state = State.END;
737          while (!this.stack.isEmpty()) {
738            final Token top = this.stack.peek();
739            assert top != null;
740            final Token.Type type = top.getType();
741            assert type != null;
742    
743            if (type == Token.Type.START_GROUP || type == Token.Type.STOP_GROUP) {
744              this.setErrorIndex(this.getIndex());
745              throw new ParseException("Mismatched parentheses", this.getIndex());
746            }
747            this.output.add(this.stack.pop());
748          }
749        }
750    
751      }

758      private final void insertBeginAtomZeroOrMoreAndCatenation() {

760        this.insertCatenation();

779        return this.handleOperator(this.tokenFor('/'));

782      /**
783       * Processes the supplied operator {@link Token} according to the <a
784       * href="http://en.wikipedia.org/wiki/Shunting-yard_algorithm">shunting-yard
785       * algorithm</a>.
786       *
787       * @param o1 the operator {@link Token}; must not be {@code null}
788       *
789       * @exception IllegalArgumentException if {@code o1} is {@code null}
790       * or if it is {@linkplain Token#isOperator() not an operator}
791       */
792      private final boolean handleOperator(final Token o1) {
793        if (o1 == null) {
794          throw new IllegalArgumentException("o1", new NullPointerException("o1"));
795        } else if (!o1.isOperator()) {
796          throw new IllegalArgumentException("!o1.isOperator()");
797        }
798        Token o2;
799        boolean added = false;
800        while (!this.stack.isEmpty() && this.assertIsOperator(o2 = this.stack.peek())) {
801          if (o1.compareTo(o2) <= 0) {
802            added = this.output.add(this.stack.pop()) || added;
803          } else {
804            break;
805          }
806        }
807        this.push(o1);
808        this.state = this.stateAfterOperator(o1);
809        return added;
810      }

812      /**
813       * Pushes the supplied {@link Token} on the {@linkplain #stack
814       * stack}, after checking for error conditions.
815       *
816       * @param token the {@link Token} to push; must not be {@code null}
817       *
818       * @exception IllegalArgumentException if {@code token} is {@code
819       * null} or is inappropriate for pushing
820       */
821      private final void push(final Token token) {
822        if (this.logger.isLoggable(Level.FINER)) {
823          this.logger.entering(this.getClass().getName(), "push", token);
824        }
825        if (token == null) {
826          throw new IllegalArgumentException("token", new NullPointerException("token"));
827        } else if (Token.Type.STOP_GROUP == token.getType()) {
828          throw new IllegalArgumentException("token.getType() == Token.Type.STOP_GROUP");
829        }
830        this.stack.push(token);
831        if (this.logger.isLoggable(Level.FINER)) {
832          this.logger.exiting(this.getClass().getName(), "push");
833        }
834      }

836      /**
837       * Returns the appropriate {@link PostfixTokenizer.State} this
838       * {@link PostfixTokenizer} should transition to after processing
839       * the supplied {@link Token}.
840       *
841       * <p>This method never returns {@code null}.</p>
842       *
843       * <p>The {@code token} parameter must be non-{@code null} and must
844       * {@linkplain Token#getType() have a <code>Type</code>} that is
845       * equal to one of the following:</p>
846       *
847       * <ul>
848       *
849       * <li>{@link Token.Type#ONE_OR_MORE}</li>
850       *
851       * <li>{@link Token.Type#ZERO_OR_MORE}</li>
852       *
853       * <li>{@link Token.Type#ZERO_OR_ONE}</li>
854       *
855       * <li>{@link Token.Type#ALTERNATION}</li>
856       *
857       * <li>{@link Token.Type#CATENATION}</li>
858       *
859       * </ul>
860       *
861       * @param token the {@link Token} to process; must not be {@code null}
862       *
863       * @return a non-{@code null} {@link PostfixTokenizer.State}
864       *
865       * @exception IllegalArgumentException if {@code token} does not
866       * meet the conditions described earlier
867       */
868      private final State stateAfterOperator(final Token token) {
869        if (token == null) {
870          throw new IllegalArgumentException("token", new NullPointerException("token"));
871        }
872        final Token.Type type = token.getType();
873        assert type != null;
874        switch (type) {
875        case ONE_OR_MORE:
876        case ZERO_OR_MORE:
877        case ZERO_OR_ONE:
878          return State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
879        case ALTERNATION:
880        case CATENATION:
881          return State.START_SAVING_OR_FILTER;
882        default:
883          throw new IllegalArgumentException("token: " + token);
884        }
885      }

887      /**
888       * Manipulates the supplied {@link Map} to contain {@link Token}s
889       * indexed under the characters (expressed as {@link Integer}s) that
890       * identify them.
891       *
892       * @param tokens the {@link Map} of {@link Token}s to manipulate;
893       * must not be {@code null}
894       *
895       * @exception IllegalArgumentException if {@code tokens} is {@code
896       * null}
897       */
898      private final void setupTokens(final Map<Integer, Token> tokens) {
899        if (tokens == null) {
900          throw new IllegalArgumentException("tokens", new NullPointerException("tokens"));
901        }
902        tokens.clear();
903        tokens.put(Integer.valueOf('^'), new Token(Token.Type.BEGIN_ATOM));
904        tokens.put(Integer.valueOf('$'), new Token(Token.Type.END_ATOM));
905        tokens.put(Integer.valueOf('+'), new Token(Token.Type.ONE_OR_MORE));
906        tokens.put(Integer.valueOf('*'), new Token(Token.Type.ZERO_OR_MORE));
907        tokens.put(Integer.valueOf('?'), new Token(Token.Type.ZERO_OR_ONE));
908        tokens.put(Integer.valueOf('|'), new Token(Token.Type.ALTERNATION));
909        final Token catenation = new Token(Token.Type.CATENATION);
910        tokens.put(Integer.valueOf('/'), catenation);
911        tokens.put(Integer.valueOf(','), catenation);
912        tokens.put(Integer.valueOf('('), new Token(Token.Type.START_SAVING));
913        tokens.put(Integer.valueOf(')'), new Token(Token.Type.STOP_SAVING));
914      }

917       * Returns a {@link Token} suitable for the supplied character.
918       *
919       * <p>This method never returns {@code null}.</p>
920       *
921       * @param c a character known to identify a {@link Token}
922       *
923       * @return a suitable {@link Token}; never {@code null}
924       *
925       * @exception IllegalArgumentException if {@code c} does not pick
926       * out a {@link Token}
927       */
928      private final Token tokenFor(final int c) {
929        final Token token = this.tokens.get(Integer.valueOf(c));
930        if (token == null) {
931          throw new IllegalArgumentException(String.format("c: %c", (char)c));
932        }
933        return token;
934      }

937       * Asserts that the supplied {@link Token} {@linkplain
938       * Token#isOperator() is an operator}.  Returns {@code true} if this
939       * is true; throws an {@link IllegalArgumentException} if this is
940       * not.
941       *
942       * @return {@code true} when invoked
943       *
944       * @exception IllegalArgumentException if {@code token} is {@code

949        if (token == null) {
950          throw new IllegalArgumentException("token", new NullPointerException("token"));

952        if (!token.isOperator()) {
953          throw new IllegalArgumentException("!token.isOperator(): " + token);

955        return true;

990       * Returns {@code true} if this {@link PostfixTokenizer} has more
991       * {@link Token}s to iterate over.
992       *
993       * @return {@code true} if this {@link PostfixTokenizer} has more
994       * {@link Token}s to iterate over
995       *
996       * @exception IllegalStateException if a prior invocation of {@link
997       * #next()} caused an error; see {@linkplain Throwable#getCause()
998       * its cause} for the root problem
999       */
1000      @Override
1001      public boolean hasNext() {
1002        if (this.error != null) {
1003          throw new IllegalStateException(this.error);
1004        }
1005        return !this.output.isEmpty();
1006      }

1008      /**
1009       * Returns the next {@link Token} lexed from this {@link
1010       * PostfixTokenizer}'s underlying {@link Reader}.
1011       *
1012       * @return a non-{@code null} {@link Token}
1013       *
1014       * @exception NoSuchElementException if the next element could not
1015       * be produced
1016       */
1017      @Override
1018      public Token next() {
1019        if (this.error != null) {
1020          throw (NoSuchElementException)new NoSuchElementException().initCause(this.error);
1021        } else if (this.output.isEmpty()) {
1022          throw new NoSuchElementException();
1023        }
1024        final Token returnValue = this.output.removeFirst();
1025        if (this.output.isEmpty()) {
1026          try {
1027            this.prime();
1028          } catch (final Exception other) {
1029            this.error = other;
1030            this.state = State.INVALID;
1031            this.output.clear();
1032          }
1033        }
1034        return returnValue;
1035      }
