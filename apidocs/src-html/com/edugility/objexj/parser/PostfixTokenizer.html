049    public class PostfixTokenizer extends ParsePosition implements Iterator<Token> {

053      private enum State {
054        BEGIN_ATOM, END_ATOM, START, START_SAVING_OR_FILTER, INVALID, FILTER, MVEL, END_OF_FILTER, OPERATOR, LEFT_PAREN, RIGHT_PAREN, STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE, END
055      }

073      public PostfixTokenizer(final PushbackReader reader) throws IOException, ParseException {
074        super(0);
075        this.logger = Logger.getLogger(this.getClass().getName());
076        assert this.logger != null;
077        if (reader == null) {
078          throw new IllegalArgumentException("reader", new NullPointerException("reader"));
079        }
080        this.tokens = new HashMap<Integer, Token>();
081        this.setupTokens();
082        this.output = new LinkedList<Token>();
083        this.stack = new ArrayDeque<Token>();
084        this.reader = reader;
085        this.state = State.START;
086        this.prime();
087      }

099      private final String buildExceptionMessage(final int c) {
100        return String.format("Unexpected character (%c) at position %d in Reader %s", c, this.getErrorIndex(), this.reader);
101      }
102    
103      private final void prime() throws IOException, ParseException {
104        if (this.state == State.INVALID) {
105          throw new IllegalStateException();
106        }
107    
108        final StringBuilder sb = new StringBuilder();
109        int parenCount = 0;
110        int c = -1;
111    
112        READ_LOOP:
113        for (; (c = this.read()) != -1; this.setIndex(this.getIndex() + 1)) {
114          switch (this.state) {
115    
116    
117            /*
118             * In this state switch block, "forwarding" states (states
119             * that peek ahead to determine what real state to forward to)
120             * are commented // LIKE THIS, and "real" states are commented
121             * //
122             * // LIKE THIS
123             * //
124             */
125    
126    
127            // START
128          case START:
129            switch (c) {
130    
131            case '^':
132              this.unread(c);
133              this.state = State.BEGIN_ATOM;
134              break;
135    
136            case '(':
137              this.insertBeginAtomZeroOrMoreAndCatenation();
138              this.unread(c);
139              this.state = State.LEFT_PAREN;
140              break;
141    
142            default:
143              if (Character.isWhitespace(c)) {
144                continue READ_LOOP;
145              } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
146                this.insertBeginAtomZeroOrMoreAndCatenation();
147                sb.append((char)c);
148                this.state = State.FILTER;
149              } else {
150                this.setErrorIndex(this.getIndex());
151                throw new ParseException(buildExceptionMessage(c), this.getIndex());
152              }
153              break;
154            }
155            break;
156    
157    
158            //
159            // BEGIN_ATOM
160            //
161          case BEGIN_ATOM:
162            assert c == '^';
163            assert sb.length() == 0;
164            this.output.add(this.tokenFor(c));
165            this.insertCatenation();
166            assert this.state == State.START_SAVING_OR_FILTER;
167            break READ_LOOP;
168    
169    
170            //
171            // END_ATOM
172            //
173          case END_ATOM:
174            assert c == '$';
175            this.insertCatenation();
176            this.output.add(this.tokenFor(c));
177            sb.setLength(0);
178            this.state = State.END;
179            break READ_LOOP;
180    
181    
182            //
183            // LEFT_PAREN
184            //
185          case LEFT_PAREN:
186            assert c == '(';
187            /*
188             * Shunting-yard algorithm: "If the token is a left
189             * parenthesis, then push it onto the stack."
190             *
191             * In our case, a parenthesis is both a grouping operator and
192             * an atom-- a signal to the parser that a START_SAVING
193             * instruction should be fired.  So we also have to add it to
194             * the output.
195             *
196             * TENTATIVE: since it's an atom, we also need to pretend we
197             * found a concatenation operator right afterwards.
198             */
199    
200            // First find a START_SAVING (atom) token.
201            final Token token = new Token(Token.Type.START_SAVING);
202            
203            // TODO: support named capture groups
204    
205            // Add it to the output (it's an atom):
206            this.output.add(token);
207    
208            // Then pretend we found a concatenation.
209            this.insertCatenation();
210    
211            // Then pretend we found a START_GROUP.
212            final Token t = new Token(Token.Type.START_GROUP);
213            assert t != null;
214            this.push(t);
215    
216            this.state = State.START_SAVING_OR_FILTER;
217            break READ_LOOP;
218    
219    
220            //
221            // RIGHT_PAREN
222            //
223          case RIGHT_PAREN:
224            assert c == ')'; // OK
225            /*
226             * Shunting-yard algorithm:
227             * "If the token is a right parenthesis:
228             * " * Until the token at the top of the stack is a left parenthesis, pop operators off the stack onto the output queue.
229             * " * Pop the left parenthesis from the stack, but not onto the output queue.
230             * " * If the token at the top of the stack is a function token, pop it onto the output queue.
231             * " * If the stack runs out without finding a left parenthesis, then there are mismatched parentheses."
232             */
233    
234            final int originalOutputSize = this.output.size();
235            while (!this.stack.isEmpty() && this.stack.peek().getType() != Token.Type.START_GROUP) {
236              this.output.add(this.stack.pop());
237            }
238            if (this.stack.isEmpty()) {
239              this.setErrorIndex(this.getIndex());
240              throw new ParseException("Mismatched parentheses", this.getIndex());
241            }
242    
243            assert Token.Type.START_GROUP == this.stack.peek().getType();
244            this.stack.pop();
245    
246            // The original shunting-yard algorithm discards the parens
247            // because in that algorithm they are used only for grouping.
248            // In regular expressions, they are also atoms.  So they
249            // belong in the output queue. We already added the
250            // START_SAVING token, so now we need to add the STOP_SAVING.
251    
252            // Concatenate...
253            this.insertCatenation();
254    
255            // ...with stop saving:
256            this.output.add(new Token(Token.Type.STOP_SAVING));
257    
258            this.state = State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
259            break READ_LOOP; // only because we added something to the output queue
260    
261    
262            // START_SAVING_OR_FILTER
263          case START_SAVING_OR_FILTER:
264            if (Character.isWhitespace(c)) {
265              continue READ_LOOP;
266            } else if (c == '(') {
267              // Forward to LEFT_PAREN state
268              this.unread(c);
269              this.state = State.LEFT_PAREN;
270            } else if (c != '$' && (c == '=' || Character.isJavaIdentifierStart(c))) {
271              // Forward to FILTER state
272              sb.append((char)c);
273              this.state = State.FILTER;
274            } else {
275              this.setErrorIndex(this.getIndex());
276              throw new ParseException(buildExceptionMessage(c), this.getIndex());
277            }
278            break;
279    
280    
281            // FILTER (really more like START_OF_FILTER?)
282          case FILTER:
283            switch (c) {
284    
285            case '(':
286              // Bumped into an MVEL guard.  Create the token, but don't
287              // return it yet.  Fill in its filterType.  Transition to
288              // the MVEL state and have it fill in the details.
289              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
290              parenCount = 1;
291              sb.setLength(0);
292              this.state = State.MVEL;
293              break;
294    
295            case ')': // un-MVELed atom; just ran into a RIGHT_PAREN; OK
296              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
297              sb.setLength(0);
298              this.unread(c);
299              this.state = State.RIGHT_PAREN;
300              break READ_LOOP;
301    
302            case '+':
303            case '*':
304            case '?':
305            case '|':
306            case '/':
307            case ',':
308              // Bumped into a genuine operator.  So create a FILTER token
309              // without an accompanying MVEL guard, and forward to the
310              // OPERATOR state.
311              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
312              sb.setLength(0);
313              this.unread(c);
314              this.state = State.OPERATOR;
315              break READ_LOOP;
316    
317            case '$':
318              this.output.add(new Token(Token.Type.FILTER, sb.toString()));
319              sb.setLength(0);
320              this.unread(c);
321              this.state = State.END_ATOM;
322              break READ_LOOP;
323    
324            default:
325              if (Character.isJavaIdentifierPart(c)) {
326                sb.append((char)c);
327              } else if (Character.isWhitespace(c)) {
328                continue READ_LOOP;
329              } else if (c == '.') {
330                sb.append('.');
331                c = this.read();
332                if (c == -1) {
333                  this.unread(c);
334                } else if (Character.isJavaIdentifierStart(c)) {
335                  sb.append((char)c);
336                } else {
337                  this.setErrorIndex(this.getIndex());
338                  throw new ParseException(buildExceptionMessage(c), this.getIndex());
339                }
340              } else {
341                this.setErrorIndex(this.getIndex());
342                throw new ParseException(buildExceptionMessage(c), this.getIndex());
343              }
344              break;
345            }
346            break;
347    
348    
349            // MVEL
350          case MVEL:
351            assert parenCount > 0;
352            switch (c) {
353    
354            case '(':
355              parenCount++;
356              sb.append((char)c);
357              break;
358    
359            case ')': // Maybe a paren inside an MVEL block; maybe the paren that closes it; OK
360              parenCount--;
361              if (parenCount == 0) {
362                final Token filter = this.output.getLast();
363                assert filter != null;
364                assert filter.getFilterType() != null;
365                assert filter.getType() == Token.Type.FILTER;
366                filter.setValue(sb.toString());
367                sb.setLength(0);
368                this.state = State.END_OF_FILTER;
369                break READ_LOOP;
370              } else if (parenCount < 0) {
371                this.setErrorIndex(this.getIndex());
372                throw new ParseException("Mismatched parentheses", this.getIndex());
373              } else {
374                sb.append((char)c);
375              }
376              break;
377    
378            default:
379              sb.append((char)c);
380              break;
381            }
382            break;
383    
384    
385            // END_OF_FILTER
386          case END_OF_FILTER:
387            switch (c) {
388    
389            case '$':
390              sb.setLength(0);
391              this.unread(c);
392              this.state = State.END_ATOM;
393              break;
394    
395            case ')': // OK
396              sb.setLength(0);
397              this.unread(c);
398              this.state = State.RIGHT_PAREN;
399              break;
400    
401            case '+':
402            case '*':
403            case '?':
404            case '|':
405            case '/':
406            case ',':
407              // Forward to OPERATOR state
408              sb.setLength(0);
409              this.unread(c);
410              this.state = State.OPERATOR;
411              break;
412    
413            default:
414              if (Character.isWhitespace(c)) {
415                continue READ_LOOP;
416              } else {
417                this.setErrorIndex(this.getIndex());
418                throw new ParseException(buildExceptionMessage(c), this.getIndex());
419              }
420    
421            }
422            break;
423    
424    
425            //
426            // OPERATOR
427            //
428          case OPERATOR:
429            sb.setLength(0);
430            if (Character.isWhitespace(c)) {
431              continue READ_LOOP;
432            } else if (this.handleOperator(this.tokenFor(c))) {
433              break READ_LOOP;
434            }
435            break;
436    
437    
438            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE
439            //
440            // STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE permits
441            // only those operators that come after a (potentially
442            // augmented) filter token, or a group of such tokens, and
443            // weeds out others before proceeding to the OPERATOR state.
444            //
445            // This handles closing of submatches (")") and end-of-input
446            // ("$") as special (legal) cases.  Otherwise the operator has
447            // to be catenation or alternation.
448            //
449            // In these cases we forward to the OPERATOR state.
450            //
451            // Any other character causes an ParseException.
452          case STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE:
453            switch (c) {
454    
455            case ')': // stop saving
456              this.unread(c);
457              this.state = State.RIGHT_PAREN;
458              break;
459    
460            case '$': // end of input
461              this.unread(c);
462              this.state = State.END_ATOM;
463              break;
464    
465            case '/': // catenation
466            case ',': // catenation
467            case '|': // alternation
468              this.unread(c);
469              this.state = State.OPERATOR;
470              break;
471    
472            default:
473              if (Character.isWhitespace(c)) {
474                continue READ_LOOP;
475              } else {
476                this.setErrorIndex(this.getIndex());
477                throw new ParseException(buildExceptionMessage(c), this.getIndex());
478              }
479            }
480            break;
481    
482    
483            //
484            // END
485            //
486          case END:
487            if (!Character.isWhitespace(c)) {
488              this.setErrorIndex(this.getIndex());
489              throw new ParseException(buildExceptionMessage(c), this.getIndex());
490            }
491            break;
492    
493            //
494            // DEFAULT
495            //
496          default:
497            this.setErrorIndex(this.getIndex());
498            throw new ParseException("Unexpected state: " + this.state, this.getIndex());
499          }
500        } // READ_LOOP
501    
502        if (c == -1) {
503          switch (this.state) {
504          case FILTER:
505            this.output.add(new Token(Token.Type.FILTER, sb.toString()));
506            sb.setLength(0);
507            break;
508          case END:
509            break;
510          default:
511            if (this.logger.isLoggable(Level.WARNING)) {
512              this.logger.logp(Level.WARNING, this.getClass().getName(), "prime", "Unhandled cleanup in state {0} with stringbuffer {1}", new Object[] { state, sb });
513            }
514            break;
515          }
516          this.state = State.END;
517          while (!this.stack.isEmpty()) {
518            final Token top = this.stack.peek();
519            assert top != null;
520            final Token.Type type = top.getType();
521            assert type != null;
522    
523            if (type == Token.Type.START_GROUP || type == Token.Type.STOP_GROUP) {
524              this.setErrorIndex(this.getIndex());
525              throw new ParseException("Mismatched parentheses", this.getIndex());
526            }
527            this.output.add(this.stack.pop());
528          }
529        }
530    
531      }

533      private final Token createFilterToken(final StringBuilder sb) {
534        if (sb == null) {
535          throw new IllegalArgumentException("sb", new NullPointerException("sb"));
536        }
537        return new Token(Token.Type.FILTER, sb.toString());
538      }
539    
540      private final void insertBeginAtomZeroOrMoreAndCatenation() {
541        this.output.add(this.tokenFor('^'));
542        this.insertCatenation();
543        this.insertZeroOrMoreAnything();
544        this.insertCatenation();
545      }
546    
547      private final void insertZeroOrMoreAnything() {
548        this.output.add(new Token(Token.Type.FILTER, "java.lang.Object"));
549        this.handleOperator(this.tokenFor('*'));
550      }
551    
552      private final boolean insertCatenation() {
553        return this.handleOperator(this.tokenFor('/'));
554      }
555    
556      private final boolean handleOperator(final Token o1) {
557        if (o1 == null) {
558          throw new IllegalArgumentException("o1", new NullPointerException("o1"));
559        } else if (!o1.isOperator()) {
560          throw new IllegalArgumentException("!o1.isOperator()");
561        }
562        Token o2;
563        boolean added = false;
564        while (!this.stack.isEmpty() && this.assertIsOperator(o2 = this.stack.peek())) {
565          if (o1.compareTo(o2) <= 0) {
566            added = this.output.add(this.stack.pop()) || added;
567          } else {
568            break;
569          }
570        }
571        this.push(o1);
572        this.state = this.stateAfterOperator(o1);
573        return added;
574      }
575    
576      private final void push(final Token token) {
577        if (this.logger.isLoggable(Level.FINER)) {
578          this.logger.entering(this.getClass().getName(), "push", token);
579        }
580        if (token == null) {
581          throw new IllegalArgumentException("token", new NullPointerException("token"));
582        } else if (Token.Type.STOP_GROUP == token.getType()) {
583          throw new IllegalArgumentException("token.getType() == Token.Type.STOP_GROUP");
584        }
585        this.stack.push(token);
586        if (this.logger.isLoggable(Level.FINER)) {
587          this.logger.exiting(this.getClass().getName(), "push");
588        }
589      }
590    
591      private final State stateAfterOperator(final Token token) {
592        if (token == null) {
593          throw new IllegalArgumentException("token", new NullPointerException("token"));
594        }
595        final Token.Type type = token.getType();
596        assert type != null;
597        switch (type) {
598        case ONE_OR_MORE:
599        case ZERO_OR_MORE:
600        case ZERO_OR_ONE:
601          return State.STOP_SAVING_OR_END_OF_INPUT_OR_NEXT_IN_SEQUENCE;
602        case ALTERNATION:
603        case CATENATION:
604          return State.START_SAVING_OR_FILTER;
605        default:
606          throw new IllegalArgumentException("token: " + token);
607        }
608      }
609    
610      private final void setupTokens() {
611        this.tokens.clear();
612        this.tokens.put(Integer.valueOf('^'), new Token(Token.Type.BEGIN_ATOM));
613        this.tokens.put(Integer.valueOf('$'), new Token(Token.Type.END_ATOM));
614        this.tokens.put(Integer.valueOf('+'), new Token(Token.Type.ONE_OR_MORE));
615        this.tokens.put(Integer.valueOf('*'), new Token(Token.Type.ZERO_OR_MORE));
616        this.tokens.put(Integer.valueOf('?'), new Token(Token.Type.ZERO_OR_ONE));
617        this.tokens.put(Integer.valueOf('|'), new Token(Token.Type.ALTERNATION));
618        final Token catenation = new Token(Token.Type.CATENATION);
619        this.tokens.put(Integer.valueOf('/'), catenation);
620        this.tokens.put(Integer.valueOf(','), catenation);
621        this.tokens.put(Integer.valueOf('('), new Token(Token.Type.START_SAVING));
622        this.tokens.put(Integer.valueOf(')'), new Token(Token.Type.STOP_SAVING));
623      }
624    
625      private final Token tokenFor(final int c) {
626        final Token token = this.tokens.get(Integer.valueOf(c));
627        if (token == null) {
628          throw new IllegalArgumentException("c: " + (char)c);
629        }
630        return token;
631      }

633      /**
634       * Asserts that the supplied {@link Token} {@linkplain
635       * Token#isOperator() is an operator}.  Returns {@code true} if this
636       * is true; throws an {@link IllegalArgumentException} if this is
637       * not.
638       *
639       * @return {@code true} when invoked
640       *
641       * @exception IllegalArgumentException if {@code token} is {@code
642       * null} or if its {@link Token#isOperator() isOperator()} method
643       * returns {@code false}
644       */
645      private final boolean assertIsOperator(final Token token) {
646        if (token == null) {
647          throw new IllegalArgumentException("token", new NullPointerException("token"));
648        }
649        if (!token.isOperator()) {
650          throw new IllegalArgumentException("!token.isOperator(): " + token);
651        }
652        return true;
653      }
654    
655      private final int read() throws IOException {
656        final int c = this.reader.read();
657        this.setIndex(this.getIndex() + 1);
658        return c;
659      }
660    
661      private final int unread(final int c) throws IOException {
662        this.reader.unread(c);
663        final int newPosition = this.getIndex() - 1;
664        this.setIndex(newPosition);
665        return newPosition;
666      }

668      /**
669       * @exception IllegalStateException if a prior invocation of {@link
670       * #next()} caused an error; see {@linkplain Throwable#getCause()
671       * its cause} for the root problem
672       */
673      @Override
674      public boolean hasNext() {
675        if (this.error != null) {
676          throw new IllegalStateException(this.error);
677        }
678        return !this.output.isEmpty();
679      }
680    
681      @Override
682      public Token next() {
683        if (this.error != null) {
684          throw (NoSuchElementException)new NoSuchElementException().initCause(this.error);
685        } else if (this.output.isEmpty()) {
686          throw new NoSuchElementException();
687        }
688        final Token returnValue = this.output.removeFirst();
689        if (this.output.isEmpty()) {
690          try {
691            this.prime();
692          } catch (final Exception other) {
693            this.error = other;
694            this.state = State.INVALID;
695            this.output.clear();
696          }
697        }
698        return returnValue;
699      }

701      @Override
702      public void remove() {
703        throw new UnsupportedOperationException();
704      }
705    
706      @Override
707      public String toString() {
708        return this.reader.toString();
709      }
