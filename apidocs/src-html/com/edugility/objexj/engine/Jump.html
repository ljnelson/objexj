028    package com.edugility.objexj.engine;
029    
030    import java.util.List;
031    
032    import java.util.regex.Matcher;
033    import java.util.regex.Pattern;
034    
035    /**
036     * An {@link Instruction} that causes an {@link InstructionContext} to
037     * {@linkplain InstructionContext#jump(int, boolean) jump} to a new
038     * location in a {@link Program}.
039     *
040     * @param <T> the type of {@link Object} managed by {@link
041     * InstructionContext}s passed to the {@link
042     * #execute(InstructionContext)} method
043     *
044     * @author <a href="http://about.me/lairdnelson">
045     * target="_parent">Laird Nelson</a>
046     *
047     * @see InstructionContext#jump(int, boolean)
048     */
049    public class Jump<T> extends Instruction<T> {
050    
051      /**
052       * The version of this class for {@linkplain Serializable
053       * serialization purposes}.
054       */
055      private static final long serialVersionUID = 1L;
056    
057      /**
058       * A {@link Pattern} for parsing a single {@link String} into
059       * operands indicating a program location and a {@code boolean}
060       * indicating whether that program location is relative to the
061       * current location.
062       *
063       * <p>This field is never {@code null}.</p>
064       */
065      private static final Pattern OPERAND_PATTERN = Pattern.compile("^\\s*(\\+?)(\\d+)");
066    
067      /**
068       * Whether or not the {@link #programLocation} field is to be
069       * interpreted relative to the current program location.
070       */
071      public final boolean relative;
072    
073      /**
074       * The new location within a {@link Program} that an {@link
075       * InstructionContext} should {@linkplain
076       * InstructionContext#jump(int, boolean) jump to}.
077       */
078      public final int programLocation;
079      
080      /**
081       * Creates a new {@link Jump} instruction.
082       *
083       * @param operand a {@link String} containing the operands for this
084       * {@link Jump} instruction; must not be {@code null}
085       *
086       * @exception IllegalArgumentException if the supplied {@code
087       * operand} {@link String} could not be parsed
088       */
089      public Jump(final String operand) {
090        super();
091        if (operand == null) {
092          throw new IllegalArgumentException("operand", new NullPointerException("operand"));
093        }
094        final Matcher m = OPERAND_PATTERN.matcher(operand);
095        if (m.find()) {
096          this.programLocation = Integer.parseInt(m.group(2));
097          this.relative = "+".equals(m.group(1)) || this.programLocation < 0;
098        } else {
099          throw new IllegalArgumentException("Bad operand: " + operand);
100        }
101      }
102    
103      /**
104       * Creates a new {@link Jump} instruction.
105       *
106       * @param programLocation the new location within a {@link Program}
107       * to {@linkplain InstructionContext#jump(int, boolean) jump to}
108       */
109      public Jump(final int programLocation) {
110        this(programLocation, false);
111      }
112    
113      /**
114       * Creates a new {@link Jump} instruction.
115       *
116       * @param programLocation the new location within a {@link Program}
117       * to {@linkplain InstructionContext#jump(int, boolean) jump to}
118       *
119       * @param relative if {@code true}, then the supplied {@code
120       * programLocation} will be interpreted as being relative to the
121       * current program location in any {@link InstructionContext} passed
122       * to the {@link #execute(InstructionContext)} method; if {@code
123       * false}, then if the supplied {@code programLocation} is positive
124       * or zero it will be treated as an absolute program location
125       */
126      public Jump(final int programLocation, final boolean relative) {
127        super();
128        this.relative = relative || programLocation < 0;
129        this.programLocation = programLocation;
130      }
131    
132      /**
133       * Calls {@link InstructionContext#jump(int, boolean)} supplying it
134       * with the values of the {@link #programLocation} field and the
135       * {@link #relative} field respectively.
136       *
137       * @param context the {@link InstructionContext} in which this
138       * {@link Jump} instruction is running; must not be {@code null}
139       *
140       * @exception IllegalArgumentException if {@code context} is {@code
141       * null}
142       */
143      @Override
144      public void execute(final InstructionContext<? extends T> context) {
145        if (context == null) {
146          throw new IllegalArgumentException("context");
147        }
148        context.jump(this.programLocation, this.relative);
149      }
150    
151      /**
152       * Returns a hashcode for this {@link Jump} instruction.
153       *
154       * @return a hashcode for this {@link Jump} instruction
155       */
156      @Override
157      public int hashCode() {
158        return this.programLocation + (relative ? 1 : 0);
159      }
160    
161      /**
162       * Returns {@code true} if the supplied {@link Object} is equal to
163       * this {@link Jump} instruction.
164       *
165       * @param other the {@link Object} to test; may be {@code null} in
166       * which case {@code false} will be returned
167       */
168      @Override
169      public boolean equals(final Object other) {
170        if (other == this) {
171          return true;
172        } else if (super.equals(other)) {
173          final Jump him = (Jump)other;
174          return this.programLocation == him.programLocation && this.relative == him.relative;
175        } else {
176          return false;
177        }
178      }
179    
180      /**
181       * Returns a non-{@code null} {@link String} representation of this
182       * {@link Jump} instruction.
183       *
184       * @return a non-{@code null} {@link String} representation of this
185       * {@link Jump} instruction
186       */
187      @Override
188      public String toString() {
189        final StringBuilder sb = new StringBuilder(super.toString()).append(" ");
190        if (this.relative && this.programLocation >= 0) {
191          sb.append("+");
192        }
193        sb.append(this.programLocation);
194        return sb.toString();
195      }
196    
197    }
