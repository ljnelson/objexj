166      public void setSource(final Object source) {
167        if (source == null) {
168          throw new IllegalArgumentException("source", new NullPointerException("source"));

177      /**
178       * Returns {@code true} if the supplied {@code programCounter} is
179       * valid&mdash;that is, greater than or equal to {@code 0} and less
180       * than {@linkplain #size() the return value of the
181       * <code>size()</code> method}.
182       * 
183       * @return {@code true} if the supplied {@code programCounter} is
184       * valid; {@code false} otherwise
185       */
186      public final boolean isValidProgramCounter(final int programCounter) {
187        return programCounter >= 0 && programCounter < this.size();
188      }

190      /**
191       * Returns the {@link Instruction} found at the supplied {@code
192       * index}, or {@code null} if there is no such {@link Instruction}.
193       *
194       * @param index the zero-based index under which an {@link
195       * Instruction} will hopefully be found
196       *
197       * @return an {@link Instruction} indexed under the supplied {@code
198       * index}, or {@code null}
199       * 
200       * @exception InvalidProgramCounterException if {@code index} is
201       * less than {@code 0} or greater than or equal to {@linkplain
202       * #size() this <code>Program</code>'s size}
203       */
204      @Override
205      public final Instruction<T> get(final int index) {
206        if (!isValidProgramCounter(index)) {
207          throw new InvalidProgramCounterException();
208        }
209        final Instruction<T> instruction = super.get(index);
210        assert instruction != null;
211        return instruction;
212      }

345      /**
346       * Returns a non-{@code null} textual representation of this {@link
347       * Program}.
348       *
349       * @return a non-{@code null} {@link String}
350       */
351      @Override
352      public String toString() {
353        final StringBuilder sb = new StringBuilder();
354        final String name = this.getName();
355        if (name != null) {
356          sb.append(String.format("%s:%n", name));
357        }
358        final Iterator<Instruction<T>> iterator = this.iterator();
359        assert iterator != null;
360        if (iterator.hasNext()) {
361          for (int i = 0; iterator.hasNext(); i++) {
362            sb.append(String.format("%3d: %s", i, iterator.next()));
363            if (iterator.hasNext()) {
364              sb.append(LS);
365            }
366          }      
367        }
368        return sb.toString();
369      }

443      /**
444       * Creates and returns a {@link Program} parsed from the supplied
445       * {@link File}.
446       *
447       * <p>This method never returns {@code null}.</p>
448       *
449       * @param <T> the type of {@link Object}s the returned {@link
450       * Program} will be able to match
451       *
452       * @param file a {@link File} containing {@link Instruction}s in
453       * {@linkplain Instruction#valueOf(String) source form}; must not be
454       * {@code null}
455       *
456       * @return a new {@link Program}; never {@code null}
457       *
458       * @exception IllegalArgumentException if {@code file} is {@code
459       * null}
460       *
461       * @exception ClassNotFoundException if there was an error
462       * assembling an {@link Instruction} in the {@link File}
463       *
464       * @exception IllegalAccessException if there was an error
465       * assembling an {@link Instruction} in the {@link File}
466       *
467       * @exception InstantiationException if there was an error
468       * assembling an {@link Instruction} in the {@link File}
469       *
470       * @exception InvocationTargetException if there was an error
471       * assembling an {@link Instruction} in the {@link File}
472       *
473       * @exception IOException if there was an error
474       * assembling an {@link Instruction} in the {@link File}
475       *
476       * @exception NoSuchMethodException if there was an error
477       * assembling an {@link Instruction} in the {@link File}
478       */
479      public static final <T> Program<T> valueOf(final File file) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
480        if (file == null) {
481          throw new IllegalArgumentException("file");
482        }
483        final BufferedReader reader = new BufferedReader(new FileReader(file));
484        try {
485          return valueOf(reader);
486        } finally {
487          try {
488            reader.close();
489          } catch (final IOException nothingToBeDone) {
490            final Logger logger = Logger.getLogger(Program.class.getName());
491            if (logger != null && logger.isLoggable(Level.SEVERE)) {
492              logger.logp(Level.SEVERE, Program.class.getName(), "parse", "Exception encountered while attempting to close reader", nothingToBeDone);
493            }
494          }
495        }
496      }

498      /**
499       * Creates and returns a {@link Program} parsed from the supplied
500       * {@link BufferedReader}.
501       *
502       * <p>This method never returns {@code null}.</p>
503       *
504       * @param <T> the type of {@link Object}s the returned {@link
505       * Program} will be able to match
506       *
507       * @param reader a {@link BufferedReader} that can read {@link
508       * Instruction}s in {@linkplain Instruction#valueOf(String) source
509       * form}; must not be {@code null} and must not be {@linkplain
510       * Reader#close() closed}
511       *
512       * @return a new {@link Program}; never {@code null}
513       *
514       * @exception IllegalArgumentException if {@code reader} is {@code
515       * null}
516       *
517       * @exception ClassNotFoundException if there was an error
518       * assembling an {@link Instruction} in the {@link File}
519       *
520       * @exception IllegalAccessException if there was an error
521       * assembling an {@link Instruction} in the {@link File}
522       *
523       * @exception InstantiationException if there was an error
524       * assembling an {@link Instruction} in the {@link File}
525       *
526       * @exception InvocationTargetException if there was an error
527       * assembling an {@link Instruction} in the {@link File}
528       *
529       * @exception IOException if there was an error
530       * assembling an {@link Instruction} in the {@link File}
531       *
532       * @exception NoSuchMethodException if there was an error
533       * assembling an {@link Instruction} in the {@link File}
534       */
535      public static final <T> Program<T> valueOf(final BufferedReader reader) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
536        if (reader == null) {
537          throw new IllegalArgumentException("reader == null");
538        }
539        Program<T> program = new Program<T>();
540        String line = null;
541        while ((line = reader.readLine()) != null) {
542          line = line.trim();
543          if (!line.isEmpty() && !line.startsWith("#")) {
544            program.add(Instruction.<T>valueOf(line));
545          }
546        }
547        return program;
548      }

664       *
665       * @exception IllegalAccessException if there was an error

682        if (lines != null && lines.hasNext()) {
683          while (lines.hasNext()) {
684            final String line = lines.next().trim();
685            if (!line.isEmpty() && !line.startsWith("#")) {

688          }
