048    public final class Program<T> extends LinkedList<Instruction<T>> {

052      private static final String LS = System.getProperty("line.separator", "\n");

054      private String name;
055

058      public Program() {
059        super();
060      }
061    
062      public Program(final Instruction<T> instruction) {
063        super();
064        if (instruction != null) {
065          this.add(instruction);
066        }
067      }
068    
069      public Program(final Collection<? extends Instruction<T>> instructions) {
070        super();
071        if (instructions != null) {
072          this.addAll(instructions);
073        }
074      }

076      public String getName() {
077        return this.name;
078      }
079    
080      public void setName(final String name) {
081        this.name = name;
082      }

084      /**
085       * Returns whatever {@link Object} it was from which this {@link
086       * Program} was compiled.  The return value is really useful only
087       * for its {@link Object#toString() toString()} method.  This method
088       * may return {@code null}.
089       *
090       * @return the source {@link Object} from which this {@link Program}
091       * was compiled, or {@code null}
092       */
093      public Object getSource() {
094        return this.source;
095      }
096    
097      public void setSource(final Object source) {
098        if (source == null) {
099          throw new IllegalArgumentException("source", new NullPointerException("source"));
100        }
101        final Object old = this.getSource();
102        if (old != null) {
103          throw new IllegalStateException("getSource() != null: " + old);
104        }
105        this.source = source;
106      }
107    
108      public final boolean isValidProgramCounter(final int programCounter) {
109        return programCounter >= 0 && programCounter < this.size();
110      }
111    
112      /**
113       * @exception InvalidProgramCounterException if {@code index} is
114       * less than {@code 0} or greater than or equal to {@linkplain
115       * #size() this <tt>Program</tt>'s size}
116       */
117      @Override
118      public final Instruction<T> get(final int index) {
119        if (!isValidProgramCounter(index)) {
120          throw new InvalidProgramCounterException();
121        }
122        final Instruction<T> instruction = super.get(index);
123        assert instruction != null;
124        return instruction;
125      }

152      }

159      }

161      @Override
162      public final boolean addAll(final Collection<? extends Instruction<T>> instructions) {
163        boolean returnValue = false;
164        if (instructions != null && !instructions.isEmpty()) {
165          returnValue = super.addAll(instructions);
166        }
167        return returnValue;
168      }
169    
170      @Override
171      public final boolean addAll(final int index, final Collection<? extends Instruction<T>> instructions) {
172        boolean returnValue = false;
173        if (instructions != null && !instructions.isEmpty()) {
174          returnValue = super.addAll(index, instructions);
175        }
176        return returnValue;
177      }
178    
179      @Override
180      public String toString() {
181        final StringBuilder sb = new StringBuilder();
182        final String name = this.getName();
183        if (name != null) {
184          sb.append(String.format("%s:%n", name));
185        }
186        final Iterator<Instruction<T>> iterator = this.iterator();
187        assert iterator != null;
188        if (iterator.hasNext()) {
189          for (int i = 0; iterator.hasNext(); i++) {
190            sb.append(String.format("%3d: %s", i, iterator.next()));
191            if (iterator.hasNext()) {
192              sb.append(LS);
193            }
194          }      
195        }
196        return sb.toString();
197      }

241      public static final <T> Program<T> valueOf(final File file) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
242        if (file == null) {
243          throw new IllegalArgumentException("file");
244        }
245        final BufferedReader reader = new BufferedReader(new FileReader(file));
246        try {
247          return valueOf(reader);
248        } finally {
249          try {
250            reader.close();
251          } catch (final IOException nothingToBeDone) {
252            final Logger logger = Logger.getLogger(Program.class.getName());
253            if (logger != null && logger.isLoggable(Level.SEVERE)) {
254              logger.logp(Level.SEVERE, Program.class.getName(), "parse", "Exception encountered while attempting to close reader", nothingToBeDone);
255            }
256          }
257        }
258      }
259    
260      public static final <T> Program<T> valueOf(final BufferedReader reader) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
261        if (reader == null) {
262          throw new IllegalArgumentException("reader == null");
263        }
264        Program<T> program = new Program<T>();
265        String line = null;
266        while ((line = reader.readLine()) != null) {
267          line = line.trim();
268          if (!line.isEmpty() && !line.startsWith("#")) {
269            program.add(Instruction.<T>valueOf(line));
270          }
271        }
272        return program;
273      }
274      
275      public static final <T> Program<T> valueOf(final String text) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
276        if (text == null) {
277          throw new IllegalArgumentException("text == null");
278        }
279        final BufferedReader reader = new BufferedReader(new StringReader(text));
280        try {
281          return valueOf(reader);
282        } finally {
283          try {
284            reader.close();
285          } catch (final IOException nothingToBeDone) {
286            final Logger logger = Logger.getLogger(Program.class.getName());
287            if (logger != null && logger.isLoggable(Level.SEVERE)) {
288              logger.logp(Level.SEVERE, Program.class.getName(), "parse", "Exception encountered while attempting to close reader", nothingToBeDone);
289            }
290          }
291        }
292      }

294      public static final <T> Program<T> valueOf(final Iterable<String> lines) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
295        return valueOf(lines == null ? (Iterator<String>)null : lines.iterator());
296      }
297    
298      public static final <T> Program<T> valueOf(final Iterator<String> lines) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
299        Program<T> program = new Program<T>();
300        if (lines != null && lines.hasNext()) {
301          while (lines.hasNext()) {
302            final String line = lines.next().trim();
303            if (!line.isEmpty() && !line.startsWith("#")) {
304              program.add(Instruction.<T>valueOf(line));
305            }
306          }
307        }
308        return program;
309      }
310    
311      public static final <T> Program<T> singleton(final Instruction<T> instruction) {
312        final Program<T> program = new Program<T>();
313        if (instruction != null) {
314          program.add(instruction);
315        }
316        return program;
317      }
