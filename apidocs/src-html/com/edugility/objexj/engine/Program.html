048    /**
049     * A {@link LinkedList} of {@link Instruction}s that can be
050     * {@linkplain Engine#run(Program, List) run} by an {@link Engine} via

142      /**
143       * Returns whatever {@link Object} it was from which this {@link
144       * Program} was compiled.  The return value is really useful only
145       * for its {@link Object#toString() toString()} method.  This method
146       * may return {@code null}.
147       *
148       * @return the source {@link Object} from which this {@link Program}
149       * was compiled, or {@code null}
150       */
151      public Object getSource() {
152        return this.source;
153      }
154    
155      /**
156       * Sets the source from which this {@link Program} was compiled,
157       * provided that this {@link Program} has not yet had its source
158       * set.
159       * 
160       * @param source the new source; may be {@code null}
161       */
162      public void setSource(final Object source) {
163        if (source == null) {
164          throw new IllegalArgumentException("source", new NullPointerException("source"));
165        }
166        final Object old = this.getSource();
167        if (old != null) {
168          throw new IllegalStateException("getSource() != null: " + old);
169        }
170        this.source = source;
171      }
172    
173      /**
174       * Returns {@code true} if the supplied {@code programCounter} is
175       * valid&mdash;that is, greater than or equal to {@code 0} and less
176       * than {@linkplain #size() the return value of the
177       * <code>size()</code> method}.
178       * 
179       * @return {@code true} if the supplied {@link programCounter} is
180       * valid; {@code false} otherwise
181       */
182      public final boolean isValidProgramCounter(final int programCounter) {
183        return programCounter >= 0 && programCounter < this.size();
184      }
185    
186      /**
187       * Returns the {@link Instruction} found at the supplied {@code
188       * index}, or {@code null} if there is no such {@link Instruction}.
189       *
190       * @param index the zero-based index under which an {@link
191       * Instruction} will hopefully be found
192       *
193       * @return an {@link Instruction} indexed under the supplied {@code
194       * index}, or {@code null}
195       * 
196       * @exception InvalidProgramCounterException if {@code index} is
197       * less than {@code 0} or greater than or equal to {@linkplain
198       * #size() this <code>Program</code>'s size}
199       */
200      @Override
201      public final Instruction<T> get(final int index) {
202        if (!isValidProgramCounter(index)) {
203          throw new InvalidProgramCounterException();
204        }
205        final Instruction<T> instruction = super.get(index);
206        assert instruction != null;
207        return instruction;
208      }

253      @Override
254      public final boolean addAll(final int index, final Collection<? extends Instruction<T>> instructions) {
255        boolean returnValue = false;
256        if (instructions != null && !instructions.isEmpty()) {
257          returnValue = super.addAll(index, instructions);
258        }
259        return returnValue;
260      }
261    
262      @Override
263      public String toString() {
264        final StringBuilder sb = new StringBuilder();
265        final String name = this.getName();
266        if (name != null) {
267          sb.append(String.format("%s:%n", name));
268        }
269        final Iterator<Instruction<T>> iterator = this.iterator();
270        assert iterator != null;
271        if (iterator.hasNext()) {
272          for (int i = 0; iterator.hasNext(); i++) {
273            sb.append(String.format("%3d: %s", i, iterator.next()));
274            if (iterator.hasNext()) {
275              sb.append(LS);
276            }
277          }      
278        }
279        return sb.toString();
280      }

354      public static final <T> Program<T> valueOf(final File file) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
355        if (file == null) {
356          throw new IllegalArgumentException("file");
357        }
358        final BufferedReader reader = new BufferedReader(new FileReader(file));
359        try {
360          return valueOf(reader);
361        } finally {
362          try {
363            reader.close();
364          } catch (final IOException nothingToBeDone) {
365            final Logger logger = Logger.getLogger(Program.class.getName());
366            if (logger != null && logger.isLoggable(Level.SEVERE)) {
367              logger.logp(Level.SEVERE, Program.class.getName(), "parse", "Exception encountered while attempting to close reader", nothingToBeDone);
368            }
369          }
370        }
371      }
372    
373      public static final <T> Program<T> valueOf(final BufferedReader reader) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, IOException, NoSuchMethodException {
374        if (reader == null) {
375          throw new IllegalArgumentException("reader == null");
376        }
377        Program<T> program = new Program<T>();
378        String line = null;
379        while ((line = reader.readLine()) != null) {
380          line = line.trim();
381          if (!line.isEmpty() && !line.startsWith("#")) {
382            program.add(Instruction.<T>valueOf(line));
383          }
384        }
385        return program;
386      }

407      public static final <T> Program<T> valueOf(final Iterable<String> lines) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
408        return valueOf(lines == null ? (Iterator<String>)null : lines.iterator());
409      }
410    
411      public static final <T> Program<T> valueOf(final Iterator<String> lines) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
412        Program<T> program = new Program<T>();
413        if (lines != null && lines.hasNext()) {
414          while (lines.hasNext()) {
415            final String line = lines.next().trim();
416            if (!line.isEmpty() && !line.startsWith("#")) {
417              program.add(Instruction.<T>valueOf(line));
418            }
419          }
420        }
421        return program;
422      }
423    
424      public static final <T> Program<T> singleton(final Instruction<T> instruction) {
425        final Program<T> program = new Program<T>();
426        if (instruction != null) {
427          program.add(instruction);
428        }
429        return program;
430      }
