028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.lang.reflect.Constructor;
033    import java.lang.reflect.InvocationTargetException;
034    
035    import java.util.regex.Matcher;
036    import java.util.regex.Pattern;
037    
038    /**
039     * An instruction {@linkplain #execute(InstructionContext) run} in the
040     * {@linkplain InstructionContext context} of a {@linkplain
041     * com.edugility.objexj.engine.Thread thread} of execution in an
042     * {@code objexj} {@linkplain Engine virtual machine}.
043     *
044     * @author <a href="http://about.me/lairdnelson">Laird Nelson</a>
045     *
046     * @see Engine
047     *
048     * @see InstructionContext
049     *
050     * @see Thread
051     */
052    public abstract class Instruction<T> implements Serializable {
053    
054      /**
055       * The version of this class for serialization purposes.
056       *
057       * @see Serializable
058       */
059      private static final long serialVersionUID = 1L;
060    
061      /**
062       * A {@link Pattern} useful in parsing an {@link Instruction} from
063       * its textual representation.  This field is never {@code null}.
064       *
065       * @see #valueOf(String)
066       */
067      private static final Pattern LINE_PATTERN = Pattern.compile("^([^\\s]+)(?:\\s+?(.*))?$");
068    
069      /**
070       * Creates a new {@link Instruction}.
071       */
072      protected Instruction() {
073        super();
074      }
075    
076      /**
077       * Executes this {@link Instruction} in the context of the supplied
078       * {@link InstructionContext}.
079       *
080       * @param context the {@link InstructionContext} in effect; must not
081       * be {@code null}
082       *
083       * @exception IllegalArgumentException if {@code context} is {@code
084       * null}
085       */
086      public abstract void execute(final InstructionContext<T> context);
087    
088      /**
089       * Loads a {@link Class}.  This implementation uses the {@linkplain
090       * java.lang.Thread#getContextClassLoader() context classloader}.
091       *
092       * @param className the name of the class to load; must not be
093       * {@code null}
094       *
095       * @return the loaded {@link Class}; never {@code null}
096       *
097       * @exception IllegalArgumentException if {@code className} is
098       * {@code null}
099       *
100       * @exception ClassNotFoundException if a {@link Class}
101       * corresponding to the supplied {@code className} parameter could
102       * not be found
103       */
104      protected Class<?> loadClass(final String className) throws ClassNotFoundException {
105        if (className == null) {
106          throw new IllegalArgumentException("className", new NullPointerException("className"));
107        }
108        return java.lang.Thread.currentThread().getContextClassLoader().loadClass(className);
109      }
110    
111      /**
112       * Checks to see if the supplied {@link Object} is equal to this
113       * {@link Instruction}.  This implementation returns {@code true} if
114       * and only if the supplied {@link Object} is non-{@code null} and
115       * has a {@link Object#getClass() Class} that is {@linkplain
116       * Class#equals(Object) equal to} this {@link Instruction}'s {@link
117       * Object#getClass() Class}.
118       *
119       * @param other the {@link Object} to test; may be {@code null}
120       *
121       * @return {@code true} if the supplied {@link Object} is equal to
122       * this {@link Instruction}; {@code false} otherwise
123       */
124      @Override
125      public boolean equals(final Object other) {
126        return other == this || this.getClass().equals(other.getClass());
127      }
128    
129      /**
130       * Returns a hashcode for this {@link Instruction}.  This
131       * implementation returns the hashcode of this {@link Instruction}'s
132       * {@link Object#getClass() Class}.
133       *
134       * @return a hashcode for this {@link Instruction}
135       */
136      @Override
137      public int hashCode() {
138        return this.getClass().hashCode();
139      }
140    
141      /**
142       * Returns a non-{@code null} {@link String} representation of this
143       * {@link Instruction}.
144       *
145       * @return a non-{@code null} {@link String} representation of this
146       * {@link Instruction}
147       */
148      @Override
149      public String toString() {
150        final String simpleName = this.getClass().getSimpleName();
151        return String.format("%s%s", simpleName.substring(0, 1).toLowerCase(), simpleName.substring(1));
152      }
153    
154      
155      /*
156       * Static methods.
157       */
158    
159    
160      /**
161       * Parses the supplied {@code line} and returns a new {@link
162       * Instruction} instance appropriate for this input.
163       *
164       * @param line the {@link String} to parse; must not be {@code null}
165       *
166       * @return a new {@link Instruction} instance; never {@code null}
167       *
168       * @exception IllegalArgumentException if {@code line} is {@code
169       * null} or unparseable
170       *
171       * @exception ClassNotFoundException if a class corresponding to a
172       * portion of the supplied {@code line} could not be found
173       *
174       * @exception IllegalAccessException if a new {@link Instruction}
175       * instance could not be created because its constructor was found
176       * but could not be called due to access restrictions
177       *
178       * @exception InstantiationException if a new {@link Instruction}
179       * instance could not be created because the constructor could not
180       * be called
181       *
182       * @exception InvocationTargetException if a new {@link Instruction}
183       * instance could not be created becase the invocation of its
184       * constructor threw an {@link Exception}
185       *
186       * @exception NoSuchMethodException if a new {@link Instruction}
187       * instance could not be created because its constructor could not
188       * be found
189       */
190      public static final <T> Instruction<T> valueOf(final String line) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
191        if (line == null) {
192          throw new IllegalArgumentException("line", new NullPointerException("line"));
193        }
194        final Matcher m = LINE_PATTERN.matcher(line);
195        assert m != null;
196        if (!m.matches()) {
197          throw new IllegalArgumentException("Bad instruction line: " + line);
198        }
199    
200        final String rawCommand = m.group(1);
201        assert rawCommand != null;
202        assert rawCommand.length() >= 2;
203    
204        final String command = String.format("com.edugility.objexj.engine.%s%s", rawCommand.substring(0, 1).toUpperCase(), rawCommand.substring(1));
205        assert command != null;
206        
207        final Class<?> instructionClass = java.lang.Thread.currentThread().getContextClassLoader().loadClass(command);
208        assert instructionClass != null;
209        if (!Instruction.class.isAssignableFrom(instructionClass)) {
210          throw new IllegalArgumentException("bad instruction: " + command);
211        }
212        
213        String operands = m.group(2);
214        if (operands == null) {
215          operands = "";
216        } else {
217          operands = operands.trim();
218        }
219    
220        Object instruction = null;
221        final Constructor<?> c;
222        if (operands.isEmpty()) {
223          c = null;
224          instruction = instructionClass.newInstance();
225        } else {
226          c = instructionClass.getConstructor(String.class);
227          assert c != null;
228          instruction = c.newInstance(operands);
229        }
230        @SuppressWarnings("unchecked")
231        final Instruction<T> temp = (Instruction<T>)instruction;
232        return temp;
233      }
234    
235    }
