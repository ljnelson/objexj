028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.lang.reflect.Constructor;
033    import java.lang.reflect.InvocationTargetException;
034    
035    import java.util.regex.Matcher;
036    import java.util.regex.Pattern;
037    
038    public abstract class Instruction<T> implements Serializable {
039    
040      private static final long serialVersionUID = 1L;
041    
042      private static final Pattern LINE_PATTERN = Pattern.compile("^([^\\s]+)(?:\\s+?(.*))?$");
043    
044      protected Instruction() {
045        super();
046      }
047    
048      public abstract void execute(final InstructionContext<T> context);
049    
050      protected Class<?> loadClass(final String className) throws ClassNotFoundException {
051        if (className == null) {
052          throw new IllegalArgumentException("className", new NullPointerException("className"));
053        }
054        return java.lang.Thread.currentThread().getContextClassLoader().loadClass(className);
055      }
056    
057      @Override
058      public boolean equals(final Object other) {
059        return other == this || this.getClass().equals(other.getClass());
060      }
061    
062      @Override
063      public int hashCode() {
064        return this.getClass().hashCode();
065      }
066    
067      @Override
068      public String toString() {
069        final String simpleName = this.getClass().getSimpleName();
070        return String.format("%s%s", simpleName.substring(0, 1).toLowerCase(), simpleName.substring(1));
071      }
072    
073      
074      /*
075       * Static methods.
076       */
077    
078    
079      public static final <T> Instruction<T> valueOf(final String line) throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException, NoSuchMethodException {
080        if (line == null) {
081          throw new IllegalArgumentException("line", new NullPointerException("line"));
082        }
083        final Matcher m = LINE_PATTERN.matcher(line);
084        assert m != null;
085        if (!m.matches()) {
086          throw new IllegalArgumentException("Bad instruction line: " + line);
087        }
088    
089        final String rawCommand = m.group(1);
090        assert rawCommand != null;
091        assert rawCommand.length() >= 2;
092    
093        final String command = String.format("com.edugility.objexj.engine.%s%s", rawCommand.substring(0, 1).toUpperCase(), rawCommand.substring(1));
094        assert command != null;
095        
096        final Class<?> instructionClass = java.lang.Thread.currentThread().getContextClassLoader().loadClass(command);
097        assert instructionClass != null;
098        if (!Instruction.class.isAssignableFrom(instructionClass)) {
099          throw new IllegalArgumentException("bad instruction: " + command);
100        }
101        
102        String operands = m.group(2);
103        if (operands == null) {
104          operands = "";
105        } else {
106          operands = operands.trim();
107        }
108    
109        Object instruction = null;
110        final Constructor<?> c;
111        if (operands.isEmpty()) {
112          c = null;
113          instruction = instructionClass.newInstance();
114        } else {
115          c = instructionClass.getConstructor(String.class);
116          assert c != null;
117          instruction = c.newInstance(operands);
118        }
119        @SuppressWarnings("unchecked")
120        final Instruction<T> temp = (Instruction<T>)instruction;
121        return temp;
122      }
123    
124    }
