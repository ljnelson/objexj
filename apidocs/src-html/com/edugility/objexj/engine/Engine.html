049    public class Engine<T> {
050    
051      public Engine() {
052        super();
053      }
054    
055      public MatchResult<T> run(final Program<T> program, final List<T> items) {
056        if (program == null) {
057          throw new IllegalArgumentException("program", new NullPointerException("program"));
058        }
059        final AtomicInteger idGenerator = new AtomicInteger();
060        final Queue<Thread<T>> threads = new LinkedList<Thread<T>>();
061        final Scheduler<T> scheduler = new Scheduler<T>(threads, idGenerator);
062    
063        scheduler.schedule(scheduler.newThread(String.format("T%d", idGenerator.getAndIncrement()),
064                                               new ProgramCounter<T>(program),
065                                               items,
066                                               0,
067                                               null,
068                                               null));
069        MatchResult<T> result = null;
070        while (!threads.isEmpty()) {
071          final Thread<T> thread = threads.remove();
072          assert thread != null;
073          thread.run();
074          if (Thread.State.MATCH == thread.getState()) {
075            result = new MatchResult<T>(thread);
076            break;
077          }
078        }
079        return result;
080      }
081    
082      private static final class Scheduler<T> implements ThreadScheduler<T> {
083    
084        private final Queue<Thread<T>> threads;
085    
086        private final AtomicInteger idGenerator;
087    
088        private final Logger logger;
089    
090        private Scheduler(final Queue<Thread<T>> threads) {
091          this(threads, new AtomicInteger());
092        }
093    
094        private Scheduler(final Queue<Thread<T>> threads, final AtomicInteger idGenerator) {
095          super();
096          this.logger = Logger.getLogger(this.getClass().getName());
097          this.threads = threads;
098          if (idGenerator == null) {
099            this.idGenerator = new AtomicInteger();
100          } else {
101            this.idGenerator = idGenerator;
102          }
103        }
104    
105        @Override
106        public final Thread<T> newThread(Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables) {
107          if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
108            this.logger.entering(this.getClass().getName(), "newThread", new Object[] { id, programCounter, items, itemPointer, captureGroups, variables });
109          }      
110          if (id == null) {
111            id = String.format("T%d", this.idGenerator.getAndIncrement());
112          }
113          final Thread<T> returnValue = new Thread<T>(id, programCounter, items, itemPointer, captureGroups, variables, this);
114          if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
115            this.logger.exiting(this.getClass().getName(), "newThread", returnValue);
116          }      
117          return returnValue;
118        }
119          
120        @Override
121        public final boolean schedule(final Thread<T> t) {
122          if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
123            this.logger.entering(this.getClass().getName(), "schedule", t);
124          }
125          if (t == null) {
126            throw new IllegalArgumentException("t", new NullPointerException("t"));
127          }
128          if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
129            this.logger.logp(Level.FINER, this.getClass().getName(), "schedule", "Scheduling thread {0}", t);
130          }
131          final boolean returnValue = this.threads != null && this.threads.add(t);
132          if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
133            this.logger.exiting(this.getClass().getName(), "schedule", Boolean.valueOf(returnValue));
134          }
135          return returnValue;
136        }
137      }
