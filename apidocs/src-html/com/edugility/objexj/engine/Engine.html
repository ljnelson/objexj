049    /**
050     * A stateless regular-expression-processing virtual machine made to
051     * run {@link Program}s against {@link List}s of input using {@link
052     * com.edugility.objexj.engine.Thread}s.
053     *
054     * @param <T> the type of {@link Object}s that {@link List}s supplied
055     * to the {@link #run(Program, List)} method will consist of
056     *
057     * @author <a href="http://about.me/lairdnelson"
058     * target="_parent">Laird Nelson</a>
059     *
060     * @see #run(Program, List)
061     *
062     * @see <a
063     * href="http://swtch.com/~rsc/regexp/regexp2.html" target="_parent">R. S. Cox. (2009,
064     * December). <span style="font-style: italic;">Regular Expression Matching: the Virtual Machine
065     * Approach</span> [Online]. Available:
066     * http://swtch.com/~rsc/regexp/regexp2.html</a>
067     */
068    public class Engine<T> {
069    
070

073       */

089      /**
090       * Runs the supplied {@link Program} against the supplied {@link
091       * List} and returns a (possibly {@code null}) {@link MatchResult}
092       * describing the run result.
093       *
094       * @param program the {@link Program} to run; must not be {@code
095       * null}
096       *
097       * @param items the input {@link List}; may be {@code null}
098       *
099       * @return a {@link MatchResult}, or {@code null} if no match
100       * occurred
101       *
102       * @exception IllegalArgumentException if {@code program} is {@code
103       * null}
104       */
105      public MatchResult<? extends T> run(final Program<T> program, final List<? extends T> items) {
106        if (program == null) {
107          throw new IllegalArgumentException("program", new NullPointerException("program"));
108        }
109        final AtomicInteger idGenerator = new AtomicInteger();
110        final Queue<Thread<T>> threads = new LinkedList<Thread<T>>();
111        final Scheduler<T> scheduler = new Scheduler<T>(threads, idGenerator);
112    
113        scheduler.schedule(scheduler.newThread(String.format("T%d", idGenerator.getAndIncrement()),
114                                               new ProgramCounter<T>(program),
115                                               items,
116                                               0,
117                                               null,
118                                               null));
119        MatchResult<? extends T> result = null;
120        while (!threads.isEmpty()) {
121          final Thread<T> thread = threads.remove();
122          if (thread != null) {
123            thread.run();
124            if (Thread.State.MATCH == thread.getState()) {
125              result = new MatchResult<T>(thread);
126              break;
127            }
128          }
129        }
130        return result;
131      }

139      /**
140       * A {@link ThreadScheduler} that schedules {@link Thread}s in a
141       * simplistic, sequential manner.
142       *
143       * @author <a href="http://about.me/lairdnelson"
144       * target="_parent">Laird Nelson</a>
145       */
146      private static final class Scheduler<T> implements ThreadScheduler<T> {
147    
148        /**
149         * A {@link Queue} of scheduled {@link Thread}s.  This field may
150         * be {@code null}.
151         */
152        private final Queue<Thread<T>> threads;
153    
154        /**
155         * An {@link AtomicInteger} used to generate {@link Thread}
156         * identifiers.  This field is never {@code null}.
157         */
158        private final AtomicInteger idGenerator;
159    
160        /**
161         * Creates a new {@link Scheduler}.
162         *
163         * @param threads a {@lik Queue} of {@link Thread}s; may be {@code
164         * null}
165         */
166        private Scheduler(final Queue<Thread<T>> threads) {
167          this(threads, new AtomicInteger());
168        }
169    
170        /**
171         * Creates a new {@link Scheduler}.
172         *
173         * @param threads a {@link Queue} of {@link Thread}s; may be
174         * {@code null}
175         *
176         * @param idGenerator an {@link AtomicInteger} to help with {@link
177         * Thread} identifier generation; may be {@code null}
178         */
179        private Scheduler(final Queue<Thread<T>> threads, final AtomicInteger idGenerator) {
180          super();
181          this.threads = threads;
182          if (idGenerator == null) {
183            this.idGenerator = new AtomicInteger();
184          } else {
185            this.idGenerator = idGenerator;
186          }
187        }
188    
189        /**
190         * Creates a new {@link Thread}.
191         *
192         * @param id the identifier for the new {@link Thread}; may be
193         * {@code null} in which case a default identifier will be used
194         * instead
195         *
196         * @param programCounter a {@link ProgramCounter}; must not be {@code null}
197         *
198         * @param items the input {@link List} to read from; may be {@code
199         * null} in which case the supplied {@code itemPointer} must be
200         * equal to {@code 0} or {@link Thread#VALID_NO_INPUT_POINTER}
201         *
202         * @param itemPointer the index within the supplied {@code items}
203         * {@link List} from which to begin {@linkplain Thread#read()
204         * reading}; must be zero or a positive integer less than the
205         * supplied {@code items} {@linkplain Collection#size() size}, or
206         * must be equal to {@link Thrad#VALID_NO_INPUT_POINTER} provided
207         * that the {@code items} {@link List} is {@code null} or
208         * {@linkplain Collection#isEmpty() empty}
209         *
210         * @param captureGroups a {@link Map} of {@link CaptureGroup}s
211         * indexed by key that this new {@link Thread} will use to store
212         * {@linkplain Thread#getSubmatches() submatches}; may be {@code
213         * null}
214         *
215         * @param variables a {@link Map} of variables the new {@link
216         * Thread} will be initialized with; may be {@code null}
217         *
218         * @exception IllegalArgumentException if any of the preconditions
219         * outlined as part of the parameter descriptions is not fulfilled
220         */
221        @Override
222        public final Thread<T> newThread(Object id, final ProgramCounter<T> programCounter, final List<? extends T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables) {
223          final String className = this.getClass().getName();
224          final Logger logger = Logger.getLogger(className);
225          final boolean finer = logger != null && logger.isLoggable(Level.FINER);
226          if (finer) {
227            logger.entering(className, "newThread", new Object[] { id, programCounter, items, itemPointer, captureGroups, variables });
228          }      
229          if (id == null) {
230            id = String.format("T%d", this.idGenerator.getAndIncrement());
231          }
232          final Thread<T> returnValue = new Thread<T>(id, programCounter, items, itemPointer, captureGroups, variables, this);
233          if (finer) {
234            logger.exiting(className, "newThread", returnValue);
235          }
236          return returnValue;
237        }
238          
239        /**
240         * Schedules the supplied {@link Thread} for execution.  This
241         * implementation of the {@link ThreadScheduler#schedule(Thread)}
242         * method adds the supplied {@link Thread} to the end of a
243         * first-in-last-out {@link Queue} of scheduled {@link Thread}s.
244         *
245         * @param t the {@link Thread} to schedule; must not be {@code
246         * null}
247         *
248         * @return {@code true} if the supplied {@link Thread} was
249         * actually scheduled; {@code false} if this {@link Scheduler}
250         * concluded that&mdash;for whatever reason&mdash;that the
251         * supplied {@link Thread} could not be scheduled
252         *
253         * @exception IllegalArgumentException if {@code t} is {@code
254         * null}
255         */
256        @Override
257        public final boolean schedule(final Thread<T> t) {
258          final String className = this.getClass().getName();
259          final Logger logger = Logger.getLogger(className);
260          final boolean finer = logger != null && logger.isLoggable(Level.FINER);
261          if (finer) {
262            logger.entering(className, "schedule", t);
263          }
264          if (t == null) {
265            throw new IllegalArgumentException("t", new NullPointerException("t"));
266          }
267          if (finer) {
268            logger.logp(Level.FINER, className, "schedule", "Scheduling thread {0}", t);
269          }
270          final boolean returnValue = this.threads != null && this.threads.add(t);
271          if (finer) {
272            logger.exiting(className, "schedule", Boolean.valueOf(returnValue));
273          }
274          return returnValue;
275        }
276      }
