028    package com.edugility.objexj.engine;
029    
030    /**
031     * An {@link Instruction} that {@linkplain
032     * InstructionContext#advanceProgramCounter() advances an
033     * <code>InstructionContext</code>'s program counter} if the {@link
034     * InstructionContext} {@linkplain InstructionContext#canRead() can be
035     * read from} and {@linkplain InstructionContext#read() produces an
036     * <code>Object</code>} that is non-{@code null}.
037     *
038     * @param <T> the type of {@link Object} that can be filtered
039     *
040     * @author <a href="http://about.me/lairdnelson"
041     * target="_parent">Laird Nelson</a>
042     */
043    public class Filter<T> extends Instruction<T> {
044    
045      /**
046       * The version of this class for {@linkplain Serializable
047       * serialization purposes}.
048       */
049      private static final long serialVersionUID = 1L;
050    
051      /**
052       * Creates a new {@link Filter}.
053       */
054      public Filter() {
055        super();
056      }
057    
058      /**
059       * {@linkplain InstructionContext#advanceProgramCounter() Advances
060       * an <code>InstructionContext</code>'s program counter} if and only
061       * if the {@link #accept(InstructionContext)} method returns {@code
062       * true}.
063       *
064       * <p>If the {@link #accept(InstructionContext)} method returns
065       * {@code false}, then this method {@linkplain
066       * InstructionContext#die() kills the
067       * <code>InstructionContext</code>}.</p>
068       *
069       * @param context an {@link InstructionContext} to execute this
070       * {@link Filter} in; must not be {@code null}
071       *
072       * @exception IllegalArgumentException if {@code context} is {@code
073       * null}
074       *
075       * @see #accept(InstructionContext)
076       */
077      @Override
078      public final void execute(final InstructionContext<T> context) {
079        if (context == null) {
080          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
081        }
082        if (this.accept(context)) {
083          context.advanceItemPointer();
084          context.advanceProgramCounter();
085        } else {
086          context.die();
087        }
088      }
089    
090      /**
091       * Returns {@code true} if this {@link Filter} accepts the supplied
092       * {@link InstructionContext}.
093       *
094       * @return {@code true} if and only if the supplied {@link
095       * InstructionContext} is not {@code null}, {@linkplain
096       * InstructionContext#canRead() can be read from}, and {@linkplain
097       * InstructionContext#read() produces an <code>Object</code>} that
098       * is not {@code null}; {@code false} in all other cases
099       */
100      public boolean accept(final InstructionContext<T> context) {
101        return context != null && context.canRead() && context.read() != null;
102      }
103    
104    }
