028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.Collections;
033    import java.util.List;
034    import java.util.Map;
035    import java.util.Set;
036    
037    import java.util.logging.Level;
038    import java.util.logging.Logger;
039    
040    import com.edugility.objexj.engine.Thread;
041    
042    /**
043     * The result of a {@link Thread} that has successfully {@linkplain
044     * Thread#match() matched} its input.
045     *
046     * @param <T> the type of {@link Object} the matching {@link Thread}
047     * works with
048     *
049     * @author <a href="http://about.me/lairdnelson"
050     * target="_parent">Laird Nelson</a>
051     *
052     * @see Thread#match()
053     */
054    public class MatchResult<T> implements Serializable {
055    
056      /**
057       * The version of this class for {@linkplain Serializable
058       * serialization purposes}.
059       */
060      private static final long serialVersionUID = 1L;
061    
062      /**
063       * The {@link Thread} that caused the match to happen.
064       *
065       * <p>This field may be {@code null}.</p>
066       */
067      private final Thread<? extends T> thread;
068    
069      /**
070       * Creates a new {@link MatchResult}.
071       */
072      MatchResult() {
073        super();
074        this.thread = null;
075      }
076    
077      /**
078       * Creates a new {@link MatchResult}.
079       *
080       * @param thread the {@link Thread} that {@linkplain Thread#match()
081       * matched}; may be {@code null}
082       */
083      MatchResult(final Thread<? extends T> thread) {
084        super();
085        if (thread != null && Thread.State.MATCH != thread.getState()) {
086          throw new IllegalArgumentException(String.format("Thread.State.MATCH != thread.getState(): %s", thread.getState()));
087        }
088        this.thread = thread;
089      }
090    
091      /**
092       * Returns a non-{@code null} {@link Logger} for logging messages
093       * from this class.
094       *
095       * <p>This method never returns {@code null}.</p>
096       *
097       * @return a non-{@code null} {@link Logger}
098       */
099      protected Logger getLogger() {
100        return Logger.getLogger(this.getClass().getName());
101      }
102    
103      /**
104       * Returns {@code true} if the {@link #lookingAt()} method returns
105       * {@code true} and if the underlying {@link Thread} was {@linkplain
106       * Thread#atEnd() at the end of its input} when the match occurred.
107       *
108       * @return {@code true} if the {@link #lookingAt()} method returns
109       * {@code true} and if the underlying {@link Thread} was {@linkplain
110       * Thread#atEnd() at the end of its input} when the match occurred.
111       */
112      public boolean matches() {
113        return this.lookingAt() && this.thread.atEnd();
114      }
115    
116      /**
117       * Returns {@code true} if the underlying {@link Thread} {@linkplain
118       * #MatchResult(Thread) supplied at construction time} has a
119       * {@linkplain Thread#getState() state} of {@link
120       * Thread.State#MATCH}.
121       *
122       * @return {@code true} if the underlying {@link Thread} {@linkplain
123       * #MatchResult(Thread) supplied at construction time} has a
124       * {@linkplain Thread#getState() state} of {@link
125       * Thread.State#MATCH}.
126       */
127      public boolean lookingAt() {
128        return this.thread != null && this.thread.getState() == Thread.State.MATCH;
129      }
130    
131      /**
132       * Returns a {@link List} of items captured as a <em>capture
133       * group</em> under the supplied {@code key}.
134       *
135       * <p>This method may return {@code null}.</p>
136       *
137       * @param key the key of the capture group {@link
138       * List} to return; may be {@code null}
139       *
140       * @return a {@link List} of items captured, or {@code null} if
141       * there is no such {@link List}
142       *
143       * @see Thread#getGroup(Object)
144       */
145      public List<? extends T> getGroup(final Object key) {
146        final String className = this.getClass().getName();
147        final Logger logger = this.getLogger();
148        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
149        if (finer) {
150          logger.entering(className, "getGroup", key);
151        }      
152        List<? extends T> result = null;
153        if (this.thread != null) {
154          result = thread.getGroup(key);
155        }
156        if (finer) {
157          logger.exiting(className, "getGroup", result);
158        }      
159        return result;
160      }
161    
162      /**
163       * Returns the total number of <em>capture groups</em> present in
164       * this {@link MatchResult}.
165       *
166       * @return the total number of capture groups present in this {@link
167       * MatchResult}; always an {@code int} greater than or equal to
168       * {@code 0}
169       *
170       * @see #getGroup(Object)
171       *
172       * @see Thread#getGroupCount()
173       */
174      public int getGroupCount() {
175        final String className = this.getClass().getName();
176        final Logger logger = this.getLogger();
177        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
178        if (finer) {
179          logger.entering(className, "getGroupCount");
180        }
181        final int result;
182        if (this.thread == null) {
183          result = 0;
184        } else {
185          result = this.thread.getGroupCount();
186        }
187        if (finer) {
188          logger.exiting(className, "getGroupCount", Integer.valueOf(result));
189        }
190        return result;
191      }
192    
193      /**
194       * Returns an {@linkplain Collections#unmodifiableSet(Set)
195       * unmodifiable} {@link Set} of {@link Object}s representing valid
196       * keys that can be used to {@linkplain #getGroup(Object) capture
197       * groups}.
198       *
199       * <p>This method never returns {@code null}.</p>
200       *
201       * @return a non-{@code null} {@link Set} of keys, each element of
202       * which can be used as an argument to the {@link #getGroup(Object)}
203       * method
204       */
205      public Set<?> getGroupKeySet() {
206        final String className = this.getClass().getName();
207        final Logger logger = this.getLogger();
208        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
209        if (finer) {
210          logger.entering(className, "getGroupKeySet");
211        }
212        final Set<?> returnValue;
213        if (this.thread == null) {
214          returnValue = Collections.emptySet();
215        } else {
216          returnValue = this.thread.getGroupKeySet();
217        }
218        if (finer) {
219          logger.exiting(className, "getGroupKeySet", returnValue);
220        }
221        return returnValue;
222      }
223    
224      /**
225       * Returns the value of a variable indexed under the supplied {@code
226       * key}.
227       *
228       * <p>This method may return {@code null}.</p>
229       *
230       * @param key the name of the variable; may be {@code null}
231       *
232       * @return the value of the variable, or {@code null} if there is no
233       * such variable
234       *
235       * @see #getVariables()
236       */
237      public Object getVariable(final Object key) {
238        final String className = this.getClass().getName();
239        final Logger logger = this.getLogger();
240        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
241        if (finer) {
242          logger.entering(className, "getVariable", key);
243        }
244        Object result = null;
245        if (this.thread != null) {
246          final Map<?, ?> variables = this.getVariables();
247          if (variables != null) {
248            result = variables.get(key);
249          }
250        }
251        if (finer) {
252          logger.exiting(className, "getVariable", result);
253        }
254        return result;
255      }
256    
257      /**
258       * Returns an {@linkplain Collections#unmodifiableMap(Map)
259       * unmodifiable <code>Map</code>} of the variables set on the {@link
260       * Thread} that this {@link MatchResult} wraps.
261       *
262       * <p>This method may return {@code null}.</p>
263       *
264       * @return a {@link Map} of variables, or {@code null}
265       */
266      public Map<?, ?> getVariables() {
267        final String className = this.getClass().getName();
268        final Logger logger = this.getLogger();
269        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
270        if (finer) {
271          logger.entering(className, "getVariables");
272        }
273        Map<?, ?> result = null;
274        if (this.thread != null) {
275          result = this.thread.getVariables();
276        }
277        if (finer) {
278          logger.exiting(className, "getVariables", result);
279        }
280        return result;
281      }
282
