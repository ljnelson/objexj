028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.Collections;
033    import java.util.List;
034    import java.util.Map;
035    import java.util.Set;
036    
037    import com.edugility.objexj.engine.Thread;
038    
039    /**
040     * The result of a {@link Thread} that has successfully {@linkplain
041     * Thread#match() matched} its input.
042     *
043     * @param <T> the type of {@link Object} the matching {@link Thread}
044     * works with
045     *
046     * @author <a href="http://about.me/lairdnelson"
047     * target="_parent">Laird Nelson</a>
048     *
049     * @see Thread#match()
050     */
051    public class MatchResult<T> implements Serializable {
052    
053      /**
054       * The version of this class for {@linkplain Serializable
055       * serialization purposes}.
056       */
057      private static final long serialVersionUID = 1L;
058    
059      /**
060       * The {@link Thread} that caused the match to happen.
061       *
062       * <p>This field may be {@code null}.</p>
063       */
064      private final Thread<T> thread;
065    
066      /**
067       * Creates a new {@link MatchResult}.
068       */
069      MatchResult() {
070        super();
071        this.thread = null;
072      }
073    
074      /**
075       * Creates a new {@link MatchResult}.
076       *
077       * @param thread the {@link Thread} that {@linkplain Thread#match()
078       * matched}; may be {@code null}
079       */
080      MatchResult(final Thread<T> thread) {
081        super();
082        if (thread != null && Thread.State.MATCH != thread.getState()) {
083          throw new IllegalArgumentException(String.format("Thread.State.MATCH != thread.getState(): %s", thread.getState()));
084        }
085        this.thread = thread;
086      }
087    
088      /**
089       * Returns {@code true} if the {@link #lookingAt()} method returns
090       * {@code true} and if the underlying {@link Thread} was {@linkplain
091       * Thread#atEnd() at the end of its input} when the match occurred.
092       *
093       * @return {@code true} if the {@link #lookingAt()} method returns
094       * {@code true} and if the underlying {@link Thread} was {@linkplain
095       * Thread#atEnd() at the end of its input} when the match occurred.
096       */
097      public boolean matches() {
098        return this.lookingAt() && this.thread.atEnd();
099      }
100    
101      /**
102       * Returns {@code true} if the underlying {@link Thread} {@linkplain
103       * #MatchResult(Thread) supplied at construction time} has a
104       * {@linkplain Thread#getState() state} of {@link
105       * Thread.State#MATCH}.
106       *
107       * @return {@code true} if the underlying {@link Thread} {@linkplain
108       * #MatchResult(Thread) supplied at construction time} has a
109       * {@linkplain Thread#getState() state} of {@link
110       * Thread.State#MATCH}.
111       */
112      public boolean lookingAt() {
113        return this.thread != null && this.thread.getState() == Thread.State.MATCH;
114      }
115    
116      /**
117       * Returns a {@link List} of items captured as a <em>capture
118       * group</em> under the supplied {@code key}.
119       *
120       * <p>This method may return {@code null}.</p>
121       *
122       * @param key the key of the capture group {@link
123       * List} to return; may be {@code null}
124       *
125       * @return a {@link List} of items captured, or {@code null} if
126       * there is no such {@link List}
127       *
128       * @see Thread#getGroup(Object)
129       */
130      public List<T> getGroup(final Object key) {
131        List<T> result = null;
132        if (this.thread != null) {
133          result = thread.getGroup(key);
134        }
135        return result;
136      }
137    
138      /**
139       * Returns the total number of <em>capture groups</em> present in
140       * this {@link MatchResult}.
141       *
142       * @return the total number of capture groups present in this {@link
143       * MatchResult}; always an {@code int} greater than or equal to
144       * {@code 0}
145       *
146       * @see #getGroup(Object)
147       *
148       * @see Thread#getGroupCount()
149       */
150      public int getGroupCount() {
151        final int result;
152        if (this.thread == null) {
153          result = 0;
154        } else {
155          result = this.thread.getGroupCount();
156        }
157        return result;
158      }
159    
160      /**
161       * Returns an {@linkplain Collections#unmodifiableSet(Set)
162       * unmodifiable} {@link Set} of {@link Object}s representing valid
163       * keys that can be used to {@linkplain #getGroup(Object) capture
164       * groups}.
165       *
166       * <p>This method never returns {@code null}.</p>
167       *
168       * @return a non-{@code null} {@link Set} of keys each element of
169       * which can be used as an argument to the {@link #getGroup(Object)}
170       * method
171       */
172      public Set<?> getGroupKeySet() {
173        final Set<?> returnValue;
174        if (this.thread == null) {
175          returnValue = Collections.emptySet();
176        } else {
177          returnValue = this.thread.getGroupKeySet();
178        }
179        return returnValue;
180      }
181    
182      /**
183       * Returns the value of a variable indexed under the supplied {@code
184       * key}.
185       *
186       * <p>This method may return {@code null}.</p>
187       *
188       * @param key the name of the variable; may be {@code null}
189       *
190       * @return the value of the variable, or {@code null} if there is no
191       * such variable
192       *
193       * @see #getVariables()
194       */
195      public Object getVariable(final Object key) {
196        Object result = null;
197        if (this.thread != null) {
198          final Map<?, ?> variables = this.getVariables();
199          if (variables != null) {
200            result = variables.get(key);
201          }
202        }
203        return result;
204      }
205    
206      /**
207       * Returns an {@linkplain Collections#unmodifiableMap(Map)
208       * unmodifiable <code>Map</code>} of the variables set on the {@link
209       * Thread} that this {@link MatchResult} wraps.
210       *
211       * <p>This method may return {@code null}.</p>
212       *
213       * @return a {@link Map} of variables, or {@code null}
214       */
215      public Map<?, ?> getVariables() {
216        Map<?, ?> result = null;
217        if (this.thread != null) {
218          result = this.thread.getVariables();
219        }
220        return result;
221      }
222    
223    }
