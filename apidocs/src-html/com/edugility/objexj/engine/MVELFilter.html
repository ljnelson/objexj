028    package com.edugility.objexj.engine;
029    
030    import java.util.HashMap;
031    import java.util.Map;
032    
033    import java.util.logging.Level;
034    import java.util.logging.Logger;
035    
036    import org.mvel2.CompileException; // for javadoc only
037    import org.mvel2.MVEL;
038    
039    import org.mvel2.integration.VariableResolverFactory; // for javadoc only
040    
041    import org.mvel2.integration.impl.MapVariableResolverFactory;
042    
043    /**
044     * A {@link Filter} that implements its {@link
045     * #accept(InstructionContext)} method by running a user-supplied <a
046     * href="http://mvel.codehaus.org/">MVEL</a> script against the {@link
047     * InstructionContext}'s {@linkplain InstructionContext#read() current
048     * <code>Object</code>}.
049     *
050     * @param <T> the type of {@link Object} that can be {@linkplain
051     * #accept(Object, Map) accepted}
052     *
053     * @author <a href="http://about.me/lairdnelson"
054     * target="_parent">Laird Nelson</a>
055     */
056    public class MVELFilter<T> extends Filter<T> {
057    
058      /**
059       * The version of this class for {@linkplain Serializable
060       * serialization purposes}.
061       */
062      private static final long serialVersionUID = 1L;
063    
064      /**
065       * The <a href="http://mvel.codehaus.org/">MVEL</a> source code used
066       * to implement the behavior of the {@link #accept(Object, Map)}
067       * method.
068       *
069       * <p>This field may be {@code null}.</p>
070       *
071       * @see #accept(Object, Map)
072       */
073      protected String mvelExpressionSource;
074    
075      /**
076       * The {@linkplain MVEL#compileExpression(String) compiled} form of
077       * the {@linkplain #mvelExpressionSource MVEL source code}.
078       *
079       * <p>This field may be {@code null}.</p>
080       *
081       * @see #mvelExpressionSource
082       */
083      protected Object mvelExpression;
084    
085      /**
086       * Creates a new {@link MVELFilter}.  This constructor is intended
087       * for subclasses only.  Subclasses should set the {@link
088       * #mvelExpressionSource} and {@link #mvelExpression} fields
089       * appropriately as soon as possible, and should treat these fields
090       * as though they were declared to be {@code final}.
091       */
092      protected MVELFilter() {
093        super();
094      }
095    
096      /**
097       * Creates a new {@link MVELFilter}.
098       *
099       * @param mvel the <a href="http://mvel.codehaus.org/">MVEL</a>
100       * source code; may be {@code null}
101       *
102       * @exception CompileException if the source code could not be
103       * compiled
104       */
105      public MVELFilter(final String mvel) {
106        super();
107        if (mvel == null) {
108          this.mvelExpression = null;
109          this.mvelExpressionSource = null;
110        } else {
111          this.mvelExpression = MVEL.compileExpression(mvel);
112          this.mvelExpressionSource = mvel;
113        }
114      }
115    
116      /**
117       * Accepts or rejects the supplied {@link InstructionContext} during
118       * execution by making sure that it {@linkplain
119       * InstructionContext#canRead() can be read from} and that the
120       * {@link #accept(Object, Map)} method also returns {@code true}.
121       *
122       * @param context the {@link InstructionContext} in which this
123       * {@link MVELFilter} is running; must not be {@code null}
124       *
125       * @exception IllegalArgumentException if {@code context} is {@code
126       * null}
127       *
128       * @exception CompileException if the {@link #accept(Object, Map)}
129       * method throws a {@link CompileException}
130       *
131       * @see #accept(Object, Map)
132       */
133      @Override
134      public boolean accept(final InstructionContext<? extends T> context) {
135        final String className = this.getClass().getName();
136        final Logger logger = this.getLogger();
137        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
138        if (finer) {
139          logger.entering(className, "accept", context);
140        }
141        if (context == null) {
142          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
143        }
144        final boolean returnValue;
145        if (this.mvelExpression == null) {
146          returnValue = context.canRead(); // no MVEL expression means no additional constraints
147        } else {
148          returnValue = context.canRead() && this.accept(context.read(), context.getVariables());
149        }
150        if (finer) {
151          logger.exiting(className, "accept", Boolean.valueOf(returnValue));
152        }
153        return returnValue;
154      }
155    
156      /**
157       * Returns {@code true} if the {@linkplain
158       * MVEL#executeExpression(Object, Object, VariableResolverFactory)
159       * execution} of the {@linkplain #mvelExpression MVEL expression}
160       * associated with this {@link MVELFilter} returned an {@link
161       * Object} {@linkplain Boolean#equals(Object) equal to} {@link
162       * Boolean#TRUE} when evaluated against the supplied {@code item}.
163       *
164       * @param item the {@link Object} that will serve as the context for
165       * the {@linkplain #mvelExpression MVEL expression} associated with
166       * this {@link MVELFilter} during its evaluation; may be {@code
167       * null}
168       *
169       * @param variables a {@link Map} of variables that may be affected
170       * by the {@linkplain #mvelExpression MVEL expression} associated
171       * with this {@link MVELFilter}; may be {@code null} in which case a
172       * new {@link HashMap} will be used instead
173       *
174       * @return {@code true} if the {@linkplain
175       * MVEL#executeExpression(Object, Object, VariableResolverFactory)
176       * execution} of the {@linkplain #mvelExpression MVEL expression}
177       * associated with this {@link MVELFilter} returned an {@link
178       * Object} {@linkplain Boolean#equals(Object) equal to} {@link
179       * Boolean#TRUE} when evaluated against the supplied {@code item};
180       * {@code false} otherwise
181       *
182       * @exception CompileException if the {@link #mvelExpression}
183       * field's value could be {@linkplain MVEL#compileExpression(String)
184       * compiled}, but referenced an object or a property that could not
185       * be resolved during {@linkplain MVEL#executeExpression(Object,
186       * Object, VariableResolverFactory) execution}; thrown by the {@link
187       * MVEL#executeExpression(Object, Object, VariableResolverFactory)}
188       * method
189       */
190      public boolean accept(final T item, Map<Object, Object> variables) {
191        final String className = this.getClass().getName();
192        final Logger logger = this.getLogger();
193        final boolean finer = logger != null && logger.isLoggable(Level.FINER);
194        if (finer) {
195          logger.entering(className, "accept", new Object[] { item, variables });
196        }
197        if (variables == null) {
198          variables = new HashMap<Object, Object>();
199        }
200        final boolean returnValue;
201        if (this.mvelExpression == null) {
202          returnValue = true;
203        } else {
204          final Object executionResult = MVEL.executeExpression(this.mvelExpression, item, new MapVariableResolverFactory(variables));
205          if (finer) {
206            logger.logp(Level.FINER, className, "accept", "Execution result: {0}", executionResult);
207          }
208          if (executionResult instanceof Boolean) {
209            returnValue = ((Boolean)executionResult).booleanValue();
210          } else {
211            returnValue = true;
212          }
213        }
214        if (finer) {
215          logger.exiting(className, "accept", Boolean.valueOf(returnValue));
216        }
217        return returnValue;
218      }
219    
220      /**
221       * Returns a hashcode for this {@link MVELFilter}.
222       *
223       * @return a hashcode for this {@link MVELFilter}
224       */
225      @Override
226      public int hashCode() {
227        if (this.mvelExpressionSource == null) {
228          return 0;
229        }
230        return this.mvelExpressionSource.hashCode();
231      }
232    
233      /**
234       * Returns {@code true} if the supplied {@link Object} is equal to
235       * this {@link MVELFilter}.
236       *
237       * @param other the {@link Object} to test; may be {@code null} in
238       * which case {@code false} will be returned
239       */
240      @Override
241      public boolean equals(final Object other) {
242        if (other == this) {
243          return true;
244        } else if (super.equals(other)) {
245          final MVELFilter him = (MVELFilter)other;
246          if (this.mvelExpressionSource == null) {
247            return him.mvelExpressionSource == null;
248          }
249          return this.mvelExpressionSource.equals(him.mvelExpressionSource);
250        } else {
251          return false;
252        }
253      }
254    
255      /**
256       * Returns a non-{@code null} {@link String} representation of this
257       * {@link MVELFilter}.  If possible, the {@linkplain
258       * #mvelExpressionSource source code} will be featured in the {@link
259       * String} that is returned.
260       *
261       * @return a non-{@code null} {@link String} representation of this
262       * {@link MVELFilter}.
263       */
264      @Override
265      public String toString() {
266        final StringBuilder sb = new StringBuilder(super.toString());
267        if (this.mvelExpressionSource != null) {
268          sb.append(" ").append(this.mvelExpressionSource);
269        }
270        return sb.toString();
271      }
