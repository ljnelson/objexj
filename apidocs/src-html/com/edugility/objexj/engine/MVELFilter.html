028    package com.edugility.objexj.engine;
029    
030    import java.util.HashMap;
031    import java.util.Map;
032    
033    import org.mvel2.MVEL;
034    
035    import org.mvel2.integration.impl.MapVariableResolverFactory;
036    
037    public class MVELFilter<T> extends Filter<T> {
038    
039      private static final long serialVersionUID = 1L;
040    
041      protected String mvelExpressionSource;
042    
043      protected Object mvelExpression;
044    
045      protected MVELFilter() {
046        super();
047      }
048      
049      public MVELFilter(final String mvel) {
050        super();
051        if (mvel == null) {
052          this.mvelExpression = null;
053          this.mvelExpressionSource = null;
054        } else {
055          this.mvelExpression = MVEL.compileExpression(mvel);
056          this.mvelExpressionSource = mvel;
057        }
058      }
059    
060      @Override
061      public boolean accept(final InstructionContext<T> context) {
062        if (context == null) {
063          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
064        }
065        final boolean returnValue;
066        if (this.mvelExpression == null) {
067          returnValue = context.canRead(); // no MVEL expression means no additional constraints
068        } else {
069          returnValue = context.canRead() && this.accept(context.read(), context.getVariables());
070        }
071        return returnValue;
072      }
073    
074      public boolean accept(final T item, Map<Object, Object> variables) {
075        if (variables == null) {
076          variables = new HashMap<Object, Object>();
077        }
078        final boolean returnValue;
079        if (this.mvelExpression == null) {
080          returnValue = true;
081        } else {
082          final Object executionResult = MVEL.executeExpression(this.mvelExpression, item, new MapVariableResolverFactory(variables));
083          returnValue = executionResult instanceof Boolean && ((Boolean)executionResult).booleanValue();
084        }
085        return returnValue;
086      }
087    
088      @Override
089      public int hashCode() {
090        if (this.mvelExpressionSource == null) {
091          return 0;
092        }
093        return this.mvelExpressionSource.hashCode();
094      }
095    
096      @Override
097      public boolean equals(final Object other) {
098        if (other == this) {
099          return true;
100        } else if (super.equals(other)) {
101          final MVELFilter him = (MVELFilter)other;
102          if (this.mvelExpressionSource == null) {
103            return him.mvelExpressionSource == null;
104          }
105          return this.mvelExpressionSource.equals(him.mvelExpressionSource);
106        } else {
107          return false;
108        }
109      }
110    
111      @Override
112      public String toString() {
113        final StringBuilder sb = new StringBuilder(super.toString());
114        if (this.mvelExpressionSource != null) {
115          sb.append(" ").append(this.mvelExpressionSource);
116        }
117        return sb.toString();
118      }
119    
120    }
