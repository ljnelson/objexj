028    package com.edugility.objexj.engine;
029    
030    import java.util.HashMap;
031    import java.util.Map;
032    
033    import org.mvel2.CompileException; // for javadoc only
034    import org.mvel2.MVEL;
035    
036    import org.mvel2.integration.VariableResolverFactory; // for javadoc only
037    
038    import org.mvel2.integration.impl.MapVariableResolverFactory;
039    
040    /**
041     * A {@link Filter} that implements its {@link
042     * #accept(InstructionContext)} method by running a user-supplied <a
043     * href="http://mvel.codehaus.org/">MVEL</a> script against the {@link
044     * InstructionContext}'s {@linkplain InstructionContext#read() current
045     * <code>Object</code>}.
046     *
047     * @param <T> the type of {@link Object} that can be {@linkplain
048     * #accept(Object, Map) accepted}
049     *
050     * @author <a href="http://about.me/lairdnelson"
051     * target="_parent">Laird Nelson</a>
052     */
053    public class MVELFilter<T> extends Filter<T> {
054    
055      /**
056       * The version of this class for {@linkplain Serializable
057       * serialization purposes}.
058       */
059      private static final long serialVersionUID = 1L;
060    
061      /**
062       * The <a href="http://mvel.codehaus.org/">MVEL</a> source code used
063       * to implement the behavior of the {@link #accept(Object, Map)}
064       * method.
065       *
066       * <p>This field may be {@code null}.</p>
067       *
068       * @see #accept(Object, Map)
069       */
070      protected String mvelExpressionSource;
071    
072      /**
073       * The {@linkplain MVEL#compileExpression(String) compiled} form of
074       * the {@linkplain #mvelExpressionSource MVEL source code}.
075       *
076       * <p>This field may be {@code null}.</p>
077       *
078       * @see #mvelExpressionSource
079       */
080      protected Object mvelExpression;
081    
082      /**
083       * Creates a new {@link MVELFilter}.  This constructor is intended
084       * for subclasses only.  Subclasses should set the {@link
085       * #mvelExpressionSource} and {@link #mvelExpression} fields
086       * appropriately as soon as possible, and should treat these fields
087       * as though they were declared to be {@code final}.
088       */
089      protected MVELFilter() {
090        super();
091      }
092      
093      /**
094       * Creates a new {@link MVELFilter}.
095       *
096       * @param mvel the <a href="http://mvel.codehaus.org/">MVEL</a>
097       * source code; may be {@code null}
098       *
099       * @exception CompileException if the source code could not be
100       * compiled
101       */
102      public MVELFilter(final String mvel) {
103        super();
104        if (mvel == null) {
105          this.mvelExpression = null;
106          this.mvelExpressionSource = null;
107        } else {
108          this.mvelExpression = MVEL.compileExpression(mvel);
109          this.mvelExpressionSource = mvel;
110        }
111      }
112    
113      /**
114       * Accepts or rejects the supplied {@link InstructionContext} during
115       * execution by making sure that it {@linkplain
116       * InstructionContext#canRead() can be read from} and that the
117       * {@link #accept(Object, Map)} method also returns {@code true}.
118       *
119       * @param context the {@link InstructionContext} in which this
120       * {@link MVELFilter} is running; must not be {@code null}
121       *
122       * @exception IllegalArgumentException if {@code context} is {@code
123       * null}
124       *
125       * @exception CompileException if the {@link #accept(Object, Map)}
126       * method throws a {@link CompileException}
127       *
128       * @see #accept(Object, Map)
129       */
130      @Override
131      public boolean accept(final InstructionContext<? extends T> context) {
132        if (context == null) {
133          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
134        }
135        final boolean returnValue;
136        if (this.mvelExpression == null) {
137          returnValue = context.canRead(); // no MVEL expression means no additional constraints
138        } else {
139          returnValue = context.canRead() && this.accept(context.read(), context.getVariables());
140        }
141        return returnValue;
142      }
143    
144      /**
145       * Returns {@code true} if the {@linkplain
146       * MVEL#executeExpression(Object, Object, VariableResolverFactory)
147       * execution} of the {@linkplain #mvelExpression MVEL expression}
148       * associated with this {@link MVELFilter} returned an {@link
149       * Object} {@linkplain Boolean#equals(Object) equal to} {@link
150       * Boolean#TRUE} when evaluated against the supplied {@code item}.
151       *
152       * @param item the {@link Object} that will serve as the context for
153       * the {@linkplain #mvelExpression MVEL expression} associated with
154       * this {@link MVELFilter} during its evaluation; may be {@code
155       * null}
156       *
157       * @param variables a {@link Map} of variables that may be affected
158       * by the {@linkplain #mvelExpression MVEL expression} associated
159       * with this {@link MVELFilter}; may be {@code null} in which case a
160       * new {@link HashMap} will be used instead
161       * 
162       * @return {@code true} if the {@linkplain
163       * MVEL#executeExpression(Object, Object, VariableResolverFactory)
164       * execution} of the {@linkplain #mvelExpression MVEL expression}
165       * associated with this {@link MVELFilter} returned an {@link
166       * Object} {@linkplain Boolean#equals(Object) equal to} {@link
167       * Boolean#TRUE} when evaluated against the supplied {@code item};
168       * {@code false} otherwise
169       *
170       * @exception CompileException if the {@link #mvelExpression}
171       * field's value could be {@linkplain MVEL#compileExpression(String)
172       * compiled}, but referenced an object or a property that could not
173       * be resolved during {@linkplain MVEL#executeExpression(Object,
174       * Object, VariableResolverFactory) execution}; thrown by the {@link
175       * MVEL#executeExpression(Object, Object, VariableResolverFactory)}
176       * method
177       */
178      public boolean accept(final T item, Map<Object, Object> variables) {
179        if (variables == null) {
180          variables = new HashMap<Object, Object>();
181        }
182        final boolean returnValue;
183        if (this.mvelExpression == null) {
184          returnValue = true;
185        } else {
186          final Object executionResult = MVEL.executeExpression(this.mvelExpression, item, new MapVariableResolverFactory(variables));
187          returnValue = executionResult instanceof Boolean && ((Boolean)executionResult).booleanValue();
188        }
189        return returnValue;
190      }
191    
192      /**
193       * Returns a hashcode for this {@link MVELFilter}.
194       *
195       * @return a hashcode for this {@link MVELFilter}
196       */
197      @Override
198      public int hashCode() {
199        if (this.mvelExpressionSource == null) {
200          return 0;
201        }
202        return this.mvelExpressionSource.hashCode();
203      }
204    
205      /**
206       * Returns {@code true} if the supplied {@link Object} is equal to
207       * this {@link MVELFilter}.
208       *
209       * @param other the {@link Object} to test; may be {@code null} in
210       * which case {@code false} will be returned
211       */
212      @Override
213      public boolean equals(final Object other) {
214        if (other == this) {
215          return true;
216        } else if (super.equals(other)) {
217          final MVELFilter him = (MVELFilter)other;
218          if (this.mvelExpressionSource == null) {
219            return him.mvelExpressionSource == null;
220          }
221          return this.mvelExpressionSource.equals(him.mvelExpressionSource);
222        } else {
223          return false;
224        }
225      }
226    
227      /**
228       * Returns a non-{@code null} {@link String} representation of this
229       * {@link MVELFilter}.  If possible, the {@linkplain
230       * #mvelExpressionSource source code} will be featured in the {@link
231       * String} that is returned.
232       *
233       * @return a non-{@code null} {@link String} representation of this
234       * {@link MVELFilter}.
235       */
236      @Override
237      public String toString() {
238        final StringBuilder sb = new StringBuilder(super.toString());
239        if (this.mvelExpressionSource != null) {
240          sb.append(" ").append(this.mvelExpressionSource);
241        }
242        return sb.toString();
243      }
244
