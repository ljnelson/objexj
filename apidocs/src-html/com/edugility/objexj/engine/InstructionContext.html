028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.concurrent.atomic.AtomicInteger;
033    
034    import java.util.List;
035    import java.util.Map;
036    
037    public class InstructionContext<T> implements Serializable {
038    
039      private static final long serialVersionUID = 1L;
040    
041      private static final AtomicInteger idGenerator = new AtomicInteger(1);
042    
043      private final Thread<T> thread;
044    
045      InstructionContext(final Thread<T> thread) {
046        super();
047        if (thread == null) {
048          throw new IllegalArgumentException("thread == null");
049        }
050        this.thread = thread;
051      }
052    
053      public Map<Object, Object> getVariables() {
054        return this.thread.getVariables();
055      }
056    
057      public final int getProgramCounterIndex() {
058        final ProgramCounter<T> pc = this.thread.getProgramCounter();
059        assert pc != null;
060        return pc.getIndex();
061      }
062    
063      public final boolean scheduleNewThread(final int programCounterIndex, final boolean relative) {
064        return this.thread.schedule(this.thread.newThread(idGenerator.getAndIncrement(), programCounterIndex, relative));
065      }
066    
067      public final boolean jump(final int programCounter, final boolean relative) {
068        return this.thread.jump(programCounter, relative);
069      }
070    
071      public final void save(final Object key) {
072        this.thread.save(key);
073      }
074    
075      public final void stop(final Object key) {
076        this.thread.stop(key);
077      }
078    
079      public final boolean atStart() {
080        return this.thread.atStart();
081      }
082    
083      public final boolean atEnd() {
084        return this.thread.atEnd();
085      }
086    
087      public final boolean advanceProgramCounter() {
088        return this.thread.advanceProgramCounter();
089      }
090    
091      public final boolean advanceItemPointer() {
092        return this.thread.advanceItemPointer();
093      }
094    
095      public final boolean canRead() {
096        return this.thread.canRead();
097      }
098    
099      public final T read() {
100        return this.thread.read();
101      }
102    
103      public final void die() {
104        this.thread.die();
105      }
106    
107      public final boolean isDead() {
108        return Thread.State.DEAD == this.thread.getState();
109      }
110    
111      public final void match() {
112        this.thread.match();
113      }
114    
115      @Deprecated
116      private final Thread<T> getThread() {
117        return this.thread;
118      }
119    
120    }
