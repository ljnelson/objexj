028    package com.edugility.objexj.engine;
029    
030    import java.util.Map;
031    
032    import java.util.regex.Matcher;
033    import java.util.regex.Pattern;
034    
035    import org.mvel2.MVEL;
036    
037    /**
038     * An {@link MVELFilter} that conveniently checks the {@linkplain
039     * InstructionContext#read() current item} to see if it is an instance
040     * of a specified class before applying further MVEL-based filters to
041     * it.
042     *
043     * @author <a href="http://about.me/lairdnelson">Laird Nelson</a>
044     */
045    public class InstanceOfMVELFilter<T> extends MVELFilter<T> {
046    
047      private static final long serialVersionUID = 1L;
048    
049      private static final Pattern OPERAND_PATTERN = Pattern.compile("^\\s*([^\\s]+)\\s*(.*)");
050    
051      private final Class<?> cls;
052    
053      private final boolean exact;
054    
055      /**
056       * Creates a new {@link NotNullMVELFilter}.
057       *
058       * @param operands the operands to parse; must not be {@code null}
059       *
060       * @exception IllegalArgumentException if {@code operands} is {@code
061       * null}
062       */
063      public InstanceOfMVELFilter(final String operands) {
064        super();
065        if (operands == null) {
066          throw new IllegalArgumentException("operands", new NullPointerException("operands"));
067        }
068        final Matcher m = OPERAND_PATTERN.matcher(operands);
069        assert m != null;
070        if (!m.find()) {
071          throw new IllegalArgumentException("Bad operands: " + operands);
072        }
073        String className = m.group(1);
074        assert className != null;
075        this.exact = className.charAt(0) == '=';
076        if (this.exact) {
077          className = className.substring(1);
078        }
079    
080        Class<?> c = null;
081        try {
082          c = this.loadClass(className);
083        } catch (final ClassNotFoundException cnfe) {
084          throw new IllegalArgumentException("Bad operands: " + operands, cnfe);
085        } finally {
086          this.cls = c;
087        }
088    
089        final String mvel = m.group(2);
090        if (mvel == null) {
091          this.mvelExpression = null;
092          this.mvelExpressionSource = null;
093        } else {
094          this.mvelExpression = MVEL.compileExpression(mvel);
095          this.mvelExpressionSource = mvel;
096        }
097      }
098    
099      public InstanceOfMVELFilter(String className, final String mvel) {
100        super();
101        if (className == null) {
102          throw new IllegalArgumentException("className", new NullPointerException("className"));
103        }
104        this.exact = className.charAt(0) == '=';
105        if (this.exact) {
106          className = className.substring(1);
107        }
108        Class<?> c = null;
109        try {
110          c = this.loadClass(className);
111        } catch (final ClassNotFoundException cnfe) {
112          throw new IllegalArgumentException("className", cnfe);
113        } finally {
114          this.cls = c;
115        }
116        if (mvel == null) {
117          this.mvelExpression = null;
118          this.mvelExpressionSource = null;
119        } else {
120          this.mvelExpression = MVEL.compileExpression(mvel);
121          this.mvelExpressionSource = mvel;
122        }
123      }
124    
125      public InstanceOfMVELFilter(final Class<?> c) {
126        this(c, false, null);
127      }
128    
129      public InstanceOfMVELFilter(final Class<?> c, final String mvel) {
130        this(c, false, mvel);
131      }
132    
133      public InstanceOfMVELFilter(final Class<?> c, final boolean exact, final String mvel) {
134        super();
135        if (c == null) {
136          throw new IllegalArgumentException("c", new NullPointerException("c"));
137        }
138        this.cls = c;
139        this.exact = exact;
140        if (mvel == null) {
141          this.mvelExpression = null;
142          this.mvelExpressionSource = null;
143        } else {
144          this.mvelExpression = MVEL.compileExpression(mvel);
145          this.mvelExpressionSource = mvel;
146        }
147      }
148    
149      public boolean isExact() {
150        return this.exact;
151      }
152    
153      @Override
154      public boolean accept(final InstructionContext<T> context) {
155        if (context == null) {
156          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
157        }
158        if (this.mvelExpression == null) {
159          return super.accept(context);
160        }
161        return this.cls != null && context.canRead() && this.accept(context.read(), context.getVariables());
162      }
163    
164      @Override
165      public boolean accept(final T item, Map<Object, Object> variables) {
166        return
167          item != null && 
168          this.cls != null &&
169          this.isExact() ? item.getClass().equals(this.cls) : this.cls.isInstance(item) &&
170          super.accept(item, variables);
171      }
172    
173      @Override
174      public int hashCode() {
175        assert this.cls != null;
176        return 37 * super.hashCode() + this.cls.hashCode() + (this.exact ? 1 : 0);
177      }
178    
179      @Override
180      public boolean equals(final Object other) {
181        if (other == this) {
182          return true;
183        } else if (super.equals(other)) {
184          final InstanceOfMVELFilter him = (InstanceOfMVELFilter)other;
185          return this.exact == him.exact && this.cls.equals(him.cls);
186        } else {
187          return false;
188        }
189      }
190    
191      @Override
192      public String toString() {
193        final StringBuilder sb = new StringBuilder(this.getClass().getSimpleName());
194        sb.append(" ");
195        if (this.isExact()) {
196          sb.append("=");
197        }
198        sb.append(this.cls.getName());
199        if (this.mvelExpressionSource != null) {
200          sb.append(" ").append(this.mvelExpressionSource);
201        }
202        return sb.toString();
203      }
204    
205    }
