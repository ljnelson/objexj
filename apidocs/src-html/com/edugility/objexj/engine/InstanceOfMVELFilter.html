028    package com.edugility.objexj.engine;
029    
030    import java.util.Map;
031    
032    import java.util.regex.Matcher;
033    import java.util.regex.Pattern;
034    
035    import org.mvel2.MVEL;
036    
037    /**
038     * An {@link MVELFilter} that conveniently checks the {@linkplain
039     * InstructionContext#read() current item} to see if it is an instance
040     * of a specified {@link Class} before applying further <a
041     * href="http://mvel.codehaus.org/">MVEL</a>-based filters to it.
042     *
043     * @param <T> the type of {@link Object} that can be {@linkplain
044     * #accept(Object, Map) accepted}
045     *
046     * @author <a href="http://about.me/lairdnelson"
047     * target="_parent">Laird Nelson</a>
048     */
049    public class InstanceOfMVELFilter<T> extends MVELFilter<T> {
050    
051      /**
052       * The version of this class for {@linkplain Serializable
053       * serialization purposes}.
054       */
055      private static final long serialVersionUID = 1L;
056    
057      /**
058       * A {@link Pattern} for parsing a single {@link String} for the
059       * operands logically taken by this {@link InstanceOfMVELFilter}.
060       *
061       * <p>This field is never {@code null}.</p>
062       *
063       * <p>The textual form of the {@link Pattern} is: {@code ^\\s*([^\\s]+)\\s*(.*)}.</p>
064       *
065       * @see java.util.regex.Pattern#compile(String)
066       */
067      private static final Pattern OPERAND_PATTERN = Pattern.compile("^\\s*([^\\s]+)\\s*(.*)");
068    
069      /**
070       * The {@link Class} whose {@link Class#isInstance(Object)} method
071       * will be called.
072       *
073       * <p>This field is never {@code null}.</p>
074       */
075      private final Class<?> cls;
076    
077      /**
078       * Determines whether an {@link Object} will be checked to see if
079       * {@linkplain Object#getClass() its <code>Class</code>} is equal to
080       * {@linkplain #InstanceOfMVELFilter(Class, String) the
081       * <code>Class</code> supplied at construction time} or merely an
082       * {@linkplain Class#isInstance(Object) an instance of it}.
083       */
084      private final boolean exact;
085    
086      /**
087       * Creates a new {@link InstanceOfMVELFilter}.
088       *
089       * @param operands the operands to parse represented as a single
090       * {@link String}; must not be {@code null}
091       *
092       * @exception IllegalArgumentException if {@code operands} is {@code
093       * null} or if a {@link Class} could not be loaded because part of
094       * the {@code operands} {@link String} was a class name that could
095       * not be {@linkplain Instruction#loadClass(String) loaded}
096       */
097      public InstanceOfMVELFilter(final String operands) {
098        super();
099        if (operands == null) {
100          throw new IllegalArgumentException("operands", new NullPointerException("operands"));
101        }
102        final Matcher m = OPERAND_PATTERN.matcher(operands);
103        assert m != null;
104        if (!m.find()) {
105          throw new IllegalArgumentException("Bad operands: " + operands);
106        }
107        String className = m.group(1);
108        assert className != null;
109        this.exact = className.charAt(0) == '=';
110        if (this.exact) {
111          className = className.substring(1);
112        }
113    
114        Class<?> c = null;
115        try {
116          c = this.loadClass(className);
117        } catch (final ClassNotFoundException cnfe) {
118          throw new IllegalArgumentException("Bad operands: " + operands, cnfe);
119        } finally {
120          this.cls = c;
121        }
122    
123        final String mvel = m.group(2);
124        if (mvel == null) {
125          this.mvelExpression = null;
126          this.mvelExpressionSource = null;
127        } else {
128          this.mvelExpression = MVEL.compileExpression(mvel);
129          this.mvelExpressionSource = mvel;
130        }
131      }
132    
133      /**
134       * Creates a new {@link InstanceOfMVELFilter}.
135       *
136       * @param className the name of a {@link Class} against which {@link
137       * Object}s will be {@linkplain #accept(InstructionContext)
138       * checked}; must not be {@code null}; may start with an {@code =}
139       * character, in which case a class equality test will be used
140       * (otherwise {@link Class#isInstance(Object)} is used)
141       *
142       * @param mvel an <a href="http://mvel.codehaus.org/">MVEL</a>
143       * expression; may be {@code null}
144       *
145       * @exception IllegalArgumentException if {@code className} is
146       * {@code null} or designates a {@link Class} that could not be
147       * {@linkplain Instruction#loadClass(String) loaded}
148       */
149      public InstanceOfMVELFilter(String className, final String mvel) {
150        super();
151        if (className == null) {
152          throw new IllegalArgumentException("className", new NullPointerException("className"));
153        }
154        this.exact = className.charAt(0) == '=';
155        if (this.exact) {
156          className = className.substring(1);
157        }
158        Class<?> c = null;
159        try {
160          c = this.loadClass(className);
161        } catch (final ClassNotFoundException cnfe) {
162          throw new IllegalArgumentException("className", cnfe);
163        } finally {
164          this.cls = c;
165        }
166        if (mvel == null) {
167          this.mvelExpression = null;
168          this.mvelExpressionSource = null;
169        } else {
170          this.mvelExpression = MVEL.compileExpression(mvel);
171          this.mvelExpressionSource = mvel;
172        }
173      }
174    
175      /**
176       * Invokes the {@link #InstanceOfMVELFilter(Class, boolean, String)}
177       * constructor passing the supplied {@link Class}, {@code false} and
178       * {@code null}, thereby creating a new {@link
179       * InstanceOfMVELFilter}.
180       *
181       * @param c the {@link Class} to use for comparison tests; must not
182       * be {@code null}
183       *
184       * @exception IllegalArgumentException if {@code c} is {@code null}
185       */
186      public InstanceOfMVELFilter(final Class<?> c) {
187        this(c, false, null);
188      }
189    
190      /**
191       * Invokes the {@link #InstanceOfMVELFilter(Class, boolean, String)}
192       * constructor passing the supplied {@link Class}, {@code false} and
193       * the supplied <a href="http://mvel.codehaus.org/">MVEL</a> expression, thereby creating a new {@link
194       * InstanceOfMVELFilter}.
195       *
196       * @param c the {@link Class} to use for comparison tests; must not
197       * be {@code null}
198       *
199       * @param mvel the <a href="http://mvel.codehaus.org/">MVEL</a>
200       * expression; may be {@code null}
201       *
202       * @exception IllegalArgumentException if {@code c} is {@code null}
203       */
204      public InstanceOfMVELFilter(final Class<?> c, final String mvel) {
205        this(c, false, mvel);
206      }
207    
208      /**
209       * Creates a new {@link InstanceOfMVELFilter}.
210       *
211       * @param c the {@link Class} to use for comparison tests; must not
212       * be {@code null}
213       *
214       * @param exact if {@code true}, then the {@link Class} comparison
215       * test used will be {@linkplain Object#equals(Object) equality};
216       * otherwise it will be {@linkplain Class#isInstance(Object)
217       * <code>Class</code> membership}
218       *
219       * @param mvel the <a href="http://mvel.codehaus.org/">MVEL</a>
220       * expression; may be {@code null}
221       *
222       * @exception IllegalArgumentException if {@code c} is {@code null}
223       *
224       * @see <a href="http://mvel.codehaus.org/">The MVEL expression
225       * language</a>
226       */
227      public InstanceOfMVELFilter(final Class<?> c, final boolean exact, final String mvel) {
228        super();
229        if (c == null) {
230          throw new IllegalArgumentException("c", new NullPointerException("c"));
231        }
232        this.cls = c;
233        this.exact = exact;
234        if (mvel == null) {
235          this.mvelExpression = null;
236          this.mvelExpressionSource = null;
237        } else {
238          this.mvelExpression = MVEL.compileExpression(mvel);
239          this.mvelExpressionSource = mvel;
240        }
241      }
242    
243      /**
244       * Returns {@code true} if {@link Class} equality will be the
245       * comparison test used against input {@link Object}s.
246       *
247       * @return {@code true} if {@link Class} equality will be the
248       * comparison test used against input {@link Object}s; {@code false}
249       * if {@linkplain Class#isInstance(Object) class membership} will be
250       * used instead
251       */
252      public boolean isExact() {
253        return this.exact;
254      }
255    
256      /**
257       * Returns {@code true} if this {@link InstanceOfMVELFilter}
258       * notionally accepts the supplied {@link InstructionContext}.
259       *
260       * @return {@code true} if the supplied {@link InstructionContext}
261       * is non-{@code null}, {@linkplain InstructionContext#canRead() can
262       * be read from}, and if the {@link #accept(Object, Map)} method
263       * returns {@code true} as well
264       *
265       * @exception IllegalArgumentException if {@code context} is {@code
266       * null}
267       */
268      @Override
269      public boolean accept(final InstructionContext<T> context) {
270        if (context == null) {
271          throw new IllegalArgumentException("context", new NullPointerException("context == null"));
272        }
273        if (this.mvelExpression == null) {
274          return super.accept(context);
275        }
276        return this.cls != null && context.canRead() && this.accept(context.read(), context.getVariables());
277      }
278    
279      /**
280       * Returns {@code true} if the supplied {@code item} is non-{@code
281       * null} and {@linkplain Object#getClass() has a <code>Class</code>}
282       * that compares properly with the {@link Class} {@link
283       * #InstanceOfMVELFilter(Class, boolean, String) supplied at
284       * construction time}.
285       *
286       * <p>{@link Class} comparison is controlled by the return value of
287       * the {@link #isExact()} method.  If it is {@code true}, then the
288       * supplied {@code item} must {@linkplain Object#getClass() have a
289       * <code>Class</code>} that is equal to {@linkplain
290       * #InstanceOfMVELFilter(Class, String) the <code>Class</code>
291       * supplied at construction time}.  If it is {@code false}, then the
292       * supplied {@code item} must merely be an {@linkplain
293       * Class#isInstance(Object) instance of} that {@link Class}.</p>
294       *
295       * @param item the {@link Object} to be accepted; may be {@code
296       * null} in which case {@code false} will be returned
297       *
298       * @param variables a {@link Map} of variables that may be populated
299       * by the <a href="http://mvel.codehaus.org/">MVEL</a> expression
300       * {@linkplain #InstanceOfMVELFilter(Class, boolean, String)
301       * supplied at construction time}
302       *
303       * @see #isExact()
304       *
305       * @see #InstanceOfMVELFilter(Class, boolean, String)
306       */
307      @Override
308      public boolean accept(final T item, Map<Object, Object> variables) {
309        return
310          item != null && 
311          this.cls != null &&
312          this.isExact() ? item.getClass().equals(this.cls) : this.cls.isInstance(item) &&
313          super.accept(item, variables);
314      }
315    
316      /**
317       * Returns a hashcode for this {@link InstanceOfMVELFilter}.
318       *
319       * @return a hashcode for this {@link InstanceOfMVELFilter}
320       */
321      @Override
322      public int hashCode() {
323        assert this.cls != null;
324        return 37 * super.hashCode() + this.cls.hashCode() + (this.exact ? 1 : 0);
325      }
326    
327      /**
328       * Returns {@code true} if the supplied {@link Object} is equal to
329       * this {@link InstanceOfMVELFilter}.
330       *
331       * @param other the {@link Object} to test; may be {@code null} in
332       * which case {@code false} will be returned
333       *
334       * @return {@code true} if the supplied {@link Object} is non-{@code
335       * null}, {@linkplain Object#getClass() has a <code>Class</code>}
336       * that is equal to this {@link InstanceOfMVELFilter}'s {@link
337       * Class}, has the same return value from its {@link #isExact()}
338       * method, and has the same {@link Class} supplied to it at
339       * {@linkplain #InstanceOfMVELFilter(Class, boolean, String)
340       * construction time}; {@code false} otherwise
341       */
342      @Override
343      public boolean equals(final Object other) {
344        if (other == this) {
345          return true;
346        } else if (super.equals(other)) {
347          final InstanceOfMVELFilter him = (InstanceOfMVELFilter)other;
348          return this.exact == him.exact && this.cls.equals(him.cls);
349        } else {
350          return false;
351        }
352      }
353    
354      /**
355       * Returns a non-{@code null} {@link String} representation of this
356       * {@link InstanceOfMVELFilter}.
357       *
358       * @return a non-{@code null} {@link String} representation of this
359       * {@link InstanceOfMVELFilter}
360       */
361      @Override
362      public String toString() {
363        final StringBuilder sb = new StringBuilder(this.getClass().getSimpleName());
364        sb.append(" ");
365        if (this.isExact()) {
366          sb.append("=");
367        }
368        sb.append(this.cls.getName());
369        if (this.mvelExpressionSource != null) {
370          sb.append(" ").append(this.mvelExpressionSource);
371        }
372        return sb.toString();
373      }
