028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.AbstractList; // for javadoc only
033    import java.util.Iterator;
034    import java.util.NoSuchElementException;
035    
036    /**
037     * A marriage of a {@link Program} and an index into that {@link
038     * Program}, and hence also an {@link Iterator} and {@link Iterable}
039     * of {@link Instruction}s.
040     *
041     * @param <T> the type of {@link Object} processed by the {@link
042     * Program} affiliated with {@link ProgramCounter} instances
043     *
044     * @author <a href="http://about.me/lairdnelson"
045     * target="_parent">Laird Nelson</a>
046     *
047     * @see Program
048     *
049     * @see Instruction
050     */
051    public class ProgramCounter<T> implements Cloneable, Serializable, Iterable<Instruction<T>>, Iterator<Instruction<T>> {
052    
053      /**
054       * The version of this class for {@link Serializable serialization
055       * purposes}.
056       */
057      private static final long serialVersionUID = 1L;
058    
059      /**
060       * The index managed by this {@link ProgramCounter}.
061       */
062      private int index;
063    
064      /**
065       * The {@link Program} managed by this {@link ProgramCounter}.
066       *
067       * <p>This field will never be {@code null}.</p>
068       */
069      private final Program<T> program;
070    
071      /**
072       * Creates a new {@link ProgramCounter}.
073       *
074       * @param program the {@link Program} to be affiliated with this
075       * {@link ProgramCounter}; must not be {@code null}
076       *
077       * @exception IllegalArgumentException if {@code program} is {@code
078       * null}
079       */
080      public ProgramCounter(final Program<T> program) {
081        this(program, 0);
082      }
083    
084      /**
085       * Creates a new {@link ProgramCounter}.
086       *
087       * @param programCounter a {@link ProgramCounter} whose {@linkplain
088       * #getProgram() affiliated <code>Program</code>} will be used and
089       * whose {@linkplain #getIndex() index} will also be used; must not
090       * be {@code null}
091       *
092       * @exception IllegalArgumentException if {@code programCounter} is
093       * {@code null} or {@linkplain #getProgram() somehow has a
094       * <code>Program</code>} that is {@code null}
095       */
096      public ProgramCounter(final ProgramCounter<T> programCounter) {
097        this(programCounter == null ? null : programCounter.getProgram(), programCounter == null ? 0 : programCounter.getIndex());
098      }
099    
100      /**
101       * Creates a new {@link ProgramCounter}.
102       *
103       * @param program the {@link Program} to be affiliated with this
104       * {@link ProgramCounter}; must not be {@code null}
105       *
106       * @param index the index to be {@linkplain #setIndex(int)
107       * affiliated with} this {@link ProgramCounter}
108       *
109       * @exception IllegalArgumentException if {@code program} is {@code
110       * null}, or if the supplied {@code index} {@linkplain #isValid(int)
111       * is not valid}
112       *
113       * @see #isValid(int)
114       *
115       * @see #setIndex(int)
116       */
117      public ProgramCounter(final Program<T> program, final int index) {
118        super();
119        if (program == null) {
120          throw new IllegalArgumentException("program", new NullPointerException("program == null"));
121        }
122        this.program = program;
123        this.setIndex(index);
124      }
125    
126    
127      /**
128       * Returns a {@linkplain #clone() clone} of this {@link
129       * ProgramCounter} which is itself an {@link Iterator} of {@link
130       * Instruction}s.
131       *
132       * @return a non-{@code null} {@link Iterator} of {@link
133       * Instruction}s not identical to this {@link ProgramCounter}
134       */
135      @Override
136      public final Iterator<Instruction<T>> iterator() {
137        return this.clone();
138      }
139    
140      /**
141       * Returns the result of invoking the {@link #canAdvance()} method.
142       *
143       * @return the result of invoking the {@link #canAdvance()} method
144       */
145      @Override
146      public final boolean hasNext() {
147        return this.canAdvance();
148      }
149    
150      /**
151       * Throws an {@link UnsupportedOperationException} when invoked.
152       */
153      @Override
154      public final void remove() {
155        throw new UnsupportedOperationException();
156      }
157    
158      /**
159       * Calls the {@link #advance()} method, and if it returns {@code
160       * true}, returns the result of calling the {@link
161       * #getInstruction()} method.  Otherwise throws a {@link
162       * NoSuchElementException}.
163       *
164       * @return an {@link Instruction}, or {@code null}
165       *
166       * @exception NoSuchElementException if there are no further {@link
167       * Instruction}s
168       *
169       * @see #advance()
170       *
171       * @see #getInstruction()
172       *
173       * @see #hasNext()
174       */
175      @Override
176      public final Instruction<T> next() {
177        if (this.advance()) {
178          return this.getInstruction();
179        }
180        throw new NoSuchElementException();
181      }
182    
183      /**
184       * Returns the index managed by this {@link ProgramCounter}.
185       *
186       * <p>The index returned by this method is guaranteed to be
187       * {@linkplain #isValid(int) valid}.  Subclasses who choose to
188       * override this method must preserve these semantics.</p>
189       *
190       * @return the (valid) index managed by this {@link ProgramCounter}
191       *
192       * @see #setIndex(int)
193       */
194      public int getIndex() {
195        return this.index;
196      }
197    
198      /**
199       * Sets the index managed by this {@link ProgramCounter} to the
200       * supplied {@code index}, provided that it is {@linkplain
201       * #isValid(int) valid}.
202       *
203       * @param index the new index; must be {@linkplain #isValid(int)
204       * valid}
205       *
206       * @exception IllegalArgumentException if {@code index} {@linkplain
207       * #isValid(int) not valid}
208       */
209      public final void setIndex(final int index) {
210        if (!this.isValid(index)) {
211          throw new IllegalArgumentException("index: " + index);
212        }
213        this.index = index;
214      }
215    
216      /**
217       * Returns {@code true} if this {@link ProgramCounter} can advance
218       * its internal index by one.
219       *
220       * @return {@code true} if the {@linkplain #getIndex() current
221       * index} plus one {@linkplain #isValid(int) is valid}; {@code
222       * false} otherwise
223       */
224      public final boolean canAdvance() {
225        return this.isValid(this.getIndex() + 1);
226      }
227      
228      /**
229       * Attempts to advance the index and returns {@code true} if the
230       * attempt was successful.
231       *
232       * @return {@code true} if this {@link ProgramCounter} was in a
233       * state where it {@linkplain #canAdvance() could advance} and did;
234       * {@code false} otherwise
235       */
236      public boolean advance() {
237        final boolean returnValue;
238        final int newIndex = this.getIndex() + 1;
239        if (this.isValid(newIndex)) {
240          this.setIndex(newIndex);
241          returnValue = true;
242        } else {
243          returnValue = false;
244        }
245        return returnValue;
246      }
247    
248      /**
249       * Returns the {@link Instruction} located at the {@linkplain
250       * #getIndex() current index}, or {@code null} if no such {@link
251       * Instruction} was found.
252       *
253       * <p>This method may return {@code null}.</p>
254       *
255       * @return the {@link Instruction} at the {@linkplain #getIndex()
256       * current index}, or {@code null}
257       *
258       * @exception InvalidProgramCounterException if this {@link
259       * ProgramCounter} {@linkplain #isValid() is not valid}
260       */
261      public final Instruction<T> getInstruction() {
262        if (!this.isValid()) {
263          throw new InvalidProgramCounterException();
264        }
265        return this.getProgram().get(this.getIndex());
266      }
267    
268      /**
269       * Returns {@code true} if this {@link ProgramCounter} is valid.
270       *
271       * @return {@code true} if the {@link #isValid(int)} method returns
272       * {@code true} when supplied with the {@link #getIndex() current
273       * index}; {@code false} otherwise
274       */
275      public final boolean isValid() {
276        return this.isValid(this.getIndex());
277      }
278    
279      /**
280       * Returns {@code true} if the {@link #isValid(Program, int)} method
281       * returns {@code true} when supplied with the {@linkplain
282       * #getProgram() current <code>Program</code>} and the supplied
283       * {@code index}.
284       *
285       * @return {@code true} if the {@link #isValid(Program, int)} method
286       * returns {@code true} when supplied with the {@linkplain
287       * #getProgram() current <code>Program</code>} and the supplied
288       * {@code index}
289       */
290      public final boolean isValid(final int index) {
291        return isValid(this.getProgram(), index);
292      }
293    
294      /**
295       * Returns {@code true} if the supplied {@link Program} and index
296       * represent a valid combination.
297       *
298       * <p>Effectively this means the supplied {@link Program}:</p>
299       *
300       * <ul>
301       *
302       * <li>Must not be {@code null}.</li>
303       *
304       * </ul>
305       *
306       * <p>The supplied {@code index}:</p>
307       *
308       * <ul>
309       *
310       * <li>Must be greater than or equal to {@code 0}</li>
311       *
312       * <li>Must be less than the {@linkplain AbstractList#size() size of
313       * the supplied <code>Program</code>}</li>
314       *
315       * </ul>
316       *
317       * @param program the {@link Program} to test; may be {@code null}
318       * in which case {@code false} will be returned
319       *
320       * @param index the index to test; may be less than {@code 0} in
321       * which case {@code false} will be returned
322       *
323       * @return {@code true} if the supplied {@link Program} and {@code
324       * index} represent a valid combination; {@code false} otherwise
325       */
326      public static final boolean isValid(final Program<?> program, final int index) {
327        return program != null && index >= 0 && index < program.size();
328      }
329       
330      /**
331       * Returns the {@link Program} affiliated with this {@link
332       * ProgramCounter}.
333       *
334       * <p>This method never returns {@code null}.</p>
335       *
336       * @return a non-{@code null} {@link Program}
337       */
338      private final Program<T> getProgram() {
339        return this.program;
340      }
341    
342      /**
343       * {@linkplain Object#clone() Clones} this {@link ProgramCounter}
344       * and returns the clone.
345       *
346       * <p>This method never returns {@code null}.</p>
347       *
348       * @return a non-{@code null} {@link ProgramCounter} not identical
349       * to this {@link ProgramCounter} but {@linkplain #equals(Object)
350       * equal to it}
351       *
352       * @see Object#clone()
353       */
354      @Override
355      public ProgramCounter<T> clone() {
356        Object clone = null;
357        try {
358          clone = super.clone();
359        } catch (final CloneNotSupportedException neverHappens) {
360          throw (InternalError)new InternalError().initCause(neverHappens);
361        }
362        assert clone instanceof ProgramCounter;
363        @SuppressWarnings("unchecked")
364        final ProgramCounter<T> returnValue = (ProgramCounter<T>)clone;
365        return returnValue;
366      }
367    
368      /**
369       * {@linkplain #clone() Clones} this {@link ProgramCounter} and then
370       * {@linkplain #setIndex(int) sets a new index on it}.
371       *
372       * <p>This method never returns {@code null}.</p>
373       *
374       * @return a non-{@code null} {@link ProgramCounter} not identical
375       * to this {@link ProgramCounter} but {@linkplain #equals(Object)
376       * equal to it} except for its associated {@linkplain #getIndex()
377       * index}
378       */
379      public ProgramCounter<T> clone(final int newIndex) {
380        if (!this.isValid(newIndex)) {
381          throw new IllegalArgumentException("newIndex: " + newIndex);
382        }
383        final ProgramCounter<T> clone = this.clone();
384        assert clone != null;
385        clone.setIndex(newIndex);
386        return clone;
387      }
388    
389      /**
390       * Returns a hashcode for this {@link ProgramCounter}.
391       *
392       * @return a hashcode for this {@link ProgramCounter}
393       */
394      @Override
395      public int hashCode() {
396        int result = 23273 + this.getIndex();
397        final Program<T> program = this.getProgram();
398        if (program != null) {
399          result += program.hashCode();
400        }
401        return result;
402      }
403    
404      /**
405       * Returns {@code true} if the supplied {@link Object} is equal to
406       * this {@link ProgramCounter}; {@code false} otherwise
407       *
408       * @param other the {@link Object} to test; may be {@code null} in
409       * which case {@code false} will be returned
410       *
411       * @return {@code true} if the supplied {@link Object} is equal to
412       * this {@link ProgramCounter}; {@code false} otherwise
413       */
414      @Override
415      public boolean equals(final Object other) {
416        if (other == this) {
417          return true;
418        } else if (other != null && other.getClass().equals(this.getClass())) {
419          final ProgramCounter him = (ProgramCounter)other;
420          final int myIndex = this.getIndex();
421          if (myIndex != him.getIndex()) {
422            return false;
423          }
424          final Program<T> myProgram = this.getProgram();
425          if (myProgram == null) {
426            if (him.getProgram() != null) {
427              return false;
428            }
429          } else if (!myProgram.equals(him.getProgram())) {
430            return false;
431          }
432          return true;
433        } else {
434          return false;
435        }
436      }

438      /**
439       * Returns a non-{@code null} {@link String} representation of this
440       * {@link ProgramCounter}.
441       *
442       * @return a non-{@code null} {@link String} representation of this
443       * {@link ProgramCounter}
444       */
445      @Override
446      public String toString() {
447        final StringBuilder sb = new StringBuilder();
448        final Program<?> program = this.getProgram();
449        if (program == null) {
450          sb.append("null");
451        } else {
452          sb.append(program.toString(this.getIndex()));
453        }
454        /*

461          sb.append(" (null)");
462        } else {
463          sb.append(" (").append(instruction).append(")");
464          }
465        */
466    
467        return sb.toString();
468      }
