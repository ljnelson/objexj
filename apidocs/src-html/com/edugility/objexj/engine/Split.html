028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.List;
033    
034    import java.util.regex.Matcher;
035    import java.util.regex.Pattern;
036    
037    /**
038     * A {@link Jump} instruction that also {@link
039     * InstructionContext#scheduleNewThread(int, boolean) schedules a new
040     * <code>Thread</code>} for execution at a different program location.
041     *
042     * @param <T> the type of {@link Object} managed by {@link
043     * InstructionContext}s passed to the {@link
044     * #execute(InstructionContext)} method
045     *
046     * @author <a href="http://about.me/lairdnelson"
047     * target="_parent">Laird Nelson</a>
048     */
049    public class Split<T> extends Jump<T> {
050    
051      /**
052       * The version of this class for {@linkplain Serializable
053       * serialization purposes}.
054       */
055      private static final long serialVersionUID = 1L;
056    
057      /**
058       * The program location for the new {@link Thread} spawned by this
059       * {@link Split} instruction.
060       */
061      public final int newThreadProgramLocation;
062    
063      /**
064       * Whether or not the {@link Jump#programLocation} and {@link
065       * #newThreadProgramLocation} fields are relative to an {@link
066       * InstructionContext}'s current program location.
067       */
068      public final boolean relative;
069    
070      /**
071       * A {@link Pattern} for parsing a single operand {@link String}
072       * into arguments for new {@link Split} instances.
073       *
074       * <p>This field is never {@code null}.</p>
075       */
076      private static final Pattern OPERAND_PATTERN = Pattern.compile("^\\s*\\+?\\d+\\s*,\\s*(\\+?)(\\d+)");
077    
078      /**
079       * Creates a new {@link Split} instruction.
080       *
081       * @param operands a {@link String} containing values for the {@link
082       * Jump#programLocation}, {@link #newThreadProgramLocation} and
083       * {@link #relative} fields; must not be {@code null}
084       *
085       * @exception IllegalArgumentException if after parsing the values of
086       * the {@link Jump#programLocation} and {@link
087       * #newThreadProgramLocation} fields are the same
088       */
089      public Split(String operands) {
090        super(operands);
091        assert operands != null;
092        final Matcher m = OPERAND_PATTERN.matcher(operands);
093        assert m != null;
094        if (!m.find()) {
095          throw new IllegalArgumentException("Bad operands: " + operands);
096        } else {
097          this.newThreadProgramLocation = Integer.parseInt(m.group(2));
098          this.relative = "+".equals(m.group(1)) || this.newThreadProgramLocation < 0 || this.programLocation < 0;
099        }
100        if (this.programLocation == this.newThreadProgramLocation) {
101          throw new IllegalArgumentException("this.programLocation == this.newThreadProgramLocation: " + this.newThreadProgramLocation);
102        }
103      }
104    
105      /**
106       * Creates a new {@link Split} instruction.
107       *
108       * @param programLocation the location an {@link InstructionContext}
109       * should {@linkplain InstructionContext#jump(int, boolean) jump to}
110       *
111       * @param newThreadProgramLocation the location a new {@link Thread}
112       * should be positioned at
113       *
114       * @exception IllegalArgumentException if {@code programLocation}
115       * and {@code newThreadProgramLocation} are equal
116       */
117      public Split(final int programLocation, final int newThreadProgramLocation) {
118        this(programLocation, newThreadProgramLocation, false);
119      }
120    
121      /**
122       * Creates a new {@link Split} instruction.
123       *
124       * @param  programLocation the location an {@link InstructionContext}
125       * should {@linkplain InstructionContext#jump(int, boolean) jump to}
126       *
127       * @param newThreadProgramLocation the location a new {@link Thread}
128       * should be positioned at
129       *
130       * @param relative whether the {@code programLocation} and {@code
131       * newThreadProgramLocation} parameters should be considered to be
132       * relative locations
133       *
134       * @exception IllegalArgumentException if {@code programLocation}
135       * and {@code newThreadProgramLocation} are equal
136       */
137      public Split(final int programLocation, final int newThreadProgramLocation, final boolean relative) {
138        super(programLocation, relative);
139        this.relative = super.relative || newThreadProgramLocation < 0 || programLocation < 0;
140        if (programLocation == newThreadProgramLocation) {
141          throw new IllegalArgumentException("programLocation == newThreadProgramLocation: " + programLocation);
142        }
143        this.newThreadProgramLocation = newThreadProgramLocation;
144      }
145    
146      /**
147       * Calls {@link InstructionContext#scheduleNewThread(int, boolean)},
148       * supplying it with the values of the {@link
149       * #newThreadProgramLocation} and {@link #relative} fields
150       * respectively, and then calls {@link
151       * Jump#execute(InstructionContext)}.
152       *
153       * @exception IllegalArgumentException if {@code context} is {@code
154       * null}
155       */
156      @Override
157      public final void execute(final InstructionContext<T> context) {
158        if (context == null) {
159          throw new IllegalArgumentException("context");
160        }
161        context.scheduleNewThread(this.newThreadProgramLocation, this.relative);
162        super.execute(context);
163      }
164    
165      /**
166       * Returns a hashcode for this {@link Split} instruction.
167       *
168       * @return a hashcode for this {@link Split} instruction.
169       */
170      @Override
171      public int hashCode() {
172        return 37 * super.hashCode() + this.newThreadProgramLocation + (this.relative ? 1 : 0);
173      }
174    
175      /**
176       * Returns {@code true} if the supplied {@link Object} is equal to
177       * this {@link Split} instruction.
178       *
179       * @param other the {@link Object} to test; may be {@code null} in
180       * which case {@code false} will be returned
181       *
182       * @return {@code true} if the supplied {@link Object} is equal to
183       * this {@link Split} instruction; {@code false} otherwise
184       */
185      @Override
186      public boolean equals(final Object other) {
187        if (other == this) {
188          return true;
189        } else if (super.equals(other)) {
190          final Split him = (Split)other;
191          return this.newThreadProgramLocation == him.newThreadProgramLocation && this.relative == him.relative;
192        } else {
193          return false;
194        }
195      }
196    
197      /**
198       * Returns a non-{@code null} {@link String} representation of this
199       * {@link Split} instruction.
200       *
201       * @return a non-{@code null} {@link String} representation of this
202       * {@link Split} instruction
203       */
204      @Override
205      public String toString() {
206        final StringBuilder sb = new StringBuilder(super.toString()).append(", ");
207        if (this.relative && this.newThreadProgramLocation >= 0) {
208          sb.append("+");
209        }
210        sb.append(this.newThreadProgramLocation);
211        return sb.toString();
212      }
213    
214    
215    }
