028    package com.edugility.objexj.engine;
029    
030    import java.util.List;
031    import java.util.Map;
032    
033    /**
034     * An interface indicating that implementations are capable of
035     * scheduling {@link Thread}s.
036     *
037     * @param <T> the type of {@link Object} manipulated by {@link
038     * Thread}s scheduled by instances of this class
039     *
040     * @author <a href="http://about.me/lairdnelson"
041     * target="_parent">Laird Nelson</a>
042     *
043     * @see Thread
044     */
045    public interface ThreadScheduler<T> {
046    
047      /**
048       * Creates a new {@link Thread} and returns it.  Implementations of
049       * this method must not return {@code null}.
050       *
051       * @param id the identifier for the new {@link Thread}; may be
052       * {@code null} and used only for its {@link Object#toString()}
053       * method
054       *
055       * @param programCounter a non-{@code null} {@link ProgramCounter}
056       * to initialize the new {@link Thread} with
057       *
058       * @param items a possibly {@code null} {@link List} of items that
059       * the new {@link Thread} may {@linkplain Thread#read() read from}
060       *
061       * @param itemPointer the starting index within the supplied {@link
062       * List} of items from which the new {@link Thread} will {@linkplain
063       * Thread#read() read}
064       *
065       * @param captureGroups any {@link CaptureGroup}s that the new
066       * {@link Thread} should start out with; may be {@code null} or
067       * {@linkplain Map#isEmpty() empty}
068       *
069       * @return a new non-{@code null} {@link Thread}
070       */
071      public Thread<T> newThread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables);
072    
073      /**
074       * Schedules the supplied {@link Thread} for execution immediately
075       * or later.
076       *
077       * @return {@code true} if the supplied {@link Thread} was
078       * scheduled; {@code false} otherwise
079       *
080       * @exception IllegalArgumentException if {@code t} is {@code null}
081       */
082      public boolean schedule(final Thread<T> t);
083      
084    }
