028    package com.edugility.objexj.engine;
029    
030    import java.io.Serializable;
031    
032    import java.util.ArrayList;
033    import java.util.Collections;
034    import java.util.List;
035    
036    /**
037     * A sub-{@link List} of items as captured by a {@link
038     * com.edugility.objexj.engine.Thread}.
039     *
040     * @param <T> the type of {@link Object} which will govern {@link
041     * List}s produced by this {@link CaptureGroup}
042     *
043     * @author <a href="http://about.me/lairdnelson"
044     * target="_parent">Laird Nelson</a>
045     */
046    public final class CaptureGroup<T> implements Cloneable, Serializable {
047    
048      /**
049       * The version of this class for {@linkplain Serializable
050       * serialization purposes}.
051       */
052      private static final long serialVersionUID = 1L;
053    
054      /**
055       * The inclusive zero-based index at which this {@link CaptureGroup}
056       * starts.  This field will always be greater than or equal to
057       * {@code 0} and less than the {@linkplain List#size() size} of the
058       * {@linkplain #getItems() <code>List</code> of items}.
059       *
060       * @see #items
061       */
062      private final int startIndex;
063    
064      /**
065       * The exclusive zero-based index at which this {@link CaptureGroup}
066       * ends.  This field will always be greater than or equal to
067       * {@linkplain #startIndex the start index} and less than or
068       * equal to the {@linkplain List#size() size} of the {@linkplain
069       * #getItems() <code>List</code> of items}.
070       *
071       * @see #items
072       */
073      private int endIndex;
074    
075      /**
076       * The {@link List} of items from which this {@link CaptureGroup}
077       * will capture a sub-{@link List}.  This field is never {@code
078       * null}.
079       */
080      private List<T> items;
081    
082      /**
083       * Creates a new {@link CaptureGroup}.
084       *
085       * @param items the {@link List} of items from which to capture;
086       * must not be {@code null} and must not be {@linkplain
087       * List#isEmpty() empty}
088       *
089       * @param startIndex the zero-based index at which capturing will
090       * start; must be greater than or equal to {@code 0} and less than
091       * {@link List#size() items.size()}
092       *
093       * @exception IllegalArgumentException if {@code items} is {@code
094       * null} or {@code items} {@linkplain List#isEmpty() is empty} or
095       * {@code startIndex} is less than {@code 0} or {@code startIndex}
096       * is greater than or equal to {@link List#size() items.size()}
097       */ 
098      public CaptureGroup(final List<T> items, final int startIndex) {
099        super();
100        if (items == null) {
101          throw new IllegalArgumentException("items", new NullPointerException("items"));
102        }
103        if (items.isEmpty()) {
104          throw new IllegalArgumentException("items.isEmpty()");
105        }
106        if (startIndex < 0) {
107          throw new IllegalArgumentException("startIndex < 0: " + startIndex);
108        }
109        if (startIndex >= items.size()) {
110          throw new IllegalArgumentException("startIndex >= items.size(): " + startIndex + " >= " + items.size());
111        }
112    
113        this.items = items;
114        this.startIndex = startIndex;
115        this.endIndex = -1;
116      }
117    
118      /**
119       * Returns a non-{@code null} {@linkplain
120       * Collections#unmodifiableList(List) unmodifiable
121       * <code>List</code>} of the items this {@link CaptureGroup}
122       * captured.
123       *
124       * @return a non-{@code null} {@link List} of items captured by this
125       * {@link CaptureGroup}.  The {@link List} returned is that returned
126       * by the {@link List#subList(int, int)} method (or {@link
127       * Collections#emptyList()})
128       */
129      public final List<T> getItems() {
130        if (this.items == null) {
131          return Collections.emptyList();
132        } else {
133          return Collections.unmodifiableList(this.items.subList(this.startIndex, this.endIndex < 0 ? this.items.size() : this.endIndex));
134        }
135      }
136    
137      /**
138       * Sets the {@link List} of items this {@link CaptureGroup} will
139       * capture from.
140       *
141       * @param items the {@link List} of items; must not be {@code null}
142       *
143       * @exception IllegalArgumentException if {@code items} is {@code
144       * null}
145       */
146      private final void setItemList(final List<T> items) {
147        if (items == null) {
148          throw new IllegalArgumentException("items", new NullPointerException("items == null"));
149        }
150        this.items = items;
151      }
152    
153      /**
154       * Sets the exclusive zero-based index that marks the end of the
155       * items this {@link CaptureGroup} will capture.
156       *
157       * @param endIndex the exclusive zero-based end index; must be
158       * greater than {@code 0}, less than the value of the {@code
159       * startIndex} supplied at {@linkplain #CaptureGroup(List, int)
160       * construction time} and less than the {@linkplain List#size()
161       * size} of the {@link List} of items supplied at {@linkplain
162       * #CaptureGroup(List, int) construction time}
163       *
164       * @exception IllegalArgumentException if {@code endIndex} is
165       * invalid
166       */
167      public final void setEndIndex(final int endIndex) {
168        if (endIndex < this.startIndex) {
169          throw new IllegalArgumentException("endIndex < startIndex: " + endIndex + " < " + this.startIndex);
170        }
171        if (this.items == null) {
172          throw new IllegalArgumentException("endIndex", new IllegalStateException("items == null; endIndex == " + endIndex));
173        }
174        this.endIndex = endIndex;
175      }
176    
177      /**
178       * Returns a non-{@code null} {@linkplain Object#clone() clone} of
179       * this {@link CaptureGroup}.
180       *
181       * @return a non-{@code null} {@linkplain Object#clone() clone} of
182       * this {@link CaptureGroup}
183       */
184      @Override
185      public final CaptureGroup<T> clone() {
186        CaptureGroup<T> clone = null;
187        try {
188          @SuppressWarnings("unchecked")
189          final CaptureGroup<T> temp = (CaptureGroup<T>)super.clone();
190          clone = temp;
191        } catch (final CloneNotSupportedException severeError) {
192          throw (InternalError)new InternalError().initCause(severeError);
193        }
194        assert clone != null;
195        if (this.items != null) {
196          clone.setItemList(new ArrayList<T>(this.items));
197        }
198        return clone;
199      }
200    
201      /**
202       * Returns a hashcode for this {@link CaptureGroup}.
203       *
204       * @return a hashcode for this {@link CaptureGroup}.
205       */
206      @Override
207      public final int hashCode() {
208        int result = 17;
209    
210        // startIndex
211        result = 37 * result + startIndex;
212    
213        // endIndex
214        result = 37 * result + endIndex;
215    
216        // items ignored on purpose
217    
218        return result;
219      }
220    
221      /**
222       * Returns {@code true} if the supplied {@link Object} is equal to
223       * this {@link CaptureGroup}. Two {@link CaptureGroup}s are equal if
224       * their start and end indices are the same.  Notably, the {@link
225       * List} of items supplied {@linkplain #CaptureGroup(List, int) at
226       * construction time} is deliberately excluded from consideration.
227       *
228       * @return {@code true} if the supplied {@link Object} is equal to
229       * this {@link CaptureGroup}; {@code false} otherwise
230       */
231      @Override
232      public final boolean equals(final Object other) {
233        if (other == this) {
234          return true;
235        } else if (this.getClass().equals(other.getClass())) {
236          final CaptureGroup him = (CaptureGroup)other;
237            
238          // startIndex
239          if (this.startIndex != him.startIndex) {
240            return false;
241          }
242    
243          // endIndex
244          if (this.endIndex != him.endIndex) {
245            return false;
246          }
247    
248          // items ignored on purpose
249    
250          // All tests pass
251          return true;
252    
253        } else {
254          return false;
255        }
256      }
257    
258      /**
259       * Returns a non-{@code null} {@link String} representation of this
260       * {@link CaptureGroup}.
261       *
262       * @return a non-{@code null} representation of this {@link
263       * CaptureGroup}
264       */
265      @Override
266      public final String toString() {
267        return String.format("(%d, %d)", this.startIndex, this.endIndex);
268      }
269    
270    }
