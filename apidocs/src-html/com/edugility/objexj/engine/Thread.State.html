042    /**
043     * A virtual thread for running {@link Instruction}s from a regular
044     * expression {@link Program}.
045     *
046     * A {@link Thread} may be in several states:
047     *
048     * <ul>
049     *
050     * <li><strong>{@linkplain State#VIABLE Viable}</strong>.  A {@link
051     * Thread} is <em>viable</em> when its {@linkplain #getState() state}
052     * is {@link State#VIABLE VIABLE} and when it {@linkplain
053     * #getProgramCounter has a valid program counter}.</li>
054     *
055     * <li><strong>{@linkplain State#DEAD Dead}</strong>.  A {@link
056     * Thread} is <em>{@linkplain State#DEAD dead}</em> when its
057     * {@linkplain #getState() state} is {@link State#DEAD DEAD}.  Death
058     * occurs when the {@link #die()} method is invoked.  Internally,
059     * {@link #die()} is invoked in some cases by the {@link #step()}
060     * method and the {@link #advanceProgramCounter()} method.  A dead
061     * {@link Thread} will return {@code false} from the following
062     * methods:
063     *
064     * <ul>
065     *
066     * <li>{@link #canRead()}</li>
067     *
068     * <li>{@link #isViable()}</li>
069     *
070     * </ul>
071     *
072     * Additionally, a {@linkplain State#DEAD dead} {@link Thread} will
073     * throw an {@link IllegalStateException} from any other method that
074     * is invoked.</li>
075     *
076     * <li><strong>{@linkplain State#MATCH Match}</strong>.  A {@link
077     * Thread} is a <em>match</em> when its {@link #getState()} method
078     * returns {@link State#MATCH MATCH}.  A {@link Thread} is placed in
079     * the {@link State#MATCH MATCH} state by invoking the {@link
080     * #match()} method.  A {@link Thread} in the {@link State#MATCH
081     * MATCH} state effectively cannot do anything other than {@linkplain
082     * #getSubmatches() report its submatches}.</li>
083     *
084     * </ul>
085     *
086     * @param <T> the type of items this {@link Thread} will {@linkplain
087     * #read() read}
088     *
089     * @author <a href="http://about.me/lairdnelson" target="_parent">Laird Nelson</a>
090     *
091     * @see <a
092     * href="http://swtch.com/~rsc/regexp/regexp2.html" target="_parent">R. S. Cox. (2009,
093     * December). <span style="font-style: italic;">Regular Expression Matching: the Virtual Machine
094     * Approach</span> [Online]. Available:
095     * http://swtch.com/~rsc/regexp/regexp2.html</a>
096     */
097    public class Thread<T> implements Cloneable, Runnable, ThreadScheduler<T> {

294      /**
295       * Creates a new {@link Thread}.
296       *
297       * @param id the identifier for the new {@link Thread}; used only
298       * for its {@link Object#toString()} output; may be {@code null}
299       *
300       * @param programCounter the {@link ProgramCounter} that {@linkplain
301       * ProgramCounter#getProgram() houses} the {@link Program} that will
302       * be run by this {@link Thread}; must not be {@code null}
303       *
304       * @param items the input to read from; may be {@code null} in which
305       * case the supplied {@code itemPointer} must be equal to {@code 0}
306       * or {@link #VALID_NO_INPUT_POINTER}
307       *
308       * @param itemPointer the index within the supplied {@code items}
309       * {@link List} from which to begin {@linkplain #read() reading};
310       * must be zero or a positive integer less than the supplied {@code
311       * items} {@linkplain Collection#size() size}, or must be equal to
312       * {@link #VALID_NO_INPUT_POINTER} provided that the {@code items}
313       * {@link List} is {@code null} or {@linkplain Collection#isEmpty()
314       * empty}
315       *
316       * @param captureGroups a {@link Map} of {@link CaptureGroup}s
317       * indexed by key that this new {@link Thread} will use to store
318       * {@linkplain #getSubmatches() submatches}; may be {@code null}
319       *
320       * @param variables a {@link Map} of variables this {@link Thread}
321       * will be initialized with; may be {@code null}
322       *
323       * @param threadScheduler the {@link ThreadScheduler} this {@link
324       * Thread} will use to {@linkplain ThreadScheduler#newThread(Object,
325       * ProgramCounter, List, int, Map, Map) spawn} and {@linkplain
326       * ThreadScheduler#schedule(Thread) schedule} new {@link Thread}s;
327       * must not be {@code null}
328       *
329       * @exception IllegalArgumentException if any of the preconditions
330       * outlined as part of the parameter descriptions is not fulfilled
331       *
332       * @see #getSubmatches()
333       *
334       * @see #getVariables()
335       */
336      public Thread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables, final ThreadScheduler<T> threadScheduler) {
337        super();
338        this.state = State.VIABLE;
339    
340        this.id = id;
341    
342        if (captureGroups != null) {
343          this.captureGroups = captureGroups;
344        }
345    
346        if (variables != null) {
347          this.variables = variables;
348        }
349    
350        this.items = items;
351    
352        if (programCounter == null) {
353          throw new IllegalArgumentException("programCounter == null");
354        }
355        this.programCounter = programCounter;
356    
357        if (threadScheduler == null) {
358          throw new IllegalArgumentException("threadScheduler == null");
359        }
360        this.threadScheduler = threadScheduler;
361    
362        /*
363         * For checking the incoming itemPointer, we reject truly insane
364         * values (like huge integers when a small items List is passed
365         * in), but we accept an itemPointer that is "just off the end" of
366         * the incoming List (e.g. itemPointer == items.size()).  This
367         * state of affairs might come about when a Thread has just read
368         * the last item, but the subsequent instruction it's supposed to
369         * run doesn't actually need any input.
370         *
371         * It is certainly the case that if the caller decided to try to
372         * read something with, say, an itemPointer of 0 from an empty
373         * list, the read would fail.  But not all Threads run
374         * Instructions that perform reads.
375         */
376    
377        if (this.items == null || this.items.isEmpty()) {
378          if (itemPointer != VALID_NO_INPUT_POINTER && itemPointer != 0) {
379            throw new IllegalArgumentException("An empty or null input list requires an inputPointer of 0 or VALID_NO_INPUT_POINTER");
380          }
381        } else if (itemPointer < 0) {
382          throw new IllegalArgumentException("itemPointer < 0: " + itemPointer);
383        } else if (itemPointer == this.items.size()) {
384          // WARN: the next read will fail
385        } else if (itemPointer > this.items.size()) {
386          throw new IllegalArgumentException("itemPointer > items.size(): " + itemPointer + " > " + this.items.size());
387        }
388        this.itemPointer = itemPointer;
389      }

422      /**
423       * Returns an {@linkplain Collections#unmodifiableMap(Map)
424       * unmodifiable <tt>Map</tt>} of submatches found by this {@link
425       * Thread}.
426       *
427       * <p>This method never returns {@code null}.</p>
428       *
429       * <p>The {@linkplain Map#values() values} in the {@link Map} that
430       * is returned are guaranteed not to be {@code null} and are
431       * {@linkplain Collections#unmodifiableList(List) unmodifiable}.</p>
432       *
433       * @return a non-{@code null} {@link Map} of submatches indexed by
434       * the keys they were stored under as a result of the {@link
435       * #save(Object)} and {@link #stop(Object)} operations
436       */
437      public final Map<Object, List<T>> getSubmatches() {
438        if (State.MATCH != this.getState()) {
439          throw new IllegalStateException();
440        }
441        Map<Object, List<T>> returnValue = null;
442        if (this.captureGroups != null && !this.captureGroups.isEmpty()) {
443          final Set<Entry<Object, CaptureGroup<T>>> entries = this.captureGroups.entrySet();
444          if (entries != null) {
445            for (final Entry<Object, CaptureGroup<T>> entry : entries) {
446              assert entry != null;
447              final CaptureGroup<T> cg = entry.getValue();
448              if (cg != null) {
449                final Object key = entry.getKey();
450                if (returnValue == null) {
451                  returnValue = new HashMap<Object, List<T>>();
452                }
453                returnValue.put(key, cg.getItems());
454              }
455            }
456          }
457        }
458        if (returnValue == null) {
459          returnValue = Collections.emptyMap();
460        } else {
461          returnValue = Collections.unmodifiableMap(returnValue);
462        }
463        return returnValue;
464      }

496      /**
497       * Marks a position in the input such that corresponding items
498       * {@linkplain #getSubmatches() can be retrieved later}.
499       *
500       * @param key the {@link Object} under which to index the position;
501       * may be {@code null}
502       *
503       * @see #getSubmatches()
504       */
505      public final void save(final Object key) {
506        if (key == null) {
507          throw new IllegalArgumentException("key", new NullPointerException("key"));
508        }
509        this.ensureViable();
510        if (this.captureGroups == null) {
511          this.captureGroups = new HashMap<Object, CaptureGroup<T>>();
512          this.captureGroups.put(key, new CaptureGroup<T>(this.items, this.getItemPointer()));
513        } else {
514          final CaptureGroup<T> cg = this.captureGroups.get(key);
515          if (cg == null) {
516            this.captureGroups.put(key, new CaptureGroup<T>(this.items, this.getItemPointer()));
517          } else {
518            cg.setEndIndex(this.getItemPointer());
519          }
520        }
521      }
522    
523      /**
524       * Marks the end of a region in the input such that corresponding
525       * items {@linkplain #getSubmatches() can be retrieved later}.
526       *
527       * @param key the {@link Object} that was passed to a previous
528       * {@link #save(Object)} call; may be {@code null}
529       */
530      public final void stop(final Object key) {
531        if (key == null) {
532          throw new IllegalArgumentException("key", new NullPointerException("key"));
533        }
534        this.ensureViable();
535        if (this.captureGroups != null) {
536          final CaptureGroup<T> cg = this.captureGroups.get(key);
537          if (cg != null) {
538            cg.setEndIndex(this.getItemPointer());
539          }
540        }
541      }

543      /**
544       * Creates a new {@link Thread} and returns it.
545       *
546       * @param id the id for the new {@link Thread}; may be {@code null}
547       * in which case a generated identifier will be used instead; only
548       * its {@link Object#toString()} method will be called
549       *
550       * @param programCounterIndex the position within the associaged
551       * {@link #getProgramCounter() ProgramCounter} from which to begin
552       * execution; may be relative or absolute depending on the value of
553       * the {@code relative} parameter
554       *
555       * @param relative whether the {@code programCounterIndex} parameter
556       * represents a relative position or an absolute index
557       *
558       * @return a non-{@code null} new {@link Thread} not identical to
559       * this one
560       *
561       * @exception IllegalStateException if this {@link Thread}
562       * {@linkplain #isViable() is not viable}
563       */
564      public final Thread<T> newThread(final Object id, final int programCounterIndex, final boolean relative) {
565        this.ensureViable();
566        ProgramCounter<T> programCounter = this.getProgramCounter();
567        if (programCounter != null) {
568          final int absoluteProgramCounterIndex;
569          if (relative) {
570            absoluteProgramCounterIndex = programCounter.getIndex() + programCounterIndex;
571          } else {
572            absoluteProgramCounterIndex = programCounterIndex;
573          }
574          programCounter = programCounter.clone(absoluteProgramCounterIndex);
575        }
576        // TODO: we're not cloning the variables; is that OK?
577        return this.newThread(id, programCounter, this.items, this.getItemPointer(), deepClone(this.captureGroups), this.variables);
578      }

581       * {@inheritDoc}
582       *
583       * @exception IllegalStateException if this {@link Thread}
584       * {@linkplain #isViable() is not viable}
585       */
586      @Override
587      public final Thread<T> newThread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables) {
588        this.ensureViable();
589        if (this.threadScheduler == null) {
590          throw new IllegalStateException("threadScheduler == null");
591        }
592        return this.threadScheduler.newThread(id, programCounter, items, itemPointer, captureGroups, variables);
593      }

595      /**
596       * Schedules the supplied {@link Thread} for execution.
597       *
598       * <p>This method may be invoked only when {@link #isViable()}
599       * returns {@code true}.</p>
600       *
601       * <p>This method performs no validity checking of any kind on the
602       * supplied {@link Thread}, relying instead upon the underlying
603       * {@link ThreadScheduler}'s {@link
604       * ThreadScheduler#schedule(Thread)} method to perform any
605       * checks.</p>
606       *
607       * @exception IllegalStateException if this {@link Thread}
608       * {@linkplain #isViable() is not viable}
609       */
610      @Override
611      public final boolean schedule(final Thread<T> t) {
612        this.ensureViable();
613        if (this.threadScheduler == null) {
614          throw new IllegalStateException("threadScheduler == null");
615        }
616        return this.threadScheduler.schedule(t);
617      }

619      /**
620       * Returns {@code true} if this {@link Thread}'s {@linkplain
621       * #getItemPointer() item pointer} is {@code 0} (in the case of
622       * there being input) or {@link #VALID_NO_INPUT_POINTER} (in the case
623       * of there being null or {@linkplain Collection#isEmpty() empty}
624       * input).
625       *
626       * @return {@code true} if this {@link Thread} is positioned at the
627       * beginning of the input; {@code false} otherwise
628       *
629       * @exception IllegalStateException if this {@link Thread}'s
630       * {@linkplain #getState() state} is {@link State#DEAD}
631       */
632      public final boolean atStart() {
633        if (Thread.State.DEAD == this.getState()) {
634          throw new IllegalStateException(Thread.State.DEAD.toString());
635        }
636        if (this.items == null || this.items.isEmpty()) {
637          return this.getItemPointer() == VALID_NO_INPUT_POINTER;
638        } else {
639          return this.getItemPointer() == 0;
640        }
641      }

643      /**
644       * Returns {@code true} if this {@link Thread}'s input pointer is
645       * equal to this {@link Thread}'s input size.
646       *
647       * <p><strong>Note:</strong> This method may be invoked in any
648       * {@linkplain State state} except {@link State#DEAD}.</p>
649       *
650       * @return {@code true} if this {@link Thread}'s input pointer is
651       * equal to this {@link Thread}'s input size; {@code false}
652       * otherwise
653       */
654      public final boolean atEnd() {
655        if (Thread.State.DEAD == this.getState()) {
656          throw new IllegalStateException(Thread.State.DEAD.toString());
657        }
658        if (this.items == null || this.items.isEmpty()) {
659          return this.getItemPointer() == VALID_NO_INPUT_POINTER;
660        } else {
661          return this.getItemPointer() == this.items.size();
662        }
663      }

665      /**
666       * Returns {@code true} if this {@link Thread} {@linkplain
667       * #isViable() is viable}, and has more items to read.
668       *
669       * <p>This method may be invoked in all states.</p>
670       *
671       * @return {@code true} if this {@link Thread} can read an item
672       */
673      public final boolean canRead() {
674        return this.isViable() && this.isValidItemPointer(this.getItemPointer());
675      }

677      /**
678       * Reads (but does not "consume") the current item.
679       *
680       * <p>This method may be invoked only when {@link #canRead()}
681       * returns {@code true}.</p>
682       *
683       * @return the current item which may be {@code null}
684       *
685       * @exception IllegalStateException if the {@link #canRead()} method
686       * returns {@code false}
687       *
688       * @see #canRead()
689       */
690      public final T read() {
691        if (!this.canRead()) {
692          throw new IllegalStateException("Thread cannot read");
693        }
694        if (this.items == null || this.items.isEmpty()) {
695          // This is an edge case.  If we are originally supplied with no
696          // input, we need to be able to use an itemPointer that is both
697          // valid, but doesn't point into the (non-existent) input list.
698          assert this.getItemPointer() == VALID_NO_INPUT_POINTER;
699          return null;
700        }
701        assert this.getItemPointer() >= 0 && this.getItemPointer() < this.items.size();
702        return this.items.get(this.getItemPointer());
703      }

705      /**
706       * Returns {@code true} if the supplied {@code itemPointer}
707       * parameter is valid.
708       *
709       * @param itemPointer the item pointer to consider
710       *
711       * @return {@code true} if the supplied {@code itemPointer}
712       * parameter is valid; {@code false} otherwise
713       */
714      private final boolean isValidItemPointer(final int itemPointer) {
715        if (this.items == null || this.items.isEmpty()) {
716          // We can have null or empty input.  In such a case, a value
717          // equal to VALID_NO_INPUT_POINTER is OK; all others are not.
718          return itemPointer == VALID_NO_INPUT_POINTER;
719        }
720        return itemPointer >= 0 && itemPointer < this.items.size();
721      }

723      /**
724       * Advances this {@link Thread}'s item pointer, provided it was
725       * valid to begin with.  Returns {@code true} if the resulting item
726       * pointer is valid.
727       *
728       * <p>Please note that a return value of {@code false} does
729       * <em>not</em> indicate that this {@link Thread} is dead.</p>
730       *
731       * @return {@code true} if this {@link Thread}'s item pointer was
732       * successfully advanced; {@code false} otherwise
733       *
734       * @exception IllegalStateException if this {@link Thread}
735       * {@linkplain #isViable() is not viable}
736       */
737      public final boolean advanceItemPointer() {
738        this.ensureViable();
739        final boolean returnValue;
740        int itemPointer = this.getItemPointer();
741        if (itemPointer == INVALID_INPUT_POINTER || itemPointer == VALID_NO_INPUT_POINTER) {
742          returnValue = false;
743        } else {
744          ++this.itemPointer;
745          returnValue = this.isValidItemPointer(this.getItemPointer());
746        }
747        return returnValue;
748      }

756       *
757       * @return this {@link Thread}'s associated {@link ProgramCounter};

761        this.ensureViable();

765      /**
766       * Advances the program counter to point at the next {@link
767       * Instruction} in the {@link Program}.  If this cannot
768       * happen&mdash;as in the case where there are no more {@link
769       * Instruction}s in the {@link Program}&mdash;then the {@link
770       * #die()} method is invoked.
771       *
772       * <p>This method may be invoked only when {@link #isViable()}
773       * returns {@code true}.</p>
774       *
775       * @return {@code true} if the program counter was successfully
776       * advanced and therefore if there is another {@link Instruction}
777       * ready to be executed; {@code false} if this {@link Thread} has
778       * been killed via the {@link #die()} method since there are no more
779       * {@link Instruction}s in the {@link Program} to be executed
780       *
781       * @exception IllegalStateException if this {@link Thread}
782       * {@linkplain #isViable() is not viable}
783       */
784      public final boolean advanceProgramCounter() {
785        this.ensureViable();
786        boolean returnValue = false;
787        final ProgramCounter<T> programCounter = this.getProgramCounter();
788        if (programCounter != null) {
789          returnValue = programCounter.advance();
790        }
791        if (!returnValue) {
792          // The program counter could not be incremented, so there are no
793          // further instructions to run, so we're dead.
794          this.die();
795        }
796        return returnValue;
797      }

799      /**
800       * Sets this {@link Thread}'s program counter to the supplied value,
801       * provided it is valid.  If the supplied value is <em>not</em>
802       * valid, then this {@link Thread} {@linkplain #die() dies}.
803       *
804       * <p>This method may be invoked only when {@link #isViable()}
805       * returns {@code true}.</p>
806       *
807       * @param programCounter
808       *
809       * @param relative
810       *
811       * @return {@code true} if the program counter was actually set;
812       * {@code false} if it was not (thus indicating that this {@link
813       * Thread} is now dead)
814       *
815       * @exception IllegalArgumentException if the supplied {@code
816       * programCounter} is not valid
817       *
818       * @exception IllegalStateException if this {@link Thread}
819       * {@linkplain #isViable() is not viable}
820       */
821      public final boolean jump(final int programCounter, final boolean relative) {
822        this.ensureViable();
823        final boolean returnValue;
824        final ProgramCounter<T> pc = this.getProgramCounter();
825        final int absoluteProgramCounter;
826        if (relative) {
827          absoluteProgramCounter = pc.getIndex() + programCounter;
828        } else {
829          absoluteProgramCounter = programCounter;
830        }
831        if (pc != null && pc.isValid(absoluteProgramCounter)) {
832          pc.setIndex(absoluteProgramCounter);
833          returnValue = true;
834        } else {
835          this.die();
836          returnValue = false;
837        }
838        return returnValue;
839      }

845       * returns {@code true}.</p>

847       * @return the current {@link Instruction}, or {@code null} if the
848       * {@link #isViable()} method returns {@code false}

851       * {@linkplain #isViable() is not viable}
852       *

855      private final Instruction<T> getInstruction() {
856        this.ensureViable();
857        final Instruction<T> returnValue;
858        final ProgramCounter<T> programCounter = this.getProgramCounter();
859        if (programCounter != null && this.isViable()) {
860          returnValue = programCounter.getInstruction();
861        } else {
862          returnValue = null;
863        }
864        return returnValue;

867      /**
868       * Retrieves the current {@link Instruction} and {@linkplain
869       * Instruction#execute(InstructionContext) executes} it.
870       *
871       * <p>If the {@link #isViable()} method returns {@code false},
872       * then invoking this method will cause the {@link #die()} method to
873       * be invoked and no further action will take place.</p>
874       *
875       * <p>This method may be invoked only when {@link #isViable()}
876       * returns {@code true}.</p>
877       *
878       * @exception IllegalStateException if this {@link Thread}
879       * {@linkplain #isViable() is not viable}
880       */
881      public final void step() {
882        this.ensureViable();
883        final Instruction<T> instruction = this.getInstruction();
884        assert instruction != null;
885        if (this.ic == null) {
886          this.ic = new InstructionContext<T>(this);
887        }
888        instruction.execute(this.ic);
889      }

891      /**
892       * Repeatedly calls the {@link #step()} method until this {@link
893       * Thread} has completed execution.
894       *
895       * <p>This method may be invoked only when {@link #isViable()}
896       * returns {@code true}.</p>
897       *
898       * @exception IllegalStateException if this {@link Thread}
899       * {@linkplain #isViable() is not viable}
900       */
901      @Override
902      public final void run() {
903        this.ensureViable();
904        final Logger logger = Logger.getLogger(this.getClass().getName());
905        assert logger != null;
906        final boolean finer = logger.isLoggable(Level.FINER);
907        final ProgramCounter<T> pc = this.getProgramCounter();
908        assert pc != null;
909        while (this.isViable()) {
910          final Instruction<T> currentInstruction = pc.getInstruction();
911          assert currentInstruction != null;
912          if (finer) {
913            final int index = pc.getIndex();
914            logger.logp(Level.FINER, this.getClass().getName(), "run", "Before running {0} {1} ({2}) at input position {3}", new Object[] { this, currentInstruction, Integer.valueOf(index), Integer.valueOf(this.getItemPointer()) });
915          }
916          this.step();
917          if (finer) {
918            if (this.isViable()) {
919              logger.logp(Level.FINER, this.getClass().getName(), "run", "After running {0} {1}; input position is now {2}", new Object[] { this, currentInstruction, Integer.valueOf(this.getItemPointer()) });
920            } else {
921              logger.logp(Level.FINER, this.getClass().getName(), "run", "Thread {0} is no longer viable", this);
922            }
923          }
924        }
925      }

1061      /**
1062       * {@linkplain Object#clone() Clones} this {@link Thread} and
1063       * returns the clone.
1064       *
1065       * <p>This method never returns {@code null}.</p>
1066       *
1067       * <h2>Design Notes</h2>
1068       *
1069       * <p>{@link Thread}s internally store {@link Map}s of {@link
1070       * CaptureGroup}s.  These {@link Map}s are {@linkplain
1071       * #deepClone(Map) deeply cloned}&mdash;that is, the {@link Map}
1072       * itself is {@link Object#clone() cloned} as well as the {@link
1073       * CaptureGroup#clone() CaptureGroup}s inside them.</p>
1074       *
1075       * <p>As of this writing, the internal {@link Map} of variables is
1076       * not deeply cloned.  This means that two {@link Thread}s might
1077       * share variable map keys and values.</p>
1078       *
1079       * @return a non-{@code null} clone of this {@link Thread}
1080       */
1081      @Override
1082      public Thread<T> clone() {
1083        Thread<T> clone = null;
1084        try {
1085          @SuppressWarnings("unchecked")
1086          final Thread<T> temp = (Thread<T>)super.clone();
1087          clone = temp;
1088        } catch (final CloneNotSupportedException cannotHappen) {
1089          throw (InternalError)new InternalError().initCause(cannotHappen);
1090        }
1091        assert clone != null;
1092        
1093        // Clone our ProgramCounter.
1094        final ProgramCounter<T> programCounter = this.getProgramCounter();
1095        if (programCounter != null) {
1096          clone.programCounter = programCounter.clone();
1097        }
1098    
1099        // TODO: Clone our variables?
1100    
1101        // Clone our capture groups.
1102        if (this.captureGroups != null) {
1103          clone.captureGroups = deepClone(this.captureGroups);
1104        }
1105    
1106        // With all of his state set, make sure that he's VIABLE.
1107        assert clone.isViable();
1108    
1109        return clone;
1110      }

1112      /**
1113       * Returns a hashcode for this {@link Thread}.
1114       *
1115       * @return a hashcode for this {@link Thread}
1116       */
1117      @Override
1118      public int hashCode() {
1119        int result = 17;
1120        
1121        final Object id = this.getId();
1122        int c = id == null ? 0 : id.hashCode();
1123        result = result * 37 + c;
1124    
1125        c = this.getItemPointer();
1126        result = result * 37 + c;
1127    
1128        c = this.getState().hashCode();
1129        result = result * 37 + c;
1130    
1131        c = this.items == null ? 0 : this.items.hashCode();
1132        result = result * 37 + c;
1133    
1134        // We don't call getProgramCounter() because hashCode() must be a
1135        // method that is permissible to call in any state.
1136        c = this.programCounter.hashCode();
1137        result = result * 37 + c;
1138    
1139        c = this.captureGroups == null ? 0 : this.captureGroups.hashCode();
1140        result = result * 37 + c;
1141    
1142        return result;
1143      }

1145      /**
1146       * Returns {@code true} if the supplied {@link Object} is equal to
1147       * this {@link Thread}.
1148       *
1149       * @return {@code true} if the supplied {@link Object} is equal to
1150       * this {@link Thread}.
1151       */
1152      @Override
1153      public boolean equals(final Object other) {
1154        if (other == this) {
1155          return true;
1156        } else if (this.getClass().equals(other.getClass())) {
1157          final Thread him = (Thread)other;
1158    
1159          final int myItemPointer = this.getItemPointer();
1160          if (him.getItemPointer() != myItemPointer) {
1161            return false;
1162          }
1163    
1164          final Object id = this.getId();
1165          if (id == null) {
1166            if (him.getId() != null) {
1167              return false;
1168            }
1169          }
1170    
1171          final State state = this.getState();
1172          assert state != null;
1173          if (state != him.getState()) {
1174            return false;
1175          }
1176    
1177          if (this.items == null) {
1178            if (him.items != null) {
1179              return false;
1180            }
1181          } else if (!this.items.equals(him.items)) {
1182            return false;
1183          }
1184    
1185          // We don't call getProgramCounter() because hashCode() must be
1186          // a method that is permissible to call in any state.
1187          if (this.programCounter == null) {
1188            if (him.programCounter != null) {
1189              return false;
1190            }
1191          } else if (!this.programCounter.equals(him.programCounter)) {
1192            return false;
1193          }
1194    
1195          if (this.captureGroups == null) {
1196            if (him.captureGroups != null) {
1197              return false;
1198            }
1199          } else if (!this.captureGroups.equals(him.captureGroups)) {
1200            return false;
1201          }
1202    
1203          return true;
1204    
1205        } else {
1206          return false;
1207        }
1208      }

1233      /**
1234       * {@linkplain Object#clone() Clones} the supplied {@link Map} of
1235       * {@link CaptureGroups} by creating a new {@link HashMap} using its
1236       * {@linkplain HashMap#HashMap(Map) copy constructor} supplied with
1237       * the value of the {@code suppliedCaptureGroups} parameter.
1238       *
1239       * <p>In addition, each {@link CaptureGroup} in the supplied {@link
1240       * Map} is {@linkplain CaptureGroup#clone() cloned}.</p>
1241       *
1242       * <p>The end result is a {@link Map} of {@link CaptureGroup}s
1243       * indexed by {@link Object}s that is {@linkplain
1244       * Object#equals(Object) equal to} the supplied {@link
1245       * suppliedCaptureGroups} parameter, but not identical to it, and
1246       * containing {@linkplain Map#values() values} that are not
1247       * identical to the {@linkplain Map#values() values} in the supplied
1248       * {@code suppliedCaptureGroups} parameter.</p>
1249       *
1250       * @param suppliedCaptureGroups a {@link Map} of {@link
1251       * CaptureGroup}s indexed by arbitrary, possibly {@code null}, keys;
1252       * may be {@code null} in which case {@code null} will be returned
1253       *
1254       * @return a {@link Map} equal to the supplied {@link
1255       * suppliedCaptureGroups} parameter, but guaranteed not to be
1256       * identical to it; or {@code null} if the supplied {@code
1257       * suppliedCaptureGroups} parameter is {@code null}
1258       */
1259      private static final <T> Map<Object, CaptureGroup<T>> deepClone(final Map<Object, CaptureGroup<T>> suppliedCaptureGroups) {
1260        final Map<Object, CaptureGroup<T>> captureGroups;
1261        if (suppliedCaptureGroups != null) {
1262          captureGroups = new HashMap<Object, CaptureGroup<T>>(suppliedCaptureGroups.size());
1263          final Set<Entry<Object, CaptureGroup<T>>> entries = suppliedCaptureGroups.entrySet();
1264          if (entries != null && !entries.isEmpty()) {
1265            for (final Entry<Object, CaptureGroup<T>> entry : entries) {
1266              if (entry != null) {
1267                final CaptureGroup<T> cg = entry.getValue();
1268                if (cg == null) {
1269                  captureGroups.put(entry.getKey(), null);
1270                } else {
1271                  captureGroups.put(entry.getKey(), cg.clone());
1272                }
1273              }
1274            }
1275          }
1276        } else {
1277          captureGroups = null;
1278        }
1279        return captureGroups;
1280      }
