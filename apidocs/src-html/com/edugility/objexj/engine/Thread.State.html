042    /**
043     * A virtual thread for running {@link Instruction}s from a regular
044     * expression {@link Program}.
045     *
046     * A {@link Thread} may be in several states:
047     *
048     * <ul>
049     *
050     * <li><strong>{@linkplain State#VIABLE Viable}</strong>.  A {@link
051     * Thread} is <em>viable</em> when its {@linkplain #getState() state}
052     * is {@link State#VIABLE VIABLE} and when it {@linkplain
053     * #getProgramCounter has a valid program counter}.</li>
054     *
055     * <li><strong>{@linkplain State#DEAD Dead}</strong>.  A {@link
056     * Thread} is <em>{@linkplain State#DEAD dead}</em> when its
057     * {@linkplain #getState() state} is {@link State#DEAD DEAD}.  Death
058     * occurs when the {@link #die()} method is invoked.  Internally,
059     * {@link #die()} is invoked in some cases by the {@link #step()}
060     * method and the {@link #advanceProgramCounter()} method.  A dead
061     * {@link Thread} will return {@code false} from the following
062     * methods:
063     *
064     * <ul>
065     *
066     * <li>{@link #canRead()}</li>
067     *
068     * <li>{@link #isViable()}</li>
069     *
070     * </ul>
071     *
072     * Additionally, a {@linkplain State#DEAD dead} {@link Thread} will
073     * throw an {@link IllegalStateException} from any other method that
074     * is invoked.</li>
075     *
076     * <li><strong>{@linkplain State#MATCH Match}</strong>.  A {@link
077     * Thread} is a <em>match</em> when its {@link #getState()} method
078     * returns {@link State#MATCH MATCH}.  A {@link Thread} is placed in
079     * the {@link State#MATCH MATCH} state by invoking the {@link
080     * #match()} method.  A {@link Thread} in the {@link State#MATCH
081     * MATCH} state effectively cannot do anything other than {@linkplain
082     * #getSubmatches() report its submatches}.</li>
083     *
084     * </ul>
085     *
086     * @param <T> the type of items this {@link Thread} will {@linkplain
087     * #read() read}
088     *
089     * @author <a href="http://about.me/lairdnelson">Laird Nelson</a>
090     *
091     * @see <a
092     * href="http://swtch.com/~rsc/regexp/regexp2.html">R. S. Cox. (2009,
093     * December). <span style="font-style: italic;">Regular Expression Matching: the Virtual Machine
094     * Approach</span> [Online]. Available:
095     * http://swtch.com/~rsc/regexp/regexp2.html</a>
096     */
097    public class Thread<T> implements Cloneable, Runnable, ThreadScheduler<T> {

261      /**
262       * Creates a new {@link Thread}.
263       *
264       * @param id the identifier for the new {@link Thread}; used only
265       * for its {@link Object#toString()} output; may be {@code null}
266       *
267       * @param programCounter the {@link ProgramCounter} that {@linkplain
268       * ProgramCounter#getProgram() houses} the {@link Program} that will
269       * be run by this {@link Thread}; must not be {@code null}
270       *
271       * @param items the input to read from; may be {@code null} in which
272       * case the supplied {@code itemPointer} must be equal to {@code 0}
273       * or {@link #VALID_NO_INPUT_POINTER}
274       *
275       * @param itemPointer the index within the supplied {@code items}
276       * {@link List} from which to begin {@linkplain #read() reading};
277       * must be zero or a positive integer less than the supplied {@code
278       * items} {@linkplain Collection#size() size}, or must be equal to
279       * {@link #VALID_NO_INPUT_POINTER} provided that the {@code items}
280       * {@link List} is {@code null} or {@linkplain Collection#isEmpty()
281       * empty}
282       *
283       * @param captureGroups a {@link Map} of {@link CaptureGroup}s
284       * indexed by key that this new {@link Thread} will use to store
285       * {@linkplain #getSubmatches() submatches}; may be {@code null}
286       *
287       * @param variables a {@link Map} of variables this {@link Thread}
288       * will be initialized with; may be {@code null}
289       *
290       * @param threadScheduler the {@link ThreadScheduler} this {@link
291       * Thread} will use to {@linkplain ThreadScheduler#newThread(Object,
292       * ProgramCounter, List, int, Map, Map) spawn} and {@linkplain
293       * ThreadScheduler#schedule(Thread) schedule} new {@link Thread}s;
294       * must not be {@code null}
295       *
296       * @exception IllegalArgumentException if any of the preconditions
297       * outlined as part of the parameter descriptions is not fulfilled
298       *
299       * @see #getSubmatches()
300       *
301       * @see #getVariables()
302       */
303      public Thread(final Object id, final ProgramCounter<T> programCounter, final List<T> items, final int itemPointer, final Map<Object, CaptureGroup<T>> captureGroups, final Map<Object, Object> variables, final ThreadScheduler<T> threadScheduler) {
304        super();
305        this.state = State.VIABLE;
306    
307        this.id = id;
308    
309        if (captureGroups != null) {
310          this.captureGroups = captureGroups;
311        }
312    
313        if (variables != null) {
314          this.variables = variables;
315        }
316    
317        this.items = items;
318    
319        if (programCounter == null) {
320          throw new IllegalArgumentException("programCounter == null");
321        }
322        this.programCounter = programCounter;
323    
324        if (threadScheduler == null) {
325          throw new IllegalArgumentException("threadScheduler == null");
326        }
327        this.threadScheduler = threadScheduler;
328    
329        /*
330         * For checking the incoming itemPointer, we reject truly insane
331         * values (like huge integers when a small items List is passed
332         * in), but we accept an itemPointer that is "just off the end" of
333         * the incoming List (e.g. itemPointer == items.size()).  This
334         * state of affairs might come about when a Thread has just read
335         * the last item, but the subsequent instruction it's supposed to
336         * run doesn't actually need any input.
337         *
338         * It is certainly the case that if the caller decided to try to
339         * read something with, say, an itemPointer of 0 from an empty
340         * list, the read would fail.  But not all Threads run
341         * Instructions that perform reads.
342         */
343    
344        if (this.items == null || this.items.isEmpty()) {
345          if (itemPointer != VALID_NO_INPUT_POINTER && itemPointer != 0) {
346            throw new IllegalArgumentException("An empty or null input list requires an inputPointer of 0 or VALID_NO_INPUT_POINTER");
347          }
348        } else if (itemPointer < 0) {
349          throw new IllegalArgumentException("itemPointer < 0: " + itemPointer);
350        } else if (itemPointer == this.items.size()) {
351          // WARN: the next read will fail
352        } else if (itemPointer > this.items.size()) {
353          throw new IllegalArgumentException("itemPointer > items.size(): " + itemPointer + " > " + this.items.size());
354        }
355        this.itemPointer = itemPointer;
356      }

389      /**
390       * Returns an {@linkplain Collections#unmodifiableMap(Map)
391       * unmodifiable <tt>Map</tt>} of submatches found by this {@link
392       * Thread}.
393       *
394       * <p>This method never returns {@code null}.</p>
395       *
396       * <p>The {@linkplain Map#values() values} in the {@link Map} that
397       * is returned are guaranteed not to be {@code null} and are
398       * {@linkplain Collections#unmodifiableList(List) unmodifiable}.</p>
399       *
400       * @return a non-{@code null} {@link Map} of submatches indexed by
401       * the keys they were stored under as a result of the {@link
402       * #save(Object)} and {@link #stop(Object)} operations
403       */
404      public final Map<Object, List<T>> getSubmatches() {
405        if (State.MATCH != this.getState()) {
406          throw new IllegalStateException();
407        }
408        Map<Object, List<T>> returnValue = null;
409        if (this.captureGroups != null && !this.captureGroups.isEmpty()) {
410          final Set<Entry<Object, CaptureGroup<T>>> entries = this.captureGroups.entrySet();
411          if (entries != null) {
412            for (final Entry<Object, CaptureGroup<T>> entry : entries) {
413              assert entry != null;
414              final CaptureGroup<T> cg = entry.getValue();
415              if (cg != null) {
416                final Object key = entry.getKey();
417                if (returnValue == null) {
418                  returnValue = new HashMap<Object, List<T>>();
419                }
420                returnValue.put(key, cg.getItems());
421              }
422            }
423          }
424        }
425        if (returnValue == null) {
426          returnValue = Collections.emptyMap();
427        } else {
428          returnValue = Collections.unmodifiableMap(returnValue);
429        }
430        return returnValue;
431      }

444      /**
445       * Marks a position in the input such that corresponding items
446       * {@linkplain #getSubmatches() can be retrieved later}.
447       *
448       * @param key the {@link Object} under which to index the position;
449       * may be {@code null}
450       *
451       * @see #getSubmatches()
452       */
453      public final void save(final Object key) {
454        if (key == null) {
455          throw new IllegalArgumentException("key", new NullPointerException("key"));
456        }
457        this.ensureViable();
458        if (this.captureGroups == null) {
459          this.captureGroups = new HashMap<Object, CaptureGroup<T>>();
460          this.captureGroups.put(key, new CaptureGroup<T>(this.items, this.getItemPointer()));
461        } else {
462          final CaptureGroup<T> cg = this.captureGroups.get(key);
463          if (cg == null) {
464            this.captureGroups.put(key, new CaptureGroup<T>(this.items, this.getItemPointer()));
465          } else {
466            cg.setEndIndex(this.getItemPointer());
467          }
468        }
469      }
470    
471      /**
472       * Marks the end of a region in the input such that corresponding
473       * items {@linkplain #getSubmatches() can be retrieved later}.
474       *
475       * @param key the {@link Object} that was passed to a previous
476       * {@link #save(Object)} call; may be {@code null}
477       */
478      public final void stop(final Object key) {
479        if (key == null) {
480          throw new IllegalArgumentException("key", new NullPointerException("key"));
481        }
482        this.ensureViable();
483        if (this.captureGroups != null) {
484          final CaptureGroup<T> cg = this.captureGroups.get(key);
485          if (cg != null) {
486            cg.setEndIndex(this.getItemPointer());
487          }
488        }
489      }

491      /**
492       * @exception IllegalStateException if this {@link Thread}
493       * {@linkplain #isViable() is not viable}
494       */
495      public final Thread<T> newThread(final Object id, final int programCounterIndex, final boolean relative) {
496        this.ensureViable();
497        ProgramCounter<T> programCounter = this.getProgramCounter();
498        if (programCounter != null) {
499          final int absoluteProgramCounterIndex;
500          if (relative) {
501            absoluteProgramCounterIndex = programCounter.getIndex() + programCounterIndex;
502          } else {
503            absoluteProgramCounterIndex = programCounterIndex;
504          }
505          programCounter = programCounter.clone(absoluteProgramCounterIndex);
506        }
507        // TODO: we're not cloning the variables; is that OK?
508        return this.newThread(id, programCounter, this.items, this.getItemPointer(), deepClone(this.captureGroups), this.variables);
509      }

526      /**
527       * Schedules the supplied {@link Thread} for execution.
528       *
529       * <p>This method may be invoked only when {@link #isViable()}
530       * returns {@code true}.</p>
531       *
532       * <p>This method performs no validity checking of any kind on the
533       * supplied {@link Thread}, relying instead upon the underlying
534       * {@link ThreadScheduler}'s {@link
535       * ThreadScheduler#schedule(Thread)} method to perform any
536       * checks.</p>
537       *
538       * @exception IllegalStateException if this {@link Thread}
539       * {@linkplain #isViable() is not viable}
540       */
541      @Override
542      public final boolean schedule(final Thread<T> t) {
543        this.ensureViable();
544        if (this.threadScheduler == null) {
545          throw new IllegalStateException("threadScheduler == null");
546        }
547        return this.threadScheduler.schedule(t);
548      }

550      public final boolean atStart() {
551        if (Thread.State.DEAD == this.getState()) {
552          throw new IllegalStateException(Thread.State.DEAD.toString());
553        }
554        if (this.items == null || this.items.isEmpty()) {
555          return this.getItemPointer() == VALID_NO_INPUT_POINTER;
556        } else {
557          return this.getItemPointer() == 0;
558        }
559      }

561      /**
562       * Returns {@code true} if this {@link Thread}'s input pointer is
563       * equal to this {@link Thread}'s input size.
564       *
565       * <p><strong>Note:</strong> This method may be invoked in any
566       * {@linkplain State state} except {@link State#DEAD}.</p>
567       *
568       * @return {@code true} if this {@link Thread}'s input pointer is
569       * equal to this {@link Thread}'s input size; {@code false}
570       * otherwise
571       */
572      public final boolean atEnd() {
573        if (Thread.State.DEAD == this.getState()) {
574          throw new IllegalStateException(Thread.State.DEAD.toString());
575        }
576        if (this.items == null || this.items.isEmpty()) {
577          return this.getItemPointer() == VALID_NO_INPUT_POINTER;
578        } else {
579          return this.getItemPointer() == this.items.size();
580        }
581      }

583      /**
584       * Returns {@code true} if this {@link Thread} {@linkplain
585       * #isViable() is viable}, and has more items to read.
586       *
587       * <p>This method may be invoked in all states.</p>
588       *
589       * @return {@code true} if this {@link Thread} can read an item
590       */
591      public final boolean canRead() {
592        return this.isViable() && this.isValidItemPointer(this.getItemPointer());
593      }
594    
595      /**
596       * Reads (but does not "consume") the current item.
597       *
598       * <p>This method may be invoked only when {@link #canRead()}
599       * returns {@code true}.</p>
600       *
601       * @return the current item which may be {@code null}
602       *
603       * @exception IllegalStateException if the {@link #canRead()} method
604       * returns {@code false}
605       *
606       * @see #canRead()
607       */
608      public final T read() {
609        if (!this.canRead()) {
610          throw new IllegalStateException("Thread cannot read");
611        }
612        if (this.items == null || this.items.isEmpty()) {
613          // This is an edge case.  If we are originally supplied with no
614          // input, we need to be able to use an itemPointer that is both
615          // valid, but doesn't point into the (non-existent) input list.
616          assert this.getItemPointer() == VALID_NO_INPUT_POINTER;
617          return null;
618        }
619        assert this.getItemPointer() >= 0 && this.getItemPointer() < this.items.size();
620        return this.items.get(this.getItemPointer());
621      }
622    
623      private final boolean isValidItemPointer(final int itemPointer) {
624        if (this.items == null || this.items.isEmpty()) {
625          // We can have null or empty input.  In such a case, a value
626          // equal to VALID_NO_INPUT_POINTER is OK; all others are not.
627          return itemPointer == VALID_NO_INPUT_POINTER;
628        }
629        return itemPointer >= 0 && itemPointer < this.items.size();
630      }

632      /**
633       * Advances this {@link Thread}'s item pointer, provided it was
634       * valid to begin with.  Returns {@code true} if the resulting item
635       * pointer is valid.
636       *
637       * <p>Please note that a return value of {@code false} does
638       * <em>not</em> indicate that this {@link Thread} is dead.</p>
639       *
640       * @return {@code true} if this {@link Thread}'s item pointer was
641       * successfully advanced; {@code false} otherwise
642       *
643       * @exception IllegalStateException if this {@link Thread}
644       * {@linkplain #isViable() is not viable}
645       */
646      public final boolean advanceItemPointer() {
647        this.ensureViable();
648        final boolean returnValue;
649        int itemPointer = this.getItemPointer();
650        if (itemPointer == INVALID_INPUT_POINTER || itemPointer == VALID_NO_INPUT_POINTER) {
651          returnValue = false;
652        } else {
653          ++this.itemPointer;
654          returnValue = this.isValidItemPointer(this.getItemPointer());
655        }
656        return returnValue;
657      }

674      /**
675       * Advances the program counter to point at the next {@link
676       * Instruction} in the {@link Program}.  If this cannot
677       * happen&mdash;as in the case where there are no more {@link
678       * Instruction}s in the {@link Program}&mdash;then the {@link
679       * #die()} method is invoked.
680       *
681       * <p>This method may be invoked only when {@link #isViable()}
682       * returns {@code true}.</p>
683       *
684       * @return {@code true} if the program counter was successfully
685       * advanced and therefore if there is another {@link Instruction}
686       * ready to be executed; {@code false} if this {@link Thread} has
687       * been killed via the {@link #die()} method since there are no more
688       * {@link Instruction}s in the {@link Program} to be executed
689       *
690       * @exception IllegalStateException if this {@link Thread}
691       * {@linkplain #isViable() is not viable}
692       */
693      public final boolean advanceProgramCounter() {
694        this.ensureViable();
695        boolean returnValue = false;
696        final ProgramCounter<T> programCounter = this.getProgramCounter();
697        if (programCounter != null) {
698          returnValue = programCounter.advance();
699        }
700        if (!returnValue) {
701          // The program counter could not be incremented, so there are no
702          // further instructions to run, so we're dead.
703          this.die();
704        }
705        return returnValue;
706      }

708      /**
709       * Sets this {@link Thread}'s program counter to the supplied value,
710       * provided it is valid.  If the supplied value is <em>not</em>
711       * valid, then this {@link Thread} {@linkplain #die() dies}.
712       *
713       * <p>This method may be invoked only when {@link #isViable()}
714       * returns {@code true}.</p>
715       *
716       * @param programCounter
717       *
718       * @param relative
719       *
720       * @return {@code true} if the program counter was actually set;
721       * {@code false} if it was not (thus indicating that this {@link
722       * Thread} is now dead)
723       *
724       * @exception IllegalArgumentException if the supplied {@code
725       * programCounter} is not valid
726       *
727       * @exception IllegalStateException if this {@link Thread}
728       * {@linkplain #isViable() is not viable}
729       */
730      public final boolean jump(final int programCounter, final boolean relative) {
731        this.ensureViable();
732        final boolean returnValue;
733        final ProgramCounter<T> pc = this.getProgramCounter();
734        final int absoluteProgramCounter;
735        if (relative) {
736          absoluteProgramCounter = pc.getIndex() + programCounter;
737        } else {
738          absoluteProgramCounter = programCounter;
739        }
740        if (pc != null && pc.isValid(absoluteProgramCounter)) {
741          pc.setIndex(absoluteProgramCounter);
742          returnValue = true;
743        } else {
744          this.die();
745          returnValue = false;
746        }
747        return returnValue;
748      }

751       * Returns the current {@link Instruction}.
752       *
753       * <p>This method may be invoked only when {@link #isViable()}
754       * returns {@code true}.</p>
755       *
756       * @return the current {@link Instruction}, or {@code null} if the
757       * {@link #isViable()} method returns {@code false}
758       *
759       * @exception IllegalStateException if this {@link Thread}
760       * {@linkplain #isViable() is not viable}
761       *
762       * @see #isViable()
763       */
764      private final Instruction<T> getInstruction() {
765        this.ensureViable();
766        final Instruction<T> returnValue;
767        final ProgramCounter<T> programCounter = this.getProgramCounter();
768        if (programCounter != null && this.isViable()) {
769          returnValue = programCounter.getInstruction();
770        } else {
771          returnValue = null;
772        }
773        return returnValue;
774      }

776      /**
777       * Retrieves the current {@link Instruction} and {@linkplain
778       * Instruction#execute(InstructionContext) executes} it.
779       *
780       * <p>If the {@link #isViable()} method returns {@code false},
781       * then invoking this method will cause the {@link #die()} method to
782       * be invoked and no further action will take place.</p>
783       *
784       * <p>This method may be invoked only when {@link #isViable()}
785       * returns {@code true}.</p>
786       *
787       * @exception IllegalStateException if this {@link Thread}
788       * {@linkplain #isViable() is not viable}
789       */
790      public final void step() {
791        this.ensureViable();
792        final Instruction<T> instruction = this.getInstruction();
793        assert instruction != null;
794        if (this.ic == null) {
795          this.ic = new InstructionContext<T>(this);
796        }
797        instruction.execute(this.ic);
798      }

800      /**
801       * Repeatedly calls the {@link #step()} method until this {@link
802       * Thread} has completed execution.
803       *
804       * <p>This method may be invoked only when {@link #isViable()}
805       * returns {@code true}.</p>
806       *
807       * @exception IllegalStateException if this {@link Thread}
808       * {@linkplain #isViable() is not viable}
809       */
810      @Override
811      public final void run() {
812        this.ensureViable();
813        final Logger logger = Logger.getLogger(this.getClass().getName());
814        assert logger != null;
815        final boolean finer = logger.isLoggable(Level.FINER);
816        final ProgramCounter<T> pc = this.getProgramCounter();
817        assert pc != null;
818        while (this.isViable()) {
819          final Instruction<T> currentInstruction = pc.getInstruction();
820          assert currentInstruction != null;
821          if (finer) {
822            final int index = pc.getIndex();
823            logger.logp(Level.FINER, this.getClass().getName(), "run", "Before running {0} {1} ({2}) at input position {3}", new Object[] { this, currentInstruction, Integer.valueOf(index), Integer.valueOf(this.getItemPointer()) });
824          }
825          this.step();
826          if (finer) {
827            if (this.isViable()) {
828              logger.logp(Level.FINER, this.getClass().getName(), "run", "After running {0} {1}; input position is now {2}", new Object[] { this, currentInstruction, Integer.valueOf(this.getItemPointer()) });
829            } else {
830              logger.logp(Level.FINER, this.getClass().getName(), "run", "Thread {0} is no longer viable", this);
831            }
832          }
833        }
834      }

837       * Returns {@code true} if and only if this {@link Thread}'s
838       * {@linkplain #getState() associated state} is {@link State#VIABLE
839       * VIABLE} and if its associated {@link ProgramCounter}'s {@link
840       * ProgramCounter#isValid()} method returns {@code true}.
841       *
842       * <p>This method may be invoked in all states.</p>
843       *
844       * @return {@code true} if and only if this {@link Thread} is viable
845       */
846      public final boolean isViable() {
847        // Do not call getProgramCounter() here; just access the field
848        // directly.  This avoids an infinite loop.
849        return this.getState() == State.VIABLE && this.programCounter != null && this.programCounter.isValid();
850      }

868       * Causes this {@link Thread} to irrevocably enter the {@link
869       * State#DEAD DEAD} state.
870       *
871       * <p>This method may be invoked in all states except {@link
872       * State#MATCH MATCH}.</p>
873       *
874       * @exception IllegalStateException if this {@link Thread} is
875       * currently in the {@link State#MATCH MATCH} state
876       */
877      public final void die() {
878        if (State.MATCH == this.getState()) {
879          throw new IllegalStateException();
880        }
881        this.state = State.DEAD;
882        this.itemPointer = INVALID_INPUT_POINTER;
883        if (this.captureGroups != null) {
884          this.captureGroups.clear();
885        }

942      @Override
943      public Thread<T> clone() {
944        this.ensureViable();
945        Thread<T> clone = null;
946        try {
947          @SuppressWarnings("unchecked")
948          final Thread<T> temp = (Thread<T>)super.clone();
949          clone = temp;
950        } catch (final CloneNotSupportedException cannotHappen) {
951          throw (InternalError)new InternalError().initCause(cannotHappen);
952        }
953        assert clone != null;
954        
955        // Clone our ProgramCounter.
956        final ProgramCounter<T> programCounter = this.getProgramCounter();
957        if (programCounter != null) {
958          clone.programCounter = programCounter.clone();
959        }
960    
961        // TODO: Clone our variables?
962    
963        // Clone our capture groups.
964        if (this.captureGroups != null) {
965          clone.captureGroups = deepClone(this.captureGroups);
966        }
967    
968        // With all of his state set, make sure that he's VIABLE.
969        assert clone.isViable();
970    
971        return clone;
972      }

978        final Object id = this.getId();

980        result = result * 37 + c;

983        result = result * 37 + c;

997        result = result * 37 + c;

1002      @Override
1003      public boolean equals(final Object other) {
1004        if (other == this) {
1005          return true;
1006        } else if (this.getClass().equals(other.getClass())) {
1007          final Thread him = (Thread)other;
1008    
1009          final int myItemPointer = this.getItemPointer();
1010          if (him.getItemPointer() != myItemPointer) {
1011            return false;
1012          }
1013    
1014          final Object id = this.getId();
1015          if (id == null) {
1016            if (him.getId() != null) {
1017              return false;
1018            }
1019          }
1020    
1021          final State state = this.getState();
1022          assert state != null;
1023          if (state != him.getState()) {
1024            return false;
1025          }
1026    
1027          if (this.items == null) {
1028            if (him.items != null) {
1029              return false;
1030            }
1031          } else if (!this.items.equals(him.items)) {
1032            return false;
1033          }
1034    
1035          // We don't call getProgramCounter() because hashCode() must be
1036          // a method that is permissible to call in any state.
1037          if (this.programCounter == null) {
1038            if (him.programCounter != null) {
1039              return false;
1040            }
1041          } else if (!this.programCounter.equals(him.programCounter)) {
1042            return false;
1043          }
1044    
1045          if (this.captureGroups == null) {
1046            if (him.captureGroups != null) {
1047              return false;
1048            }
1049          } else if (!this.captureGroups.equals(him.captureGroups)) {
1050            return false;
1051          }
1052    
1053          return true;
1054    
1055        } else {
1056          return false;
1057        }
1058      }

1076      private static final <T> Map<Object, CaptureGroup<T>> deepClone(final Map<Object, CaptureGroup<T>> suppliedCaptureGroups) {
1077        final Map<Object, CaptureGroup<T>> captureGroups;
1078        if (suppliedCaptureGroups != null) {
1079          captureGroups = new HashMap<Object, CaptureGroup<T>>(suppliedCaptureGroups.size());
1080          final Set<Entry<Object, CaptureGroup<T>>> entries = suppliedCaptureGroups.entrySet();
1081          if (entries != null && !entries.isEmpty()) {
1082            for (final Entry<Object, CaptureGroup<T>> entry : entries) {
1083              if (entry != null) {
1084                final CaptureGroup<T> cg = entry.getValue();
1085                if (cg == null) {
1086                  captureGroups.put(entry.getKey(), null);
1087                } else {
1088                  captureGroups.put(entry.getKey(), cg.clone());
1089                }
1090              }
1091            }
1092          }
1093        } else {
1094          captureGroups = null;
1095        }
1096        return captureGroups;
1097      }
