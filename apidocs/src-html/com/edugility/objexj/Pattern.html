028    package com.edugility.objexj;
029    
030    import java.io.IOException;
031    
032    import java.util.List;
033    
034    import com.edugility.objexj.engine.Engine;
035    import com.edugility.objexj.engine.Program;
036    
037    import com.edugility.objexj.parser.Parser;
038    
039    /**
040     * A regular expression pattern for arbitrary {@link Object}s.
041     *
042     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
043     */
044    public class Pattern<T> {
045    
046      /**
047       * The {@link Engine} to use to {@linkplain Engine#run(Program,
048       * List) run <tt>Program</tt>s}.  This field is never {@code null}.
049       */
050      private final Engine<T> engine;
051    
052      /**
053       * The {@link Program} this {@link Pattern} will {@linkplain
054       * Engine#run(Program, List) run}.  This field is never {@code
055       * null}.
056       */
057      private final Program<T> program;
058    
059      /**
060       * Creates a new {@link Pattern} with the supplied {@link Program}.
061       * A new {@link Engine} will be used to {@linkplain
062       * Engine#run(Program, List) run} the supplied {@link Program}.
063       *
064       * @param program the {@link Program} to {@linkplain
065       * Engine#run(Program, List) run}; must not be {@code null}
066       *
067       * @exception IllegalArgumentException if {@code program} is {@code
068       * null}
069       *
070       * @see #Pattern(Engine, Program)
071       *
072       * @see Pattern#compile(String)
073       */
074      private Pattern(final Program<T> program) {
075        this(null, program);
076      }
077    
078      /**
079       * Creates a new {@link Pattern} with the supplied {@link Engine}
080       * and {@link Program}.
081       *
082       * @param engine the {@link Engine} that will be used to {@linkplain
083       * Engine#run(Program, List) run} the supplied {@link Program}; if
084       * {@code null} a new {@link Engine} will be used instead
085       *
086       * @param program the {@link Program} to {@linkplain
087       * Engine#run(Program, List) run}; must not be {@code null}
088       *
089       * @exception IllegalArgumentException if {@code program} is {@code
090       * null}
091       *
092       * @see Pattern#compile(String)
093       */
094      private Pattern(final Engine<T> engine, final Program<T> program) {
095        super();
096        if (program == null) {
097          throw new IllegalArgumentException("program", new NullPointerException("program"));
098        }
099        this.program = program;
100        if (engine == null) {
101          this.engine = new Engine<T>();
102        } else {
103          this.engine = engine;
104        }
105      }
106    
107      /**
108       * Returns a {@link Matcher} initialized to match the supplied
109       * {@link List} of items.  This method never returns {@code null}.
110       *
111       * @param items the input; may be {@code null}
112       *
113       * @return a new {@link Matcher}; never {@code null}
114       */
115      public final Matcher<T> matcher(final List<T> items) {
116        return new Matcher<T>(this, items);
117      }
118    
119      /**
120       * Returns the {@link Engine} that will be used to {@linkplain
121       * Engine#run(Program, List) run} this {@link Pattern}'s {@linkplain
122       * #getProgram() affiliated <tt>Program</tt>}.  This method never
123       * returns {@code null}.
124       *
125       * @return a non-{@code null} {@link Engine}
126       */
127      final Engine<T> getEngine() {
128        assert this.engine != null;
129        return this.engine;
130      }
131    
132      /**
133       * Returns the {@link Program} that this {@link Pattern} will cause
134       * to be {@linkplain Engine#run(Program, List) run} by {@link
135       * Matcher}s {@linkplain #matcher(List) supplied by its
136       * <tt>matcher(List)</tt> method}.  This method never returns {@code
137       * null}.
138       *
139       * @return a non-{@code null} {@link Program}
140       */
141      final Program<T> getProgram() {
142        assert this.program != null;
143        return this.program;
144      }
145    
146      /**
147       * Returns a non-{@code null} {@link String} representation of this
148       * {@link Pattern}.
149       *
150       * <p>This implementation attempts to return the original source
151       * code used to produce this {@link Pattern}.  If that fails for
152       * some reason, then the normal {@link Object#toString()} method
153       * return value is returned instead.</p>
154       *
155       * @return a non-{@code null} {@link String} representation of this
156       * {@link Pattern}
157       */
158      @Override
159      public String toString() {
160        final Program<T> program = this.getProgram();
161        assert program != null;
162        final String returnValue;
163        final Object source = program.getSource();
164        if (source == null) {
165          returnValue = super.toString();
166        } else {
167          returnValue = source.toString();
168        }
169        return returnValue;
170      }
171    
172    
173      /*
174       * Static methods.
175       */
176    
177    
178      /**
179       * Compiles a new {@link Pattern} from the supplied source code.
180       *
181       * @param source the source code for the {@link Pattern}; must not
182       * be {@code null}
183       *
184       * @return a new, non-{@code null} {@link Pattern}
185       * 
186       * @exception IllegalArgumentException if {@code source} is {@code
187       * null}
188       *
189       * @exception IOException if the source code could not be compiled
190       *
191       * @see <a href="../../../../syntax.html">Syntax Guide</a>
192       */
193      public static final <T> Pattern<T> compile(final String source) throws IOException {
194        if (source == null) {
195          throw new IllegalArgumentException("source", new NullPointerException("source"));
196        }
197        final Program<T> p = new Parser().parse(source);
198        assert p != null;
199        return new Pattern<T>(p);
200      }
201    
202    }
